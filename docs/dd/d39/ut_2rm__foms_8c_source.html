76 /*
77  * Allocate desired FOP and populate test-data in it.
78  */
79 static struct m0_fop *fop_alloc(enum m0_rm_incoming_type fomtype)
80 {
81  struct m0_fop *fop = NULL;
82 
83  switch (fomtype) {
84  case M0_RIT_BORROW:
85  fop = m0_fop_alloc(&m0_rm_fop_borrow_fopt, NULL, machine);
86  M0_UT_ASSERT(fop != NULL);
87  break;
88  case M0_RIT_REVOKE:
89  fop = m0_fop_alloc(&m0_rm_fop_revoke_fopt, NULL, machine);
90  M0_UT_ASSERT(fop != NULL);
91  break;
92  default:
93  M0_IMPOSSIBLE("Invalid RM-FOM type");
94  break;
95  }
96  return fop;
97 }

109 /*
110  * Create and initialise RM FOMs.
111  */
112 static void fom_create(enum m0_rm_incoming_type fomtype,
113  bool err_test,
114  struct m0_fop *fop,
115  struct m0_fom **fom)
116 {
117  int rc = 0;
118  struct m0_fom *base_fom;
119 
120  if (err_test)
121  m0_fi_enable_once("request_fom_create", "fom_alloc_failure");
122 
123  switch (fomtype) {
124  case M0_RIT_BORROW:
125  rc = borrow_fom_create(fop, fom,
126  &rm_ctxs[SERVER_1].rc_rmach_ctx.rmc_reqh);
127  break;
128  case M0_RIT_REVOKE:
129  rc = revoke_fom_create(fop, fom,
130  &rm_ctxs[SERVER_1].rc_rmach_ctx.rmc_reqh);
131  break;
132  default:
133  M0_IMPOSSIBLE("Invalid RM-FOM type");
134  }
135  M0_UT_ASSERT(ergo(err_test, rc == -ENOMEM));
136  M0_UT_ASSERT(ergo(!err_test, rc == 0));
137 
138  if (!err_test) {
139  (*fom)->fo_service = &dummy_svc;
140  if (dummy_svc.rs_fom_key == 0)
141  dummy_svc.rs_fom_key = m0_locality_lockers_allot();
142  dummy_svc.rs_reqh = &rm_ctxs[SERVER_1].rc_rmach_ctx.rmc_reqh;
143  base_fom = *fom;
144  base_fom->fo_fop = fop;
145  base_fom->fo_loc = &dummy_loc;
146  M0_CNT_INC(base_fom->fo_loc->fl_foms);
147  m0_fom_sm_init(base_fom);
148  m0_fom_locality_inc(base_fom);
149  }
150 }

154  m0_ut_fom_phase_set(fom, M0_FOPH_FINISH);

156  m0_fop_put_lock(fom->fo_fop);

164  break;
165  default:
166  M0_IMPOSSIBLE("Invalid RM-FOM type");
167  }
168 }

261 /*
262  * Validate the test results.
263  */
264 static void brw_fom_state_validate(struct m0_fom *fom, int32_t rc,
265  enum test_type test)
266 {
267  struct m0_rm_fop_borrow *brw_fop;
268  struct m0_sm_group *grp = &dummy_loc.fl_group;
269 
270  m0_rm_owner_lock(rm_test_data.rd_owner);
271  switch (test) {
272  case RM_UT_FULL_CREDITS_TEST:
273  M0_UT_ASSERT(m0_fom_phase(fom) == FOPH_RM_REQ_FINISH);
274  M0_UT_ASSERT(rc == M0_FSO_WAIT);
275  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
276  &rm_test_data.rd_owner->ro_sublet));
277  M0_UT_ASSERT(m0_rm_ur_tlist_is_empty(
278  &rm_test_data.rd_owner->ro_owned[OWOS_CACHED]));
279  M0_UT_ASSERT(m0_rm_ur_tlist_is_empty(
280  &rm_test_data.rd_owner->ro_owned[OWOS_HELD]));
281  break;
282  case RM_UT_PARTIAL_CREDITS_TEST:
283  M0_UT_ASSERT(m0_fom_phase(fom) == FOPH_RM_REQ_FINISH);
284  M0_UT_ASSERT(rc == M0_FSO_WAIT);
285  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
286  &rm_test_data.rd_owner->ro_sublet));
287  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
288  &rm_test_data.rd_owner->ro_owned[OWOS_CACHED]));
289  M0_UT_ASSERT(m0_rm_ur_tlist_is_empty(
290  &rm_test_data.rd_owner->ro_owned[OWOS_HELD]));
291  break;
292  case RM_UT_INVALID_CREDITS_TEST:
293  case RM_UT_MEMFAIL_TEST:
294  M0_UT_ASSERT(rc == M0_FSO_WAIT);
295  break;
296  }
297  m0_rm_owner_unlock(rm_test_data.rd_owner);
298  brw_fop = m0_fop_data(fom->fo_fop);
299  M0_UT_ASSERT(brw_fop != NULL);
300  m0_buf_free(&brw_fop->bo_base.rrq_credit.cr_opaque);
301  /* RM code posts AST to wakeup FOM, delete it. AST is already freed */
302  grp->s_forkq = grp->s_forkq->sa_next;
303 }

305 /*
306  * Test function for testing BORROW FOM functions.
307  */
308 static void brw_fom_state_test(enum test_type test)
309 {
310  struct m0_fom *fom;
311  struct m0_fop *fop;
312  int rc;
313 
314  /* Initialise hierarchy of RM objects with rings resource */
315  rings_utdata_ops_set(&rm_test_data);
316  rm_utdata_init(&rm_test_data, OBJ_OWNER);
317 
318  /* Add self-loan to the test owner object */
319  rm_test_owner_capital_raise(rm_test_data.rd_owner,
320  &rm_test_data.rd_credit);
321 
322  fop = fop_alloc(M0_RIT_BORROW);
323 
324  /*
325  * Create FOM and set the FOM phase to start processing of the request.
326  */
327  fom_create(M0_RIT_BORROW, false, fop, &fom);
328  M0_UT_ASSERT(fom != NULL);
329  brw_fop_populate(fom, test);
330 
331  /*
332  * Call the first phase of FOM.
333  */
334  m0_fi_enable_once("request_pre_process", "no-subscription");
335  rc = borrow_fom_tick(fom);
336  M0_UT_ASSERT(m0_fom_phase(fom) == FOPH_RM_REQ_CREDIT_GET);
337  M0_UT_ASSERT(rc == M0_FSO_AGAIN);
338 
339  /*
340  * Call the second phase of FOM.
341  */
342  if (test == RM_UT_MEMFAIL_TEST)
343  m0_fi_enable_once("rings_credit_copy", "fail_copy");
344  rc = borrow_fom_tick(fom);
345  M0_UT_ASSERT(m0_fom_phase(fom) == FOPH_RM_REQ_WAIT);
346  M0_UT_ASSERT(rc == M0_FSO_WAIT);
347 
348 
349  /*
350  * Call the third phase of FOM.
351  */
352  m0_fi_enable_once("request_post_process", "no_rpc_reply_post");
353  rc = borrow_fom_tick(fom);
354  brw_fom_state_validate(fom, rc, test);
355 
356  fom_fini(fom, M0_RIT_BORROW);
357  brw_test_cleanup();
358  rm_utdata_owner_windup_fini(&rm_test_data);
359 }
