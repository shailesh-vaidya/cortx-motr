57  uint32_t op,

59 {
60  static struct m0_fop_type *sns_fop_type[] = {
61  [CM_OP_REPAIR] = &m0_sns_repair_trigger_fopt,
62  [CM_OP_REPAIR_QUIESCE] = &m0_sns_repair_quiesce_fopt,
63  [CM_OP_REPAIR_RESUME] = &m0_sns_repair_trigger_fopt,
64  [CM_OP_REBALANCE] = &m0_sns_rebalance_trigger_fopt,
65  [CM_OP_REBALANCE_QUIESCE]= &m0_sns_rebalance_quiesce_fopt,
66  [CM_OP_REBALANCE_RESUME] = &m0_sns_rebalance_trigger_fopt,
67  [CM_OP_REPAIR_STATUS] = &m0_sns_repair_status_fopt,
68  [CM_OP_REBALANCE_STATUS] = &m0_sns_rebalance_status_fopt,
69  [CM_OP_REPAIR_ABORT] = &m0_sns_repair_abort_fopt,
70  [CM_OP_REBALANCE_ABORT] = &m0_sns_rebalance_abort_fopt
71  };
72  M0_ENTRY();
73  M0_PRE(IS_IN_ARRAY(op, sns_fop_type));
74 
75  *fop = m0_fop_alloc(sns_fop_type[op], NULL, mach);
76  return *fop == NULL ? M0_ERR(-ENOMEM) : M0_RC(0);
77 }
