44 M0_INTERNAL int m0_spiel__ut_reqh_init(struct m0_spiel_ut_reqh *spl_reqh,
45  const char *ep_addr)
46 {
47  enum { NR_TMS = 1 };
48  int rc;
49 
50  M0_SET0(spl_reqh);
51  rc = m0_net_domain_init(&spl_reqh->sur_net_dom,
52  m0_net_xprt_default_get());
53  if (rc != 0)
54  return rc;
55 
56  rc = m0_rpc_net_buffer_pool_setup(&spl_reqh->sur_net_dom,
57  &spl_reqh->sur_buf_pool,
58  m0_rpc_bufs_nr(
59  M0_NET_TM_RECV_QUEUE_DEF_LEN,
60  NR_TMS),
61  NR_TMS);
62  if (rc != 0)
63  goto net;
64 
65  rc = M0_REQH_INIT(&spl_reqh->sur_reqh,
66  .rhia_dtm = (void *)1,
67  .rhia_mdstore = (void *)1,
68  .rhia_fid = &g_process_fid,
69  );
70  if (rc != 0)
71  goto buf_pool;
72 
73  rc = m0_rpc_machine_init(&spl_reqh->sur_rmachine,
74  &spl_reqh->sur_net_dom, ep_addr,
75  &spl_reqh->sur_reqh, &spl_reqh->sur_buf_pool,
76  M0_BUFFER_ANY_COLOUR,
77  M0_RPC_DEF_MAX_RPC_MSG_SIZE,
78  M0_NET_TM_RECV_QUEUE_DEF_LEN);
79  if (rc != 0)
80  goto reqh;
81  m0_reqh_start(&spl_reqh->sur_reqh);
82  rc = m0_spiel__ut_rms_start(&spl_reqh->sur_reqh);
83  if (rc != 0)
84  goto rm;
85  return 0;
86 rm:
87  m0_reqh_services_terminate(&spl_reqh->sur_reqh);
88 reqh:
89  m0_reqh_fini(&spl_reqh->sur_reqh);
90 buf_pool:
91  m0_rpc_net_buffer_pool_cleanup(&spl_reqh->sur_buf_pool);
92 net:
93  m0_net_domain_fini(&spl_reqh->sur_net_dom);
94  return rc;
95 }

106 M0_INTERNAL int m0_spiel__ut_rpc_server_start(struct m0_rpc_server_ctx *rpc_srv,

123  "%d", M0_RPC_DEF_MAX_RPC_MSG_SIZE);
124 
125 #define NAME(ext) "ut_spiel" ext
126  char *argv[] = {
127  NAME(""), "-T", "AD", "-D", NAME(".db"), "-S", NAME(".stob"),
128  "-A", "linuxstob:"NAME("-addb_stob"), "-w", "10",
129  "-e", full_ep, "-H", (char *)ha_ep, "-f", M0_UT_CONF_PROCESS,
130  "-m", max_rpc_size, "-G", full_ep,
131  "-c", (char *)confdb_path
132  };
133 #undef NAME
134 
135  M0_SET0(rpc_srv);
136 
137  rpc_srv->rsx_xprts = m0_net_all_xprt_get();
138  rpc_srv->rsx_xprts_nr = m0_net_xprt_nr();
139  rpc_srv->rsx_argv = argv;
140  /* if !run_io_srv then remove last row of argv array */
141  rpc_srv->rsx_argc = ARRAY_SIZE(argv);
142  snprintf(log_name, LOG_NAME_MAX_LEN, "confd_%s.log", ha_ep);
143  rpc_srv->rsx_log_file_name = log_name;
144 
145  return m0_rpc_server_start(rpc_srv);
146 }
