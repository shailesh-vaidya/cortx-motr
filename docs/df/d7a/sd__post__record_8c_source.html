101 void fdmi_sd_post_record(void)
102 {
103  struct m0_fdmi_src *src = src_alloc();
104  int rc;
105 
106  static char fdmi_data[] = "hello, FDMI";
107 
108  M0_ENTRY();
109  fdmi_serv_start_ut(&filterc_stub_ops);
110  m0_semaphore_init(&g_sem, 0);
111  m0_semaphore_init(&g_sem2, 0);
112  rc = m0_fdmi_source_register(src);
113  M0_UT_ASSERT(rc == 0);
114  g_src_rec = (struct m0_fdmi_src_rec) {
115  .fsr_src = src,
116  .fsr_data = fdmi_data,
117  };
118  M0_FDMI_SOURCE_POST_RECORD(&g_src_rec);
119  /* Wait until record is processed and released */
120  m0_semaphore_down(&g_sem);
121  m0_semaphore_down(&g_sem2);
122  m0_fdmi_source_deregister(src);
123  M0_UT_ASSERT(src->fs_record_post == NULL);
124  m0_fdmi_source_free(src);
125  m0_semaphore_fini(&g_sem);
126  m0_semaphore_fini(&g_sem2);
127  fdmi_serv_stop_ut();
128  M0_LEAVE();
129 }
