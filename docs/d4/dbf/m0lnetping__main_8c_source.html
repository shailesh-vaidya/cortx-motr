140 static int verbose_printk(const char *fmt, ...)
141 {
142  va_list varargs;
143  char *fmtbuf;
144  int rc;
145 
146  va_start(varargs, fmt);
147  fmtbuf = m0_alloc(strlen(fmt) + sizeof KERN_INFO);
148  if (fmtbuf != NULL) {
149  sprintf(fmtbuf, "%s%s", KERN_INFO, fmt);
150  fmt = fmtbuf;
151  }
152  /* call vprintk(KERN_INFO ...) */
153  rc = vprintk(fmt, varargs);
154  va_end(varargs);
155  m0_free(fmtbuf);
156  return rc;
157 }

179 static int __init m0_netst_init_k(void)
180 {
181  int rc;
182  uint64_t buffer_size;
183  M0_THREAD_ENTER;
184 
185  /* parse module options */
186  if (nr_bufs < PING_MIN_BUFS) {
187  printk(KERN_WARNING "Minimum of %d buffers required\n",
188  PING_MIN_BUFS);
189  return -EINVAL;
190  }
191  buffer_size = nlx_ping_parse_uint64(bulk_size);
192  if (buffer_size > PING_MAX_BUFFER_SIZE) {
193  printk(KERN_WARNING "Max supported bulk data size: %d\n",
194  PING_MAX_BUFFER_SIZE);
195  return -EINVAL;
196  }
197  if (nr_clients > PING_MAX_CLIENT_THREADS) {
198  printk(KERN_WARNING "Max of %d client threads supported\n",
199  PING_MAX_CLIENT_THREADS);
200  return -EINVAL;
201  }
202  if (client_only && server_only)
203  client_only = server_only = false;
204  if (server_network == NULL) {
205  printk(KERN_WARNING "Server LNet interface address missing ("
206  "e.g. 10.1.2.3@tcp0, 1.2.3.4@o2ib1)\n");
207  return -EINVAL;
208  }
209  if (!server_only && client_network == NULL) {
210  printk(KERN_WARNING "Client LNet interface address missing ("
211  "e.g. 10.1.2.3@tcp0, 1.2.3.4@o2ib1)\n");
212  return -EINVAL;
213  }
214  if (server_portal < 0)
215  server_portal = PING_SERVER_PORTAL;
216  if (client_portal < 0)
217  client_portal = PING_CLIENT_PORTAL;
218  if (server_tmid < 0)
219  server_tmid = PING_SERVER_TMID;
220  if (client_tmid < 0)
221  client_tmid = PING_CLIENT_DYNAMIC_TMID;
222  if (verbose < 0)
223  verbose = 0;
224 
225  /* set up sys fs entries? */
226 
227  nlx_ping_init();
228 
229  if (!client_only) {
230  /* set up server context */
231  m0_mutex_init(&sctx.pc_mutex);
232  m0_cond_init(&sctx.pc_cond, &sctx.pc_mutex);
233  if (!quiet)
234  sctx.pc_ops = &verbose_ops;
235  else
236  sctx.pc_ops = &quiet_ops;
237 
238  sctx.pc_nr_bufs = nr_bufs;
239  sctx.pc_nr_recv_bufs = nr_recv_bufs;
240  sctx.pc_bulk_size = buffer_size;
241  sctx.pc_msg_timeout = msg_timeout;
242  sctx.pc_bulk_timeout = bulk_timeout;
243  sctx.pc_server_bulk_delay = active_bulk_delay;
244  sctx.pc_network = server_network;
245  sctx.pc_portal = server_portal;
246  sctx.pc_tmid = server_tmid;
247  sctx.pc_dom_debug = server_debug;
248  sctx.pc_tm_debug = server_debug;
249  sctx.pc_sync_events = !async_events;
250  sctx.pc_min_recv_size = server_min_recv_size;
251  sctx.pc_max_recv_msgs = server_max_recv_msgs;
252  sctx.pc_verbose = verbose;
253  nlx_ping_server_spawn(&server_thread, &sctx);
254 
255  printk(KERN_INFO "Motr LNet System Test"
256  " Server Initialized\n");
257  }
258 
259  if (!server_only) {
260  int i;
261  int32_t client_base_tmid = client_tmid;
262  M0_ALLOC_ARR(client_thread, nr_clients);
263  M0_ALLOC_ARR(params, nr_clients);
264 
265  M0_ASSERT(client_thread != NULL);
266  M0_ASSERT(params != NULL);
267 
268  /* start all the client threads */
269  for (i = 0; i < nr_clients; ++i) {
270  if (client_base_tmid != PING_CLIENT_DYNAMIC_TMID)
271  client_tmid = client_base_tmid + i;
272 #define CPARAM_SET(f) params[i].f = f
273  CPARAM_SET(loops);
274  CPARAM_SET(nr_bufs);
275  CPARAM_SET(bulk_timeout);
276  CPARAM_SET(msg_timeout);
277  CPARAM_SET(client_network);
278  CPARAM_SET(client_portal);
279  CPARAM_SET(client_tmid);
280  CPARAM_SET(server_network);
281  CPARAM_SET(server_portal);
282  CPARAM_SET(server_tmid);
283  CPARAM_SET(send_msg_size);
284  CPARAM_SET(verbose);
285 #undef CPARAM_SET
286  params[i].bulk_size = buffer_size;
287  params[i].client_id = i + 1;
288  params[i].client_pid = M0_NET_LNET_PID;
289  params[i].server_pid = M0_NET_LNET_PID;
290  params[i].debug = client_debug;
291  if (!quiet)
292  params[i].ops = &verbose_ops;
293  else
294  params[i].ops = &quiet_ops;
295 
296  rc = M0_THREAD_INIT(&client_thread[i],
297  struct nlx_ping_client_params *,
298  NULL, &nlx_ping_client, &params[i],
299  "client_%d", params[i].client_id);
300  M0_ASSERT(rc == 0);
301  }
302  printk(KERN_INFO "Motr LNet System Test"
303  " %d Client(s) Initialized\n", nr_clients);
304  }
305 
306  return 0;
307 }

309 static void __exit m0_netst_fini_k(void)
310 {
311  M0_THREAD_ENTER;
312  if (!server_only) {
313  int i;
314  for (i = 0; i < nr_clients; ++i) {
315  m0_thread_join(&client_thread[i]);
316  if (!quiet && verbose > 0) {
317  printk(KERN_INFO "Client %d: joined\n",
318  params[i].client_id);
319  }
320  }
321  if (!quiet)
322  nlx_ping_print_qstats_total("Client total",
323  &verbose_ops);
324  m0_free(client_thread);
325  m0_free(params);
326  }
327 
328  if (!client_only) {
329  if (sctx.pc_ops->pqs != NULL)
330  (*sctx.pc_ops->pqs)(&sctx, false);
331 
332  nlx_ping_server_should_stop(&sctx);
333  m0_thread_join(&server_thread);
334  m0_cond_fini(&sctx.pc_cond);
335  m0_mutex_fini(&sctx.pc_mutex);
336  }
337 
338  nlx_ping_fini();
339  printk(KERN_INFO "Motr Kernel Messaging System Test removed\n");
340 }
