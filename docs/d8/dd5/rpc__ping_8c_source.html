129 static int build_endpoint_addr(enum ep_type type,
130  char *out_buf,
131  size_t buf_size)
132 {
133  char *ep_name;
134  char *nid;
135  int tmid;
136 
137  M0_PRE(M0_IN(type, (EP_SERVER, EP_CLIENT)));
138 
139  if (type == EP_SERVER) {
140  nid = server_nid;
141  ep_name = "server";
142  tmid = server_tmid;
143  } else {
144  nid = client_nid;
145  ep_name = "client";
146  tmid = client_tmid;
147  }
148 
149  if (buf_size > M0_NET_LNET_XEP_ADDR_LEN)
150  return -1;
151 
152  snprintf(out_buf, buf_size, "%s:%d:%d:%d", nid, M0_NET_LNET_PID,
153  M0_LNET_PORTAL, tmid);
154  if (verbose)
155  printf("%s endpoint: %s\n", ep_name, out_buf);
156 
157  return 0;
158 }

216 static void ping_reply_received(struct m0_rpc_item *item)
217 {
218  struct m0_fop_ping_rep *reply;
219  struct m0_fop *rfop;
220  int rc;
221 
222  /* typical error checking performed in replied callback */
223  rc = m0_rpc_item_error(item);
224  if (rc == 0) {
225  M0_ASSERT(item->ri_reply != NULL);
226  rfop = container_of(item->ri_reply, struct m0_fop, f_item);
227  reply = m0_fop_data(rfop);
228  M0_ASSERT(reply != NULL);
229  rc = reply->fpr_rc;
230  }
231  if (rc == 0) {
232  /* operation is successful. */;
233  } else {
234  /* operation is failed */;
235  }
236 }

242 /* Create a ping fop and post it to rpc layer */
243 static void send_ping_fop(struct m0_rpc_session *session)
244 {
245  int rc;
246  struct m0_fop *fop;
247  struct m0_fop_ping *ping_fop;
248  uint32_t nr_arr_member;
249  const size_t sz = sizeof ping_fop->fp_arr.f_data[0];
250 
251  nr_arr_member = nr_ping_bytes / sz + !(nr_ping_bytes % sz);
252 
253  fop = m0_fop_alloc_at(session, &m0_fop_ping_fopt);
254  M0_ASSERT(fop != NULL);
255 
256  ping_fop = m0_fop_data(fop);
257  ping_fop->fp_arr.f_count = nr_arr_member;
258 
259  M0_ALLOC_ARR(ping_fop->fp_arr.f_data, nr_arr_member);
260  M0_ASSERT(ping_fop->fp_arr.f_data != NULL);
261 
262  fop->f_item.ri_resend_interval = M0_TIME_ONE_MSEC * 50;
263  rc = m0_rpc_post_sync(fop, session, &ping_item_ops,
264  m0_time_from_now(1, 0));
265  M0_ASSERT(rc == 0);
266  M0_ASSERT(fop->f_item.ri_error == 0);
267  M0_ASSERT(fop->f_item.ri_reply != 0);
268  m0_fop_put_lock(fop);
269 }

271 static void rpcping_thread(struct m0_rpc_session *session)
272 {
273  int i;
274 
275  for (i = 0; i < nr_ping_item; ++i)
276  send_ping_fop(session);

279 static int run_client(void)
280 {
281  int rc;
282  int i;
283  struct m0_thread *client_thread;
284 
285  /*
286  * Declare these variables as static, to avoid on-stack allocation
287  * of big structures. This is important for kernel-space, where stack
288  * size is very small.
289  */
290  static struct m0_net_domain client_net_dom;
291  static struct m0_rpc_client_ctx cctx;
292  static struct m0_fid process_fid = M0_FID_TINIT('r', 0, 1);
293 
294  m0_time_t start;
295  m0_time_t delta;
296 
297  cctx.rcx_net_dom = &client_net_dom;
298  cctx.rcx_local_addr = client_endpoint;
299  cctx.rcx_remote_addr = server_endpoint;
300  cctx.rcx_max_rpcs_in_flight = MAX_RPCS_IN_FLIGHT;
301  cctx.rcx_recv_queue_min_length = tm_recv_queue_len;
302  cctx.rcx_max_rpc_msg_size = max_rpc_msg_size;
303  cctx.rcx_fid = &process_fid;
304 
305  rc = build_endpoint_addr(EP_SERVER, server_endpoint,
306  sizeof server_endpoint);
307  if (rc != 0)
308  return rc;
309 
310  rc = build_endpoint_addr(EP_CLIENT, client_endpoint,
311  sizeof client_endpoint);
312  if (rc != 0)
313  return rc;
314 
315  m0_ping_fop_init();
316 
317  rc = m0_net_domain_init(&client_net_dom, m0_net_xprt_default_get());
318  if (rc != 0)
319  goto fop_fini;
320 
321  rc = m0_rpc_client_start(&cctx);
322  if (rc != 0) {
323  printf("m0rpcping: client init failed \"%i\"\n", -rc);
324  goto net_dom_fini;
325  }
326  M0_ALLOC_ARR(client_thread, nr_client_threads);
327 
328  start = m0_time_now();
329  for (i = 0; i < nr_client_threads; i++) {
330  M0_SET0(&client_thread[i]);
331 
332  rc = M0_THREAD_INIT(&client_thread[i], struct m0_rpc_session *,
333  NULL, &rpcping_thread,
334  &cctx.rcx_session, "client_%d", i);
335  M0_ASSERT(rc == 0);
336  }
337 
338  for (i = 0; i < nr_client_threads; i++) {
339  m0_thread_join(&client_thread[i]);
340  m0_thread_fini(&client_thread[i]);
341  }
342 
343  delta = m0_time_sub(m0_time_now(), start);
344 
345  rc = m0_rpc_client_stop_stats(&cctx, &__print_stats);
346  if (verbose)
347  printf("Time: %lu.%2.2lu sec\n",
348  (unsigned long)m0_time_seconds(delta),
349  (unsigned long)m0_time_nanoseconds(delta) *
350  100 / M0_TIME_ONE_SECOND);
351 net_dom_fini:
352  m0_net_domain_fini(&client_net_dom);
353 fop_fini:
354  m0_ping_fop_fini();
355 
356  return rc;
357 }
