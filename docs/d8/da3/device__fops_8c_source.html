46 M0_INTERNAL int m0_sss_device_fops_init(void)
47 {
48  m0_sm_conf_extend(m0_generic_conf.scf_state, sss_device_fom_phases_desc,
49  m0_generic_conf.scf_nr_states);
50 
51  /*
52  * Extend phases this FOM needs to open separate BE transaction.
53  * Because extend phases execute before M0_FOPH_TXN_INIT phase.
54  * FOM handler catch M0_FOPH_TXN_INIT phase. If it first time then
55  * move to extended phase else simple way.
56  */
57  sss_device_fom_phases_desc[M0_FOPH_TXN_INIT].sd_allowed |=
58  M0_BITS(SSS_DFOM_DISK_OPENING);
59 
60  m0_sss_fop_device_fopt.ft_magix = 0;
61  m0_sss_fop_device_rep_fopt.ft_magix = 0;
62 
63  M0_FOP_TYPE_INIT(&m0_sss_fop_device_fopt,
64  .name = "Device-fop",
65  .opcode = M0_SSS_DEVICE_REQ_OPCODE,
66  .xt = m0_sss_device_fop_xc,
67  .rpc_flags = M0_RPC_ITEM_TYPE_REQUEST,
68  .fom_ops = &sss_device_fom_type_ops,
69  .svc_type = &m0_ss_svc_type,
70  .sm = &sss_device_fom_conf);
71 
72  M0_FOP_TYPE_INIT(&m0_sss_fop_device_rep_fopt,
73  .name = "Device-reply-fop",
74  .opcode = M0_SSS_DEVICE_REP_OPCODE,
75  .xt = m0_sss_device_fop_rep_xc,
76  .rpc_flags = M0_RPC_ITEM_TYPE_REPLY);
77  return 0;
78 }

86 M0_INTERNAL struct m0_fop *m0_sss_device_fop_create(struct m0_rpc_machine *mach,
87  uint32_t cmd,
88  const struct m0_fid *fid)
89 {
90  struct m0_fop *fop;
91  struct m0_sss_device_fop *req;
92 
93  M0_PRE(mach != NULL);
94  M0_PRE(cmd < M0_DEVICE_CMDS_NR);
95  M0_PRE(m0_conf_fid_type(fid) == &M0_CONF_DRIVE_TYPE);
96 
97  fop = m0_fop_alloc(&m0_sss_fop_device_fopt, NULL, mach);
98  if (fop == NULL)
99  return NULL;
100 
101  req = m0_sss_fop_to_dev_req(fop);
102  req->ssd_cmd = cmd;
103  req->ssd_fid = *fid;
104 
105  return fop;
106 }
