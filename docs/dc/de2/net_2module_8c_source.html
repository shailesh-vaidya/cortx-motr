88 static struct m0_module *net_module_create(struct m0 *instance)
89 {
90  struct m0_net_module *net;
91  struct m0_module *m;
92  unsigned i;
93 
94  M0_ALLOC_PTR(net);
95  if (net == NULL)
96  return NULL;
97  m0_module_setup(&net->n_module, m0_net_module_type.mt_name,
98  levels_net, ARRAY_SIZE(levels_net), instance);
99  for (i = 0; i < ARRAY_SIZE(net->n_xprts); ++i) {
100  m = &net->n_xprts[i].nx_module;
101  m0_module_setup(m, net_xprt_mods[i].name, levels_net_xprt,
102  ARRAY_SIZE(levels_net_xprt), instance);
103  m0_module_dep_add(m, M0_LEVEL_NET_DOMAIN,
104  &net->n_module, M0_LEVEL_NET);
105  }
106  instance->i_moddata[M0_MODULE_NET] = net;
107  return &net->n_module;
108 }
