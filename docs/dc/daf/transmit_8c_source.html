329 static int test_fom_tick(struct m0_fom *fom)
330 {
331  int result;
332  int idx = fom->fo_fop - &redo_fop[0];
333  struct m0_dtm_oper_descr *ode = m0_fop_data(fom->fo_fop);
334  struct m0_tl uu;
335  struct m0_dtm_oper *oper = &oper_tgt[idx];
336 
337  M0_UT_ASSERT(IS_IN_ARRAY(idx, oper_tgt));
338  M0_UT_ASSERT(ode->od_updates.ou_nr <= ARRAY_SIZE(update_tgt[idx]));
339 
340  switch (m0_fom_phase(fom)) {
341  case FOM_INIT:
342  m0_semaphore_up(&seq);
343  m0_dtm_update_list_init(&uu);
344  m0_dtm_update_link(&uu, control_tgt[idx], TGT_DELTA);
345  m0_dtm_oper_init(oper, &dtm_tgt, &uu);
346  m0_dtm_update_list_fini(&uu);
347  m0_dtm_update_list_init(&uu);
348  m0_dtm_update_link(&uu, update_tgt[idx], ode->od_updates.ou_nr);
349 
350  result = m0_dtm_oper_build(oper, &uu, ode);
351  M0_UT_ASSERT(result == 0);
352  oper->oprt_op.op_ops = &test_op_ops;
353  m0_dtm_update_list_fini(&uu);
354  m0_dtm_oper_close(oper);
355  m0_fom_phase_set(fom, FOM_READY);
356  break;
357  case FOM_READY:
358  m0_mutex_lock(&lock);
359  m0_dtm_oper_prepared(oper, NULL);
360  m0_dtm_oper_done(oper, NULL);
361  ++ticked;
362  m0_fom_phase_set(fom, M0_FOM_PHASE_FINISH);
363  m0_mutex_unlock(&lock);
364  break;
365  default:
366  M0_UT_ASSERT(0);
367  }
368  return M0_FSO_WAIT;
369 }
