43 M0_INTERNAL void m0_addb2_global_thread_enter(void)
44 {
45  struct m0_addb2_sys *sys = SYS();
46  struct m0_thread_tls *tls = m0_thread_tls();
47 
48  M0_PRE(tls->tls_addb2_mach == NULL);
49  if (sys != NULL) {
50  tls->tls_addb2_mach = m0_addb2_sys_get(sys);
51  m0_addb2_push(M0_AVI_NODE, M0_ADDB2_OBJ(&m0_node_uuid));
52  M0_ADDB2_PUSH(M0_AVI_PID, m0_pid());
53  m0_addb2_push(M0_AVI_THREAD, M0_ADDB2_OBJ(&tls->tls_self->t_h));
54  m0_addb2_clock_add(&tls->tls_clock, M0_AVI_CLOCK, -1);
55  }
56 }

58 M0_INTERNAL void m0_addb2_global_thread_leave(void)
59 {
60  struct m0_addb2_sys *sys = SYS();
61  struct m0_thread_tls *tls = m0_thread_tls();
62  struct m0_addb2_mach *mach = tls->tls_addb2_mach;
63 
64  if (mach != NULL) {
65  M0_ASSERT(sys != NULL);
66  m0_addb2_pop(M0_AVI_THREAD);
67  m0_addb2_pop(M0_AVI_PID);
68  m0_addb2_pop(M0_AVI_NODE);
69  M0_SET0(&tls->tls_clock);
70  tls->tls_addb2_mach = NULL;
71  m0_addb2_sys_put(sys, mach);
72  }
73 }
