60 double m0_net_test_stats_stddev(const struct m0_net_test_stats *stats)
61 {
62  double mean;
63  double stddev;
64  double N;
65  double sum_sqr;
66 
67  M0_PRE(m0_net_test_stats_invariant(stats));
68 
69  if (stats->nts_count == 0 || stats->nts_count == 1)
70  return 0.;
71 
72  mean = m0_net_test_stats_avg(stats);
73  N = stats->nts_count;
74  sum_sqr = double_get(&stats->nts_sum_sqr);
75  stddev = (sum_sqr - N * mean * mean) / (N - 1.);
76  stddev = stddev < 0. ? 0. : stddev;
77  stddev = sqrt(stddev);
78  return stddev;
79 }
