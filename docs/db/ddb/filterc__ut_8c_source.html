97 static int ut_filterc_start()
98 {
99  struct m0_reqh *reqh = motr2reqh(&sctx.rsx_motr_ctx);
100  struct m0_reqh_service_type *stype;
101  bool start_service = false;
102  int rc = 0;
103 
104  stype = m0_reqh_service_type_find("M0_CST_FDMI");
105  if (stype == NULL) {
106  M0_LOG(M0_ERROR, "FDMI service type is not found.");
107  return M0_ERR_INFO(-EINVAL, "Unknown reqh service type: fdmi");
108  }
109 
110  ufc_fdmi_service = m0_reqh_service_find(stype, reqh);
111  if (ufc_fdmi_service == NULL) {
112  rc = m0_reqh_service_allocate(&ufc_fdmi_service, &m0_fdmi_service_type, NULL);
113  M0_UT_ASSERT(rc == 0);
114  m0_reqh_service_init(ufc_fdmi_service, reqh, NULL);
115  start_service = true;
116  }
117 
118  /* Patch filterc instance used by source dock FOM */
119  if (ufc_fco != NULL) {
120  struct m0_reqh_fdmi_svc_params *fdms_start_params;
121  M0_ALLOC_PTR(fdms_start_params);
122  M0_ASSERT(fdms_start_params != NULL);
123  fdms_start_params->filterc_ops = ufc_fco;
124 
125  m0_buf_init(&ufc_fdmi_service->rs_ss_param,
126  fdms_start_params,
127  sizeof(*fdms_start_params));
128  }
129 
130  if (start_service) {
131  rc = m0_reqh_service_start(ufc_fdmi_service);
132  M0_UT_ASSERT(rc == 0);
133  }
134  return M0_RC(rc);
135 }

254 static void filterc_connect_to_confd(void)
255 {
256  struct m0_filterc_ops fc_ops = filterc_def_ops;
257  struct m0_fdmi_src *src = src_alloc();
258  int rc;
259  M0_ENTRY();
260 
261  m0_mutex_init(&cond_mutex);
262  m0_cond_init(&match_cond, &cond_mutex);
263 
264  ufc_fco = &fc_ops;
265  fc_ops.fco_start = ut_filterc_fco_start;
266 
267  rpc_client_and_server_start();
268 
269  rc = ut_filterc_start();
270  M0_UT_ASSERT(rc == 0);
271 
272  rc = m0_fdmi_source_register(src);
273  M0_UT_ASSERT(rc == 0);
274  g_src_rec = (struct m0_fdmi_src_rec) {
275  /* Don't send to remote ep. */
276  .fsr_dryrun = true,
277  .fsr_matched = false,
278  .fsr_src = src,
279  .fsr_data = g_fdmi_data,
280  };
281 
282  M0_FDMI_SOURCE_POST_RECORD(&g_src_rec);
283 
284  m0_mutex_lock(&cond_mutex);
285  m0_cond_wait(&match_cond);
286  m0_mutex_unlock(&cond_mutex);
287  M0_UT_ASSERT(g_src_rec.fsr_matched);
288 
289  m0_fdmi_source_deregister(src);
290  m0_fdmi_source_free(src);
291 
292  rpc_client_and_server_stop();
293  m0_cond_fini(&match_cond);
294  m0_mutex_fini(&cond_mutex);
295 
296  M0_LEAVE();
297 }
