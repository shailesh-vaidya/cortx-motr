156 M0_INTERNAL struct m0_fop *m0_ss_process_fop_create(struct m0_rpc_machine *mach,
157  uint32_t cmd,
158  const struct m0_fid *fid)
159 {
160  struct m0_fop *fop;
161  struct m0_ss_process_req *req;

163  M0_PRE(mach != NULL);
164  M0_PRE(cmd < M0_PROCESS_NR);
165 
166  fop = m0_fop_alloc(&m0_fop_process_fopt, NULL, mach);
167  if (fop == NULL)
168  return NULL;
169 
170  req = m0_ss_fop_process_req(fop);
171  req->ssp_cmd = cmd;
172  req->ssp_id = *fid;
173 
174  return fop;
175 }
