69 static void type_h(const struct ff2c_type *t, int depth)
70 {
71  const struct ff2c_field *f = t->t_field.l_head;
72 
73  indent(depth);
74  out("%s {\n", t->t_c_name);
75  if (t->t_union) {
76  field_h(f, depth + 1);
77  f = f->f_next;
78  indent(depth + 1);
79  out("union {\n");
80  depth++;
81  }
82  for (; f != NULL; f = f->f_next)
83  field_h(f, depth + 1);
84  if (t->t_union) {
85  indent(depth);
86  out("} u;\n");
87  depth--;
88  }
89  indent(depth);
90  out("}");
91 }
