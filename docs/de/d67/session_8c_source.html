155 M0_INTERNAL bool m0_rpc_session_invariant(const struct m0_rpc_session *session)
156 {
157  bool ok;
158 
159  ok = _0C(session != NULL) &&
160  _0C(session->s_conn != NULL) &&
161  _0C(ergo(rpc_session_tlink_is_in(session),
162  rpc_session_tlist_contains(&session->s_conn->c_sessions,
163  session))) &&
164  _0C(ergo(session->s_session_id != SESSION_ID_0,
165  session->s_conn->c_nr_sessions > 0));
166 
167  if (!ok)
168  return false;
169 
170  switch (session_state(session)) {
171  case M0_RPC_SESSION_INITIALISED:
172  case M0_RPC_SESSION_ESTABLISHING:
173  return _0C(session->s_session_id == SESSION_ID_INVALID) &&
174  _0C(m0_rpc_session_is_idle(session));
175 
176  case M0_RPC_SESSION_TERMINATED:
177  return _0C(session->s_session_id <= SESSION_ID_MAX) &&
178  _0C(m0_rpc_session_is_idle(session));
179 
180  case M0_RPC_SESSION_IDLE:
181  case M0_RPC_SESSION_TERMINATING:
182  return _0C(m0_rpc_session_is_idle(session)) &&
183  _0C(session->s_session_id <= SESSION_ID_MAX);
184 
185  case M0_RPC_SESSION_BUSY:
186  return _0C(!m0_rpc_session_is_idle(session)) &&
187  _0C(session->s_session_id <= SESSION_ID_MAX);
188 
189  case M0_RPC_SESSION_FAILED:
190  return _0C(session->s_sm.sm_rc != 0);
191 
192  default:
193  return _0C(false);
194  }
195  /* Should never reach here */
196  M0_ASSERT(0);
197 }

223 M0_INTERNAL int m0_rpc_session_init_locked(struct m0_rpc_session *session,
224  struct m0_rpc_conn *conn)
225 {
226  int rc;
227 
228  M0_ENTRY("session: %p, conn: %p", session, conn);
229  M0_PRE(session != NULL && conn != NULL);
230  M0_PRE(m0_rpc_machine_is_locked(conn->c_rpc_machine));
231 
232  M0_SET0(session);
233 
234  session->s_session_id = SESSION_ID_INVALID;
235  session->s_conn = conn;
236  session->s_xid = 0;
237  session->s_cancelled = false;
238

240  m0_sm_init(&session->s_sm, &session_conf,
241  M0_RPC_SESSION_INITIALISED,
242  &conn->c_rpc_machine->rm_sm_grp);
243  m0_rpc_conn_add_session(conn, session);
244  M0_ASSERT(m0_rpc_session_invariant(session));
245  rc = m0_rpc_item_cache_init(&session->s_reply_cache,
246  &conn->c_rpc_machine->rm_sm_grp.s_lock) ?:
247  m0_rpc_item_cache_init(&session->s_req_cache,
248  &conn->c_rpc_machine->rm_sm_grp.s_lock);
249  if (rc == 0)
250  M0_LOG(M0_INFO, "Session %p INITIALISED", session);
251  else
252  M0_LOG(M0_ERROR, "Session %p initialisation failed: %d",
253  session, rc);
254  m0_rpc_item_pending_cache_init(session);
255  m0_rpc_item_xid_list_init(session);
256 
257  return M0_RC(rc);
258 }

260 M0_INTERNAL void m0_rpc_session_reset(struct m0_rpc_session *session)
261 {
262  struct m0_rpc_machine *machine = session_machine(session);
263 
264  m0_rpc_machine_lock(machine);
265  if (session_state(session) == M0_RPC_SESSION_INITIALISED) {
266  m0_rpc_machine_unlock(machine);
267  return;
268  }
269  session_state_set(session, M0_RPC_SESSION_INITIALISED);
270  session->s_xid = 0;
271  session->s_cancelled = false;
272  session->s_session_id = SESSION_ID_INVALID;
273  M0_POST(m0_rpc_session_invariant(session));
274  m0_rpc_machine_unlock(machine);
275 }

311 {

313  M0_ASSERT(m0_rpc_session_invariant(session));

317

319  m0_rpc_item_cache_fini(&session->s_req_cache);
320  m0_rpc_item_pending_cache_fini(session);

323  m0_rpc_conn_remove_session(session);
324  __session_fini(session);
325  session->s_session_id = SESSION_ID_INVALID;
326  session_state_set(session, M0_RPC_SESSION_FINALISED);
327  m0_sm_fini(&session->s_sm);
328  M0_LOG(M0_INFO, "Session %p FINALISED \n", session);
329  M0_LEAVE();

390 M0_INTERNAL int m0_rpc_session_establish(struct m0_rpc_session *session,
391  m0_time_t abs_timeout)
392 {
393  struct m0_rpc_conn *conn;
394  struct m0_fop *fop;
395  struct m0_rpc_fop_session_establish *args;
396  struct fop_session_establish_ctx *ctx;
397  struct m0_rpc_session *session_0;
398  struct m0_rpc_machine *machine;
399  int rc;
400 
401  M0_ENTRY("session: %p", session);
402  M0_PRE(session != NULL);
403 
404  if (M0_FI_ENABLED("fake_error"))
405  return M0_RC(-EINVAL);
406 
407  machine = session_machine(session);
408 
409  M0_ALLOC_PTR(ctx);
410  if (ctx == NULL) {
411  rc = M0_ERR(-ENOMEM);
412  } else {
413  ctx->sec_session = session;
414  m0_fop_init(&ctx->sec_fop,
415  &m0_rpc_fop_session_establish_fopt, NULL,
416  session_establish_fop_release);
417  rc = m0_fop_data_alloc(&ctx->sec_fop);
418  if (rc != 0) {
419  m0_fop_put_lock(&ctx->sec_fop);
420  }
421  }
422 
423  m0_rpc_machine_lock(machine);
424 
425  M0_ASSERT(m0_rpc_session_invariant(session) &&
426  session_state(session) == M0_RPC_SESSION_INITIALISED);
427 
428  if (rc != 0) {
429  session_failed(session, rc);
430  m0_rpc_machine_unlock(machine);
431  return M0_RC(rc);
432  }
433 
434  conn = session->s_conn;
435 
436  M0_ASSERT(conn_state(conn) == M0_RPC_CONN_ACTIVE);
437 
438  fop = &ctx->sec_fop;
439  args = m0_fop_data(fop);
440  M0_ASSERT(args != NULL);
441 
442  args->rse_sender_id = conn->c_sender_id;
443 
444  session_0 = m0_rpc_conn_session0(conn);
445  rc = m0_rpc__fop_post(fop, session_0, &session_establish_item_ops,
446  abs_timeout);
447  if (rc == 0) {
448  session_state_set(session, M0_RPC_SESSION_ESTABLISHING);
449  } else {
450  session_failed(session, rc);
451  }
452  m0_fop_put(fop);
453 
454  M0_POST(ergo(rc != 0, session_state(session) == M0_RPC_SESSION_FAILED));
455  M0_POST(m0_rpc_session_invariant(session));
456 
457  m0_rpc_machine_unlock(machine);
458 
459  /* see m0_rpc_session_establish_reply_received() */
460  return M0_RC(rc);
461 }
462 M0_EXPORTED(m0_rpc_session_establish);

487 M0_INTERNAL void m0_rpc_session_establish_reply_received(struct m0_rpc_item
488  *item)
489 {
490  struct m0_rpc_fop_session_establish_rep *reply = NULL;
491  struct fop_session_establish_ctx *ctx;
492  struct m0_rpc_machine *machine;
493  struct m0_rpc_session *session;
494  struct m0_rpc_item *reply_item;
495  struct m0_fop *fop;
496  uint64_t session_id;
497  int32_t rc;
498 
499  M0_ENTRY("item: %p", item);
500  M0_PRE(item != NULL &&
501  item->ri_session != NULL &&
502  item->ri_session->s_session_id == SESSION_ID_0);
503 
504  fop = m0_rpc_item_to_fop(item);
505  ctx = container_of(fop, struct fop_session_establish_ctx, sec_fop);
506  session = ctx->sec_session;
507  M0_ASSERT(session != NULL);
508 
509  machine = session_machine(session);
510  M0_ASSERT(m0_rpc_machine_is_locked(machine));
511 
512  M0_ASSERT(m0_rpc_session_invariant(session));
513  M0_ASSERT_INFO(session_state(session) == M0_RPC_SESSION_ESTABLISHING,
514  "Invalid session state: expected %s, got %s",
515  m0_rpc_session_state_to_str(M0_RPC_SESSION_ESTABLISHING),
516  m0_rpc_session_state_to_str(session_state(session)));
517 
518  rc = m0_rpc_item_error(item);
519  if (rc == 0) {
520  reply_item = item->ri_reply;
521  M0_ASSERT(reply_item != NULL &&
522  item->ri_session == reply_item->ri_session);
523  reply = m0_fop_data(m0_rpc_item_to_fop(reply_item));
524  rc = reply->rser_rc;
525  }
526  if (rc == 0) {
527  M0_ASSERT(reply != NULL);
528  session_id = reply->rser_session_id;
529  if (session_id > SESSION_ID_MIN &&
530  session_id < SESSION_ID_MAX &&
531  reply->rser_sender_id != SENDER_ID_INVALID) {
532  session->s_session_id = session_id;
533  session_state_set(session, M0_RPC_SESSION_IDLE);
534  } else {
535  rc = M0_ERR(-EPROTO);
536  }
537  }
538  if (rc != 0)
539  session_failed(session, rc);
540 
541  M0_POST(m0_rpc_session_invariant(session));
542  M0_POST(M0_IN(session_state(session), (M0_RPC_SESSION_IDLE,
543  M0_RPC_SESSION_FAILED)));
544  M0_POST(m0_rpc_machine_is_locked(machine));
545  M0_LEAVE();
546 }

593  m0_time_t abs_timeout)
594 {

612  M0_TIME_NEVER);

622 M0_INTERNAL int m0_rpc_session_terminate(struct m0_rpc_session *session,

624 {
625  struct m0_fop *fop;
626  struct m0_rpc_fop_session_terminate *args;

695  * [^1]
696  * There are two choices here:
697  *
698  * 1. leave session in TERMNATING state FOREVER.
699  * Then when to fini/cleanup session.
700  * This will not allow finialising of session, in turn conn,
701  * and rpc_machine can't be finalised.

703  * 2. Move session to FAILED state.
704  * For this session the receiver side state will still
705  * continue to exist. And receiver can send one-way
706  * items, that will be received on sender i.e. current node.
707  * Current code will drop such items. When/how to fini and
708  * cleanup receiver side state? XXX
709  *
710  * For now, later is chosen. This can be changed in future
711  * to alternative 1, iff required.
712  */
