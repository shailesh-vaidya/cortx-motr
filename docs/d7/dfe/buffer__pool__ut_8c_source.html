48 static void test_init(void)
49 {
50  int rc;
51  uint32_t seg_nr = 64;
52  m0_bcount_t seg_size = 4096;
53  uint32_t colours = 10;
54  unsigned shift = 12;
55  uint32_t buf_nr = 10;
56 
57  M0_ALLOC_PTR(bp.nbp_ndom);
58  M0_UT_ASSERT(bp.nbp_ndom != NULL);
59  rc = m0_net_domain_init(bp.nbp_ndom, m0_net_xprt_default_get());
60  M0_ASSERT(rc == 0);
61  seg_nr = min32u((uint32_t)m0_net_domain_get_max_buffer_segments(
62  bp.nbp_ndom),
63  seg_nr);
64  bp.nbp_ops = &b_ops;
65  rc = m0_net_buffer_pool_init(&bp, bp.nbp_ndom,
66  M0_NET_BUFFER_POOL_THRESHOLD, seg_nr,
67  seg_size, colours, shift, false);
68  M0_UT_ASSERT(rc == 0);
69  m0_chan_init(&buf_chan, &bp.nbp_mutex);
70  m0_net_buffer_pool_lock(&bp);
71  rc = m0_net_buffer_pool_provision(&bp, buf_nr);
72  m0_net_buffer_pool_unlock(&bp);
73  M0_UT_ASSERT(rc == buf_nr);
74 }

76 static void test_get_put(void)
77 {

135 static void test_get_put_multiple(void)
136 {
137  int i;

140  struct m0_thread *client_thread;
141

144  for (i = 0; i < nr_client_threads; i++) {
145  M0_SET0(&client_thread[i]);
146  rc = M0_THREAD_INIT(&client_thread[i], int,

148  M0_BUFFER_ANY_COLOUR, "client_%d", i);
149  M0_ASSERT(rc == 0);
150  M0_SET0(&client_thread[++i]);
151  /* value of integer 'i' is used to put or get the
152  buffer in coloured list */
153  rc = M0_THREAD_INIT(&client_thread[i], int,
154  NULL, &buffers_get_put,
155  i, "client_%d", i);
156  M0_ASSERT(rc == 0);
157  }
158  for (i = 0; i < nr_client_threads; i++) {
159  m0_thread_join(&client_thread[i]);
160  }
161  m0_free(client_thread);
162  m0_net_buffer_pool_lock(&bp);
163  M0_UT_ASSERT(m0_net_buffer_pool_invariant(&bp));
164  m0_net_buffer_pool_unlock(&bp);
165 }

178 static void buffers_get_put(int rc)
179 {
180  struct m0_net_buffer *nb;
181  struct m0_clink buf_link;
182 
183  m0_clink_init(&buf_link, NULL);
184  m0_clink_add_lock(&buf_chan, &buf_link);
185  do {
186  m0_net_buffer_pool_lock(&bp);
187  nb = m0_net_buffer_pool_get(&bp, rc);
188  m0_net_buffer_pool_unlock(&bp);
189  if (nb == NULL)
190  m0_chan_wait(&buf_link);
191  } while (nb == NULL);
192  m0_nanosleep(m0_time(0, 100), NULL);
193  m0_net_buffer_pool_lock(&bp);
194  if (nb != NULL)
195  m0_net_buffer_pool_put(&bp, nb, rc);
196  m0_net_buffer_pool_unlock(&bp);
197  m0_clink_del_lock(&buf_link);
198  m0_clink_fini(&buf_link);
199 }
