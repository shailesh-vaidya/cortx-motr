55 static int fom_create(struct m0_fop *fop,
56  struct m0_fom **out,
57  struct m0_reqh *reqh)
58 {
59  struct m0_fom *fom;
60  struct m0_fop *rfop;
61 
62  M0_PRE(fop != NULL);
63  M0_PRE(out != NULL);
64 
65  M0_ALLOC_PTR(fom);
66  if (fom == NULL)
67  return -ENOMEM;
68 
69  rfop = m0_fop_reply_alloc(fop, &m0_fop_generic_reply_fopt);
70  M0_UT_ASSERT(rfop != NULL);
71 
72  m0_fom_init(fom, &fop->f_type->ft_fom_type, &ut_fom_ops,
73  fop, rfop, reqh);
74 
75  *out = fom;
76 
77  return 0;
78 }

151 static void ut_sss_req(const char *name, uint32_t cmd, int expected_rc,
152  int expected_state)
153 {
154  int rc;
155  struct m0_fop *fop;
156  struct m0_fop *rfop;
157  struct m0_rpc_item *item;
158  struct m0_sss_rep *ss_rfop;
159 
160  fop = ut_ssfop_alloc(name, cmd);
161  item = &fop->f_item;
162 
163  rc = m0_rpc_post_sync(fop, &cctx.rcx_session, NULL, 0);
164  M0_UT_ASSERT(rc == 0);
165 
166  rfop = m0_rpc_item_to_fop(item->ri_reply);
167  M0_UT_ASSERT(rfop != NULL);
168 
169  ss_rfop = m0_fop_data(rfop);
170  M0_UT_ASSERT(ss_rfop->ssr_rc == expected_rc);
171  if (expected_state != 0)
172  M0_UT_ASSERT(ss_rfop->ssr_state == expected_state);
173  m0_fop_put_lock(fop);
174 }

176 static void fop_allow_test(void)
177 {
178  int rc;
179 
180  sctx.rsx_argv = ut_server_argv;
181  sctx.rsx_argc = ARRAY_SIZE(ut_server_argv);
182

184  M0_UT_ASSERT(rc == 0);
185 
186  start_rpc_client_and_server();
187  ut_sss_req(ds1_service_type.rst_name, M0_SERVICE_STATUS, 0,
188  M0_RST_STARTED);
189  ut_sss_req(ds1_service_type.rst_name, M0_SERVICE_QUIESCE, 0,
190  M0_RST_STOPPING);
191  ut_sss_req(ds1_service_type.rst_name, M0_SERVICE_STOP, 0,
192  M0_RST_STOPPED);
193 
194  rc = send_fop();
195  M0_UT_ASSERT(rc == -ECONNREFUSED);
196 
197  /* Test reposting of the fop in case of failure. */
198  rc = send_fop();
199  M0_UT_ASSERT(rc == -ECONNREFUSED);
200 
201  ut_sss_req(ds1_service_type.rst_name,
202  M0_SERVICE_INIT, 0, M0_RST_INITIALISED);
203  ut_sss_req(ds1_service_type.rst_name,
204  M0_SERVICE_START, 0, M0_RST_STARTED);
205  rc = send_fop();
206  M0_UT_ASSERT(rc == 0);
207 
208  m0_reqhut_fop_fini();
209  stop_rpc_client_and_server();
210 }
