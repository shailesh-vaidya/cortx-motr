84 static int filterc_apply_flt_get_next(struct m0_filterc_iter *iter,
85  struct m0_conf_fdmi_filter **out)
86 {
87  int rc;
88  static bool first_filter = true;
89  struct m0_fdmi_filter *flt = &g_conf_filter.ff_filter;
90  struct m0_fdmi_flt_node *root;
91  struct m0_buf var = M0_BUF_INITS(g_var_str);
92 
93  if (first_filter) {
94  root = m0_fdmi_flt_op_node_create(
95  M0_FFO_OR,
96  m0_fdmi_flt_bool_node_create(false),
97  m0_fdmi_flt_var_node_create(&var));
98 
99  m0_fdmi_filter_init(flt);
100 
101  m0_fdmi_filter_root_set(flt, root);
102 
103  *out = &g_conf_filter;
104  rc = 1;
105  first_filter = false;
106  } else {
107  *out = NULL;
108  rc = 0;
109  }
110  return rc;
111 }

181 static void fdmi_sd_apply_filter_internal(const struct m0_filterc_ops *ops)
182 {
183  struct m0_fdmi_src *src = src_alloc();
184  int rc;
185 
186  M0_ENTRY();
187 
188  g_var_str = strdup("test");
189  M0_SET0(&g_conf_filter);
190 
191  fdmi_serv_start_ut(ops);
192  m0_semaphore_init(&g_sem, 0);
193  m0_semaphore_init(&g_sem2, 0);
194  rc = m0_fdmi_source_register(src);
195  M0_UT_ASSERT(rc == 0);
196  g_src_rec = (struct m0_fdmi_src_rec) {
197  .fsr_src = src,
198  .fsr_data = g_fdmi_data,
199  };
200  M0_FDMI_SOURCE_POST_RECORD(&g_src_rec);
201  /* Wait until record is processed and released */
202  m0_semaphore_down(&g_sem);
203  m0_semaphore_down(&g_sem2);
204  m0_fdmi_source_deregister(src);
205  m0_fdmi_source_free(src);
206  m0_semaphore_fini(&g_sem);
207  m0_semaphore_fini(&g_sem2);
208  fdmi_serv_stop_ut();
209  M0_LEAVE();
210 }
