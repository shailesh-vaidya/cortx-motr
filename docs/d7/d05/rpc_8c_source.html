63 M0_INTERNAL int m0_rpc_post(struct m0_rpc_item *item)
64 {
65  int rc;
66  struct m0_rpc_machine *machine;
67 
68  M0_ENTRY("%p[%s/%u]", item, item_kind(item),
69  item->ri_type->rit_opcode);
70  M0_PRE(item->ri_session != NULL);
71  M0_PRE(m0_rpc_conn_is_snd(item2conn(item)));
72 
73  machine = session_machine(item->ri_session);
74  M0_ASSERT(m0_rpc_item_size(item) <= machine->rm_min_recv_size);
75 
76  m0_rpc_machine_lock(machine);
77  rc = m0_rpc__post_locked(item);
78  m0_rpc_machine_unlock(machine);
79  return M0_RC(rc);
80 }

83 M0_INTERNAL int m0_rpc__post_locked(struct m0_rpc_item *item)
84 {
85  struct m0_rpc_session *session;
86  int error;
87  M0_PRE(item != NULL && item->ri_type != NULL);
88 
89  M0_ENTRY("%p[%s/%u]",
90  item, item_kind(item), item->ri_type->rit_opcode);
91  M0_PRE(m0_rpc_item_is_request(item));
92 
93  session = item->ri_session;
94  M0_ASSERT_EX(m0_rpc_session_invariant(session));
95  M0_ASSERT(m0_rpc_item_size(item) <=
96  m0_rpc_session_get_max_item_size(session));
97  M0_ASSERT(m0_rpc_machine_is_locked(session_machine(session)));
98  /*
99  * For requests an additional reference is taken, this reference is
100  * released after ->rio_replied() is called.
101  */
102  m0_rpc_item_get(item);
103  item->ri_rmachine = session_machine(session);
104  item->ri_rpc_time = m0_time_now();
105  m0_rpc_item_sm_init(item, M0_RPC_ITEM_OUTGOING);
106 
107  m0_cookie_new(&item->ri_cookid);
108  m0_cookie_init(&item->ri_header.osr_cookie, &item->ri_cookid);
109  item->ri_header.osr_uuid = session->s_conn->c_uuid;
110  item->ri_header.osr_sender_id = session->s_conn->c_sender_id;
111  item->ri_header.osr_session_id = session->s_session_id;
112 
113  if (!M0_IN(session_state(session), (M0_RPC_SESSION_IDLE,
114  M0_RPC_SESSION_BUSY))) {
115  M0_LOG(M0_DEBUG, "%p[%s/%u], fop %p, session %p, "
116  "Session isn't established. Hence, not posting the item",
117  item, item_kind(item), item->ri_type->rit_opcode,
118  m0_rpc_item_to_fop(item), session);
119  error = M0_ERR(-ENOTCONN);
120  } else if (session->s_session_id != SESSION_ID_0 &&
121  m0_rpc_session_is_cancelled(session)) {
122  M0_LOG(M0_DEBUG, "%p[%s/%u], fop %p, session %p, "
123  "Session is cancelled. Hence, not posting the item",
124  item, item_kind(item), item->ri_type->rit_opcode,
125  m0_rpc_item_to_fop(item), session);
126  error = M0_ERR(-ECANCELED);
127  } else {
128  m0_rpc_item_send(item);
129  return M0_RC(item->ri_error);
130  }
131  m0_rpc_item_failed(item, error);
132  return error;
133 }

135 void m0_rpc_reply_post(struct m0_rpc_item *request, struct m0_rpc_item *reply)
136 {
137  struct m0_rpc_machine *machine;
138 
139  M0_ENTRY("req_item: %p, rep_item: %p", request, reply);
140  M0_PRE(request != NULL && reply != NULL);
141  M0_PRE(request->ri_session != NULL);
142  M0_PRE(reply->ri_type != NULL);
143  M0_PRE(m0_rpc_item_size(reply) <=
144  m0_rpc_session_get_max_item_size(request->ri_session));
145  M0_PRE(m0_rpc_conn_is_rcv(item2conn(request)));
146 
147  if (M0_FI_ENABLED("delay_reply")) {
148  M0_LOG(M0_DEBUG, "%p reply delayed", request);
149  m0_nanosleep(m0_time(M0_RPC_ITEM_RESEND_INTERVAL,
150  200 * 1000 * 1000), NULL);
151  }
152 
153  reply->ri_resend_interval = M0_TIME_NEVER;
154  reply->ri_rpc_time = m0_time_now();
155  reply->ri_session = request->ri_session;
156  machine = reply->ri_rmachine = request->ri_rmachine;
157 
158  reply->ri_prio = request->ri_prio;
159  reply->ri_deadline = 0;
160  reply->ri_error = 0;
161 
162  m0_rpc_machine_lock(machine);
163  m0_rpc_item_sm_init(reply, M0_RPC_ITEM_OUTGOING);
164  m0_rpc_item_send_reply(request, reply);
165  m0_rpc_machine_unlock(machine);
166 }

184 M0_INTERNAL void m0_rpc_oneway_item_post_locked(const struct m0_rpc_conn *conn,
185  struct m0_rpc_item *item)
186 {
187  M0_PRE(conn != NULL &&
188  m0_rpc_machine_is_locked(conn->c_rpc_machine));
189  M0_PRE(item != NULL && m0_rpc_item_is_oneway(item));
190 
191  /*
192  * Rpc always acquires an *internal* reference to "all" items (Here
193  * one-way items). This reference is released when the item is sent.
194  */
195  m0_rpc_item_get(item);
196  item->ri_resend_interval = M0_TIME_NEVER;
197  item->ri_rpc_time = m0_time_now();
198  item->ri_rmachine = conn->c_rpc_machine;
199 
200  m0_rpc_item_sm_init(item, M0_RPC_ITEM_OUTGOING);
201  item->ri_nr_sent++;
202  m0_rpc_frm_enq_item(&conn->c_rpcchan->rc_frm, item);
203 }

229 M0_INTERNAL int m0_rpc_net_buffer_pool_setup(struct m0_net_domain *ndom,
230  struct m0_net_buffer_pool
231  *app_pool, uint32_t bufs_nr,
232  uint32_t tm_nr)
233 {
234  int rc;
235  uint32_t segs_nr;
236  m0_bcount_t seg_size;
237 
238  M0_ENTRY("net_dom: %p", ndom);
239  M0_PRE(ndom != NULL);
240  M0_PRE(app_pool != NULL);
241  M0_PRE(bufs_nr != 0);
242 
243  seg_size = m0_rpc_max_seg_size(ndom);
244  segs_nr = m0_rpc_max_segs_nr(ndom);
245  app_pool->nbp_ops = &b_ops;
246  rc = m0_net_buffer_pool_init(app_pool, ndom,
247  M0_NET_BUFFER_POOL_THRESHOLD,
248  segs_nr, seg_size, tm_nr, M0_SEG_SHIFT,
249  false);
250  if (rc != 0)
251  return M0_ERR_INFO(rc, "net_buf_pool: Initialization");
252 
253  m0_net_buffer_pool_lock(app_pool);
254  rc = m0_net_buffer_pool_provision(app_pool, bufs_nr);
255  m0_net_buffer_pool_unlock(app_pool);
256 
257  if (rc == bufs_nr)
258  return M0_RC(0);
259  m0_net_buffer_pool_fini(app_pool);
260  return M0_RC(-ENOMEM);
261 }

271 M0_INTERNAL uint32_t m0_rpc_bufs_nr(uint32_t len, uint32_t tms_nr)
272 {
273  return len +
274  /* It is used so that more than one free buffer is present
275  * for each TM when tms_nr > 8.
276  */
277  max32u(tms_nr / 4, 1) +
278  /* It is added so that frequent low_threshold callbacks of
279  * buffer pool can be reduced.
280  */
281  M0_NET_BUFFER_POOL_THRESHOLD;
282 }
