211 void test_verify(enum flock_tests test_id)
212 {
213  struct m0_rm_owner *clnt = rm_ctxs[SERVER_1].rc_test_data.rd_owner;
214  struct m0_rm_owner *srv = rm_ctxs[SERVER_2].rc_test_data.rd_owner;
215  struct m0_sm_group *smgrp;
216  bool validation_lock = false;
217 
218  switch(test_id) {
219  case LOCK_ON_CLIENT_TEST:
220  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(&srv->ro_sublet));
221  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(&clnt->ro_borrowed));
222  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
223  &clnt->ro_owned[OWOS_CACHED]));
224  break;
225  case LOCK_ON_SERVER_TEST:
226  M0_UT_ASSERT(m0_rm_ur_tlist_is_empty(&srv->ro_sublet));
227  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
228  &srv->ro_owned[OWOS_CACHED]));
229  M0_UT_ASSERT(m0_rm_ur_tlist_is_empty(&clnt->ro_borrowed));
230  break;
231  case DISTRIBUTED_LOCK_TEST:
232  smgrp = owner_grp(srv);
233  if (!m0_mutex_is_locked(&smgrp->s_lock)) {
234  validation_lock = true;
235  m0_rm_owner_lock(srv);
236  }
237  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(&srv->ro_sublet) ||
238  !m0_rm_ur_tlist_is_empty(
239  &srv->ro_owned[OWOS_CACHED]) ||
240  !m0_rm_ur_tlist_is_empty(
241  &srv->ro_owned[OWOS_HELD]));
242  if (validation_lock)
243  m0_rm_owner_unlock(srv);
244  break;
245  default:
246  break;
247  }
248 }

250 static void dlock(enum rm_server srv_id, int n)
251 {
252  struct m0_rm_owner *owner = rm_ctxs[srv_id].rc_test_data.rd_owner;
253  struct m0_rm_incoming req;
254  int rc;
255 
256  M0_SET0(&req);
257  m0_file_lock(owner, &req);
258  m0_rm_owner_lock(owner);
259  test_verify(DISTRIBUTED_LOCK_TEST);
260  rc = m0_sm_timedwait(&req.rin_sm,
261  M0_BITS(RI_SUCCESS, RI_FAILURE),
262  M0_TIME_NEVER);
263  M0_UT_ASSERT(rc == 0);
264  M0_UT_ASSERT(req.rin_rc == 0);
265  M0_UT_ASSERT(incoming_state(&req) == RI_SUCCESS);
266  m0_rm_owner_unlock(owner);
267  if (srv_id == SERVER_1) {
268  clnt_counter += n;
269  loan_session_set(SERVER_2, SERVER_1);
270  } else if (srv_id == SERVER_2) {
271  srv_counter += n;
272  }
273  m0_file_unlock(&req);
274  M0_SET0(&req);
275 }

361  m0_file_unlock(&req);

478 static void flock_utfini(void)
479 {
480  int32_t i;
481 
482  /*
483  * Windup the server first, then the client. Trying to stop
484  * client first may put it into INSOLVENT state. This will
485  * finalize UT-RM objects hierarchy.
486  */
487  rm_ctx_server_windup(SERVER_2);
488  rm_ctx_server_windup(SERVER_1);
489 
490  /*
491  * Following loops cannot be combined.
492  * The ops within the loops need sync points. Hence they are separate.
493  */
494  /* Disconnect the servers. */
495  for (i = test_servers_nr - 1; i >= 0; --i) {
496  rm_ctx_server_stop(i);
497  }
498  rm_ctxs_conf_fini(rm_ctxs, test_servers_nr);
499  /*
500  * Finalise the servers. Must be done in the reverse order, so that the
501  * first initialised reqh is finalised last.
502  */
503  for (i = test_servers_nr - 1; i >= 0; --i) {
504  rm_ctx_fini(&rm_ctxs[i]);
505  }
506  for (i = 0; i < LOCK_TESTS_NR; ++i) {
507  m0_clink_del_lock(&tests_clink[i]);
508  m0_clink_fini(&tests_clink[i]);
509  }
510  m0_chan_fini_lock(&rm_ut_tests_chan);
511  m0_mutex_fini(&rm_ut_tests_chan_mutex);
512 }
