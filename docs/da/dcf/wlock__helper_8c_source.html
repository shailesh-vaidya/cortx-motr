57 static void write_lock_complete(struct m0_rm_incoming *in,
58  int32_t rc)
59 {
60  M0_ENTRY("incoming %p, rc %d", in, rc);
61  wlx.wlc_rc = rc;
62  m0_semaphore_up(&wlx.wlc_sem);
63  M0_LEAVE();
64 }

76 static int wlock_ctx_create(struct m0_rpc_machine *rpc_mach, const char *rm_ep)
77 {
78  int rc;
79 
80  wlx.wlc_rmach = rpc_mach;
81  m0_rwlockable_domain_type_init(&wlx.wlc_dom, &wlx.wlc_rt);
82  m0_rw_lockable_init(&wlx.wlc_rwlock, &M0_RWLOCK_FID, &wlx.wlc_dom);
83  m0_fid_tgenerate(&wlx.wlc_owner_fid, M0_RM_OWNER_FT);
84  m0_rm_rwlock_owner_init(&wlx.wlc_owner, &wlx.wlc_owner_fid,
85  &wlx.wlc_rwlock, NULL);
86  wlx.wlc_rm_addr = m0_strdup(rm_ep);
87  rc = m0_semaphore_init(&wlx.wlc_sem, 0);
88  return M0_RC(rc);
89 }

111 static void _write_lock_get(struct wlock_ctx *wlx)
112 {
113  M0_ENTRY("wlx=%p", wlx);
114  M0_PRE(wlx != NULL);
115  m0_rm_rwlock_req_init(&wlx->wlc_req, &wlx->wlc_owner, &ri_ops,

122 static void wlock_ctx_destroy(struct wlock_ctx *wlx)
123 {
124  M0_ENTRY("wlx=%p", wlx);
125  M0_PRE(wlx != NULL);
126  M0_LOG(M0_DEBUG, "owner wound up");
127  m0_rm_rwlock_owner_fini(&wlx->wlc_owner);
128  m0_rw_lockable_fini(&wlx->wlc_rwlock);
129  m0_rwlockable_domain_type_fini(&wlx->wlc_dom, &wlx->wlc_rt);
130  M0_LEAVE();
131 }

133 static void wlock_ctx_disconnect(struct wlock_ctx *wlx)
134 {
135  int rc;
136 
137  M0_ENTRY("wlx=%p", wlx);
138  M0_PRE(_0C(wlx != NULL) && _0C(!M0_IS0(&wlx->wlc_sess)));
139  rc = m0_rpc_session_destroy(&wlx->wlc_sess, M0_TIME_NEVER);
140  if (rc != 0)
141  M0_LOG(M0_ERROR, "Failed to destroy wlock session");
142  rc = m0_rpc_conn_destroy(&wlx->wlc_conn, M0_TIME_NEVER);
143  if (rc != 0)
144  M0_LOG(M0_ERROR, "Failed to destroy wlock connection");
145  M0_LEAVE();
146 
147 }

149 static void wlock_ctx_owner_windup(struct wlock_ctx *wlx)
150 {

153  M0_ENTRY("wlx=%p", wlx);

157  M0_TIME_NEVER);

159  M0_LEAVE();

172 static void _write_lock_put(struct wlock_ctx *wlx)
173 {
174  struct m0_rm_incoming *req;
175 
176  M0_ENTRY("wlx=%p", wlx);
177  M0_PRE(wlx != NULL);
178  req = &wlx->wlc_req;
179  m0_rm_credit_put(req);
180  m0_rm_incoming_fini(req);
181  wlock_ctx_owner_windup(wlx);
182  wlock_ctx_creditor_unset(wlx);
183  M0_LEAVE();
184 }

195 M0_INTERNAL int rm_write_lock_get(struct m0_rpc_machine *rpc_mach,
196  const char *rm_ep)
197 {
198  int rc;
199 
200  rc = wlock_ctx_create(rpc_mach, rm_ep);
201  if (rc != 0) {
202  M0_ERR(rc);
203  goto fail;
204  }
205  rc = wlock_ctx_connect(&wlx);
206  if (rc != 0) {
207  M0_ERR_INFO(rc, "ep=%s", rm_ep);
208  goto ctx_free;
209  }
210  wlock_ctx_creditor_setup(&wlx);
211  _write_lock_get(&wlx);
212  m0_semaphore_down(&wlx.wlc_sem);
213  rc = wlx.wlc_rc;
214  if (rc != 0) {
215  M0_ERR(rc);
216  goto ctx_destroy;
217  }
218  return M0_RC(rc);
219 ctx_destroy:
220  wlock_ctx_owner_windup(&wlx);
221  wlock_ctx_creditor_unset(&wlx);
222  wlock_ctx_destroy(&wlx);
223  wlock_ctx_disconnect(&wlx);
224 ctx_free:
225  m0_free0(&wlx.wlc_rm_addr);
226 fail:
227  return M0_ERR(rc);
228 }
