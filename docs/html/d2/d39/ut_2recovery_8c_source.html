147 static void be_ut_recovery_log_fill(struct be_ut_recovery_ctx *ctx,
148  int record_nr,
149  int discard_nr,
150  bool last_incomplete)
151 {
152  struct m0_be_log *log = &ctx->burc_log;
153  struct m0_mutex *lock = &ctx->burc_lock;
154  struct m0_be_log_record *record;
155  int i;
156  int rc;
157 
158  M0_ALLOC_ARR(ctx->burc_records, record_nr);
159  M0_UT_ASSERT(ctx->burc_records != NULL);
160 
161  for (i = 0; i < record_nr; ++i) {
162  record = &ctx->burc_records[i];
163  be_ut_recovery_log_record_init_one(record, log);
164  m0_mutex_lock(lock);
165  rc = m0_be_log_reserve(log, BE_UT_RECOVERY_LOG_RESERVE_SIZE);
166  M0_UT_ASSERT(rc == 0);
167  m0_be_log_record_io_size_set(record, 0,
168  BE_UT_RECOVERY_LOG_LIO_SIZE);
169  m0_be_log_record_io_prepare(record, SIO_WRITE,
170  BE_UT_RECOVERY_LOG_RESERVE_SIZE);
171  m0_mutex_unlock(lock);
172  rc = M0_BE_OP_SYNC_RET(op,
173  m0_be_log_record_io_launch(record, &op),
174  bo_sm.sm_rc);
175  M0_UT_ASSERT(rc == 0);
176 
177  m0_mutex_lock(lock);
178  if (i < discard_nr) {
179  m0_be_log_record_discard(record->lgr_log,
180  record->lgr_size);
181  } else {
182  m0_be_log_record_skip_discard(record);
183  }
184  m0_mutex_unlock(lock);
185  }
186 
187  for (i = 0; i < record_nr; ++i) {
188  record = &ctx->burc_records[i];
189  be_ut_recovery_log_record_fini_one(record);
190  }
191  m0_free(ctx->burc_records);
192 }

194 static int be_ut_recovery_iter_count(struct be_ut_recovery_ctx *ctx)
195 {
196  struct m0_be_log *log = &ctx->burc_log;
197  struct m0_mutex *lock = &ctx->burc_lock;
198  struct m0_be_log_record record = {};
199  struct m0_be_log_record_iter iter = {};
200  int count = 0;
201  int rc;
202 
203  be_ut_recovery_log_record_init_one(&record, log);
204  rc = m0_be_log_record_iter_init(&iter);
205  M0_UT_ASSERT(rc == 0);
206  while (m0_be_log_recovery_record_available(log)) {
207  m0_be_log_recovery_record_get(log, &iter);
208  ++count;
209 
210  /* discard this record to move log's pointers */
211  m0_be_log_record_assign(&record, &iter, true);
212  m0_mutex_lock(lock);
213  m0_be_log_record_io_prepare(&record, SIO_READ, 0);
214  m0_mutex_unlock(lock);
215  rc = M0_BE_OP_SYNC_RET(op,
216  m0_be_log_record_io_launch(&record, &op),
217  bo_sm.sm_rc);
218  M0_UT_ASSERT(rc == 0);
219  m0_mutex_lock(lock);
220  m0_be_log_record_discard(record.lgr_log, record.lgr_size);
221  m0_mutex_unlock(lock);
222  m0_be_log_record_reset(&record);
223  }
224  m0_be_log_record_iter_fini(&iter);
225  be_ut_recovery_log_record_fini_one(&record);
226 
227  return count;
228 }

230 void m0_be_ut_recovery(void)
231 {
232  struct be_ut_recovery_ctx ctx = {};
233  int count;
234  int nr;
235 
236  be_ut_recovery_log_init(&ctx);
237 
238  /* empty log */
239  count = be_ut_recovery_iter_count(&ctx);
240  M0_UT_ASSERT(count == 0);
241 
242  be_ut_recovery_log_fill(&ctx, 10, 10, false);
243  be_ut_recovery_log_reopen(&ctx);
244  count = be_ut_recovery_iter_count(&ctx);
245  M0_UT_ASSERT(count == 0);
246 
247  be_ut_recovery_log_fill(&ctx, 10, 5, false);
248  be_ut_recovery_log_reopen(&ctx);
249  count = be_ut_recovery_iter_count(&ctx);
250  M0_UT_ASSERT(count == 5);
251 
252  nr = BE_UT_RECOVERY_LOG_SIZE / BE_UT_RECOVERY_LOG_RESERVE_SIZE * 2;
253  be_ut_recovery_log_fill(&ctx, nr, nr - 5, false);
254  be_ut_recovery_log_reopen(&ctx);
255  count = be_ut_recovery_iter_count(&ctx);
256  M0_UT_ASSERT(count == 5);
257 
258  be_ut_recovery_log_reopen(&ctx);
259  count = be_ut_recovery_iter_count(&ctx);
260  M0_UT_ASSERT(count == 0);
261 
262  be_ut_recovery_log_fini(&ctx);
263 }
