143 M0_INTERNAL bool m0_reqh_service_invariant(const struct m0_reqh_service *svc)
144 {
145  return _0C(m0_reqh_service_bob_check(svc)) &&
146  _0C(M0_IN(svc->rs_sm.sm_state, (M0_RST_INITIALISING, M0_RST_INITIALISED,
147  M0_RST_STARTING, M0_RST_STARTED,
148  M0_RST_STOPPING, M0_RST_STOPPED,
149  M0_RST_FAILED))) &&
150  _0C(svc->rs_type != NULL && svc->rs_ops != NULL &&
151  (svc->rs_ops->rso_start_async != NULL ||
152  svc->rs_ops->rso_start != NULL)) &&
153  _0C(ergo(M0_IN(svc->rs_sm.sm_state, (M0_RST_INITIALISED,
154  M0_RST_STARTING, M0_RST_STARTED, M0_RST_STOPPING,
155  M0_RST_STOPPED, M0_RST_FAILED)),
156  svc->rs_reqh != NULL)) &&
157  _0C(ergo(M0_IN(svc->rs_sm.sm_state, (M0_RST_STARTED, M0_RST_STOPPING,
158  M0_RST_STOPPED, M0_RST_FAILED)),
159  m0_reqh_svc_tlist_contains(&svc->rs_reqh->rh_services, svc))) &&
160  _0C(ergo(svc->rs_reqh != NULL,
161  M0_IN(m0_reqh_lockers_get(svc->rs_reqh, svc->rs_type->rst_key),
162  (NULL, svc)))) &&
163  _0C(svc->rs_level > M0_RS_LEVEL_UNKNOWN);
164 }

207 static void reqh_service_ha_event(struct m0_reqh_service *service,
208  enum m0_reqh_service_state state)
209 {
210  static const enum m0_conf_ha_service_event state2event[] = {
211  [M0_RST_STARTING] = M0_CONF_HA_SERVICE_STARTING,
212  [M0_RST_STARTED] = M0_CONF_HA_SERVICE_STARTED,
213  [M0_RST_STOPPING] = M0_CONF_HA_SERVICE_STOPPING,
214  [M0_RST_STOPPED] = M0_CONF_HA_SERVICE_STOPPED,
215  [M0_RST_FAILED] = M0_CONF_HA_SERVICE_FAILED,
216  };
217 
218  if (!M0_IN(state, (M0_RST_STARTING, M0_RST_STARTED,
219  M0_RST_STOPPING, M0_RST_STOPPED, M0_RST_FAILED)))
220  return;
221  if (m0_get()->i_ha == NULL || m0_get()->i_ha_link == NULL) {
222  M0_LOG(M0_DEBUG, "can't report service HA event=%d "
223  "service_fid="FID_F, state2event[state],
224  FID_P(&service->rs_service_fid));
225  return;
226  }
227  m0_conf_ha_service_event_post(m0_get()->i_ha, m0_get()->i_ha_link,
228  &service->rs_reqh->rh_fid,
229  &service->rs_service_fid,
230  &service->rs_service_fid,
231  m0_process(),
232  state2event[state],
233  service->rs_type->rst_typecode);
234 }

365  reqh_service_started_common(reqh, service);
366  else
367  reqh_service_failed_common(reqh, service, key);
368  M0_POST(ergo(rc == 0, m0_reqh_service_invariant(service)));
369  m0_rwlock_write_unlock(&reqh->rh_rwlock);
370 
371  return M0_RC(rc);

428 M0_INTERNAL void m0_reqh_service_init(struct m0_reqh_service *service,
429  struct m0_reqh *reqh,
430  const struct m0_fid *fid)
431 {
432  M0_PRE(service != NULL && reqh != NULL &&
433  service->rs_sm.sm_state == M0_RST_INITIALISING);
434  /* Currently fid may be NULL */
435  M0_PRE(fid == NULL || m0_fid_is_valid(fid));
436 
437  m0_sm_init(&service->rs_sm, &service_states_conf, M0_RST_INITIALISING,
438  &reqh->rh_sm_grp);
439 
440  if (fid != NULL)
441  service->rs_service_fid = *fid;
442  service->rs_reqh = reqh;
443  m0_mutex_init(&service->rs_mutex);
444  reqh_service_state_set(service, M0_RST_INITIALISED);
445 
446  /*
447  * We want to track these services externally so add them to the list
448  * just as soon as they enter the M0_RST_INITIALISED state.
449  * They will be left on the list until they get fini'd.
450  */
451  m0_reqh_svc_tlink_init_at(service, &reqh->rh_services);
452  service->rs_fom_key = m0_locality_lockers_allot();
453  M0_POST(!m0_buf_is_set(&service->rs_ss_param));
454  M0_POST(m0_reqh_service_invariant(service));
455 }

892 M0_INTERNAL void m0_reqh_service_ctxs_shutdown_prepare(struct m0_reqh *reqh)
893 {
894  struct m0_reqh_service_ctx *ctx;
895 
896  M0_ENTRY();
897  /*
898  * Step 1: Flag every context for abortion.
899  *
900  * The idea is to make any reconnecting service context to encounter the
901  * flag and go offline before real context destruction occurs.
902  */
903  m0_tl_for(pools_common_svc_ctx, &reqh->rh_pools->pc_svc_ctxs, ctx) {
904  M0_ASSERT(!reqh_service_ctx_sm_is_locked(ctx));
905  /*
906  * Need to prevent context from being activated by HA
907  * notification, service or process related.
908  */
909  m0_reqh_service_ctx_unsubscribe(ctx);
910  reqh_service_ctx_sm_lock(ctx);
911  /*
912  * Context is unconditionally flagged to abort. In case of
913  * connection broken due to network reasons, the context will be
914  * put offline and prevented from further connection.
915  *
916  * This way yet still incomplete context is commanded to go
917  * offline. Presently offline context does not require any
918  * special attention.
919  */
920  reqh_service_ctx_flag_set(ctx, M0_RSC_RLINK_ABORT);
921  reqh_service_ctx_sm_unlock(ctx);
922  } m0_tl_endfor;
923 
924  /*
925  * Step 2: Spot a still connecting context and wait for fom activity.
926  *
927  * The idea is to let not a context with dormant fom go for destruction.
928  * The rpc link fom activity has to occur first to make sure abortion
929  * flag goes in effect. Besides the current one, any other reconnecting
930  * context may be offlined in background because of abortion flag
931  * already installed on previous step.
932  */
933  m0_tl_for(pools_common_svc_ctx, &reqh->rh_pools->pc_svc_ctxs, ctx) {
934  bool connecting;
935 
936  M0_ASSERT(!reqh_service_ctx_sm_is_locked(ctx));
937  reqh_service_ctx_sm_lock(ctx);
938  connecting = CTX_STATE(ctx) == M0_RSC_CONNECTING;
939  M0_LOG(M0_DEBUG, "[%d] %s (%d)", CTX_STATE(ctx),
940  m0_rpc_link_end_point(&ctx->sc_rlink), ctx->sc_type);
941  if (connecting) {
942  /*
943  * .rlk_rc must not be 0 to meet M0_RSC_OFFLINE
944  * conditions.
945  * */
946  ctx->sc_rlink.rlk_rc = M0_ERR(-EPERM);
947  reqh_service_ctx_flag_clear(ctx, M0_RSC_RLINK_CONNECT);
948  /*
949  * From this moment we rely on rpc link timeout to wake
950  * up the link fom to be ultimately aborted by the flag
951  * specification.
952  */
953  }
954  reqh_service_ctx_sm_unlock(ctx);
955  /* Do waiting outside the sm lock. */
956  if (connecting)
957  reqh_service_ctx_state_wait(ctx, M0_RSC_OFFLINE);
958  } m0_tl_endfor;
959  M0_LEAVE();
960 }

984 static void reqh_service_ctx_ast_cb(struct m0_sm_group *grp,
985  struct m0_sm_ast *ast)
986 {
987  struct m0_reqh_service_ctx *ctx = M0_AMB(ctx, ast, sc_rlink_ast);
988  bool connected;
989  bool disconnect;
990 
991  reqh_service_ctx_sm_lock(ctx);
992  M0_ENTRY("'%s' ctx of service '%s' state: %d",
993  m0_rpc_machine_ep(ctx->sc_rlink.rlk_conn.c_rpc_machine),
994  m0_rpc_link_end_point(&ctx->sc_rlink), CTX_STATE(ctx));
995  M0_PRE(M0_IN(CTX_STATE(ctx), (M0_RSC_CONNECTING,
996  M0_RSC_DISCONNECTING)));
997 
998  connected = m0_rpc_link_is_connected(&ctx->sc_rlink);
999  disconnect = reqh_service_ctx_flag_is_set(ctx, M0_RSC_RLINK_DISCONNECT);
1000 
1001  switch (CTX_STATE(ctx)) {
1002  case M0_RSC_CONNECTING:
1003  if (connected) {
1004  M0_ASSERT(ctx->sc_rlink.rlk_rc == 0);
1005  reqh_service_ctx_state_move(ctx, M0_RSC_ONLINE);
1006  M0_LOG(M0_DEBUG, "connecting -> online: %s (%d)",
1007  m0_rpc_link_end_point(&ctx->sc_rlink),
1008  ctx->sc_type);
1009  } else {
1010  M0_ASSERT(ctx->sc_rlink.rlk_rc != 0);
1011  /* Reconnect later. */
1012  reqh_service_ctx_state_move(ctx, M0_RSC_OFFLINE);
1013  M0_LOG(M0_DEBUG, "connecting -> offline: %s (%d)",
1014  m0_rpc_link_end_point(&ctx->sc_rlink),
1015  ctx->sc_type);
1016  }
1017  break;
1018 
1019  case M0_RSC_DISCONNECTING:
1020  M0_ASSERT(!connected);
1021  reqh_service_ctx_state_move(ctx, M0_RSC_OFFLINE);
1022  reqh_service_ctx_destroy_if_abandoned(ctx);
1023  M0_LOG(M0_DEBUG, "disconnecting -> offline: %s (%d)",
1024  m0_rpc_link_end_point(&ctx->sc_rlink), ctx->sc_type);
1025  break;
1026 
1027  default:
1028  M0_IMPOSSIBLE("Invalid state: %d", CTX_STATE(ctx));
1029  }
1030  M0_LOG(M0_DEBUG, "state: %d", CTX_STATE(ctx));
1031 
1032  /* Cancel only established session. */
1033  if (reqh_service_ctx_flag_is_set(ctx, M0_RSC_RLINK_CANCEL) &&
1034  connected) {
1035  m0_rpc_session_cancel(&ctx->sc_rlink.rlk_sess);
1036  reqh_service_ctx_state_move(ctx, M0_RSC_CANCELLED);
1037  reqh_service_ctx_flag_clear(ctx, M0_RSC_RLINK_CANCEL);
1038  }
1039  /*
1040  * Reconnect every time `ctx' reaches OFFLINE state until
1041  * m0_reqh_service_disconnect() is called unless explicit abortion is
1042  * requested.
1043  */
1044  if (CTX_STATE(ctx) == M0_RSC_OFFLINE) {
1045  if (!reqh_service_ctx_flag_is_set(ctx, M0_RSC_RLINK_ABORT) &&
1046  !disconnect)
1047  reqh_service_connect_locked(ctx,
1048  m0_time_from_now(REQH_SVC_CONNECT_TIMEOUT, 0));
1049  }
1050  /* m0_reqh_service_disconnect() was called during connecting. */
1051  if (CTX_STATE(ctx) == M0_RSC_ONLINE && disconnect)
1052  reqh_service_disconnect_locked(ctx);
1053 
1054  reqh_service_ctx_sm_unlock(ctx);
1055 }

1059  struct m0_reqh_service_ctx *ctx = M0_AMB(ctx, clink, sc_rlink_wait);
1060  struct m0_locality *loc;

1062  /*

1080 static bool process_event_handler(struct m0_clink *clink)
1081 {
1082  struct m0_reqh_service_ctx *ctx = M0_AMB(ctx, clink, sc_process_event);
1083  struct m0_conf_obj *obj = M0_AMB(obj, clink->cl_chan,
1084  co_ha_chan);

1087  M0_ENTRY();
