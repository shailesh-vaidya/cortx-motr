300 M0_INTERNAL bool entity_invariant_full(struct m0_entity *ent)
301 {
302  bool rc = false;
303  struct m0_sm_group *grp;
304 
305  grp = &ent->en_sm_group;
306  if (!m0_sm_group_is_locked(grp)) {
307  m0_sm_group_lock(grp);
308  rc = entity_invariant_locked(ent);
309  m0_sm_group_unlock(grp);
310  }
311  return M0_RC(rc);

371 M0_INTERNAL void m0_entity_init(struct m0_entity *entity,
372  struct m0_realm *parent,
373  const struct m0_uint128 *id,
374  const enum m0_entity_type type)
375 {
376  struct m0_sm_group *grp;
377 
378  M0_ENTRY();
379 
380  M0_PRE(entity != NULL);
381  M0_PRE(parent != NULL);
382  M0_PRE(parent->re_instance != NULL);
383  M0_PRE(id != NULL);
384  M0_PRE(entity_id_is_valid(id));
385  M0_PRE(m0_sm_conf_is_initialized(&entity_conf));
386  M0_PRE(m0_entity_type_is_valid(type));
387 
388  /* initalise the entity */
389  entity->en_type = type;
390  entity->en_id = *id;
391  entity->en_realm = parent;
392 
393  /* initalise the state machine */
394  grp = &entity->en_sm_group;
395  m0_sm_group_init(grp);
396  m0_sm_init(&entity->en_sm, &entity_conf, M0_ES_INIT, grp);
397  m0_mutex_init(&entity->en_pending_tx_lock);
398  spti_tlist_init(&entity->en_pending_tx);
399  M0_ASSERT(entity_invariant_full(entity));
400  M0_LEAVE();
401 }

438 void m0_entity_fini(struct m0_entity *entity)
439 {
440  struct m0_reqh_service_txid *iter;
441 
442  M0_ENTRY();
443 
444  M0_PRE(entity != NULL);
445  M0_PRE(M0_IN(entity->en_sm.sm_state,
446  (M0_ES_INIT, M0_ES_OPEN,
447  M0_ES_FAILED)));
448  M0_ASSERT(entity_invariant_full(entity));

451  m0_free0(&iter);

453  m0_mutex_fini(&entity->en_pending_tx_lock);
454  m0_sm_group_lock(&entity->en_sm_group);

456  m0_sm_move(&entity->en_sm, 0, M0_ES_CLOSING);
457  m0_sm_move(&entity->en_sm, 0, M0_ES_INIT);
458  }
459  m0_sm_fini(&entity->en_sm);
460  m0_sm_group_unlock(&entity->en_sm_group);
461  m0_sm_group_fini(&entity->en_sm_group);
462  M0_SET0(entity);
463  M0_LEAVE();
464 }

568 M0_INTERNAL int m0_op_get(struct m0_op **op, size_t size)
569 {
570  int rc = 0;
571 
572  M0_ENTRY();
573  M0_PRE(op != NULL);
574 
575  /* Allocate the op if necessary. */
576  if (*op == NULL) {
577  rc = m0_op_alloc(op, size);
578  if (rc != 0)
579  return M0_ERR(rc);
580  } else {
581  size_t cached_size = (*op)->op_size;
582 
583  if ((*op)->op_size < size)
584  return M0_ERR(-EMSGSIZE);
585 
586  /* 0 the pre-allocated operation. */
587  memset(*op, 0, cached_size);
588  /* @todo This forgets the original size. */
589  (*op)->op_size = size;
590  }
591  return M0_RC(0);
592 }

598 M0_INTERNAL void m0_op_cancel_one(struct m0_op *op)
599 {
600  struct m0_op_common *oc;
601 
602  M0_ENTRY();
603 
604  M0_PRE(m0_op_invariant(op));
605  M0_PRE(M0_IN(op->op_sm.sm_state, (M0_OS_LAUNCHED,
606  M0_OS_EXECUTED,
607  M0_OS_STABLE,
608  M0_OS_FAILED)));
609 
610  oc = bob_of(op, struct m0_op_common, oc_op, &oc_bobtype);
611  M0_PRE(oc->oc_cb_cancel != NULL);
612 
613  m0_sm_group_lock(&op->op_sm_group);
614 
615  if (!M0_IN(op->op_sm.sm_state, (M0_OS_STABLE,
616  M0_OS_FAILED))) {
617  if (oc->oc_cb_cancel != NULL) {
618  /*
619  * Not implemented for opcodes M0_OC_ALLOC,
620  * M0_ET_REALM, M0_EO_INVALID,
621  * M0_EO_SYNC, M0_EO_GETATTR,
622  * M0_EO_SETATTR, M0_EO_LAYOUT_GET,
623  * M0_EO_LAYOUT_SET
624  */
625  oc->oc_cb_cancel(oc);
626  } else {
627  op->op_rc = M0_ERR(-EINVAL);
628  m0_sm_move(&op->op_sm, op->op_rc, M0_OS_FAILED);
629  }
630  }
631 
632  /* Call the op-type's cancel function */
633  m0_sm_group_unlock(&op->op_sm_group);
634 
635  M0_LEAVE();
636 }

649  M0_ET_IDX)))

652 
653  M0_LEAVE();

685 M0_INTERNAL void m0_op_launch_one(struct m0_op *op)
686 {
687  struct m0_op_common *oc;
688  struct m0_client *m0c;
689  int rc;
690 
691  M0_ENTRY();
692 
693  M0_PRE(m0_op_invariant(op));
694  M0_PRE(op->op_sm.sm_state == M0_OS_INITIALISED);
695  /* SYNC op doesn't have `entity`. */
696  M0_PRE(m0_op_entity_invariant(op));
697 
698  oc = bob_of(op, struct m0_op_common, oc_op, &oc_bobtype);
699  M0_PRE(oc->oc_cb_launch != NULL);
700 
701  m0c = m0__op_instance(op);
702  M0_PRE(m0c != NULL);
703 
704  rc = m0__io_ref_get(m0c);
705  if (rc != 0) {
706  m0_sm_group_lock(&op->op_sm_group);
707  m0_sm_move(&op->op_sm, rc, M0_OS_FAILED);
708  op->op_rc = rc;
