94 M0_INTERNAL int m0_stob_types_init(void)
95 {
96  struct m0_stob_types *types = stob_types_get();
97  struct m0_stob_type *type;
98  int i;
99  int rc = 0;
100 
101  types_tlist_init(&types->sts_stypes);
102 
103  for (i = 0; i < ARRAY_SIZE(stob_types_reg); ++i) {
104  rc = stob_type_copy(stob_types_reg[i], &type);
105  if (rc != 0)
106  break;
107  m0_stob_type_register(type);
108  }
109  if (rc != 0)
110  stob_types_destroy_list(types);
111 
112  return M0_RC(rc);
113 }
