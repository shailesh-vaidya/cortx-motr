157 static int spiel_copy_file(const char *source, const char* dest)
158 {
159  char buf[1024];
160  int in_fd;
161  int out_fd;
162  size_t result;
163  int rc = 0;
164 
165  in_fd = open(source, O_RDONLY);
166  if (in_fd < 0)
167  return -EINVAL;
168 
169  out_fd = open(dest, O_WRONLY | O_CREAT, 0666);
170  if (in_fd < 0) {
171  close(in_fd);
172  return -EINVAL;
173  }
174 
175  while (1) {
176  rc = (result = read(in_fd, &buf[0], sizeof(buf))) >= 0 ?
177  0 : -EINVAL;
178  if (result == 0 || rc != 0)
179  break;
180  rc = write(out_fd, &buf[0], result) == result ? 0 : -EINVAL;
181  if (rc != 0)
182  break;
183  }
184  close(in_fd);
185  close(out_fd);
186 
187  return rc;
188 }
