82 static struct m0_rpc_item *get_item(struct m0_rpc_item_source *ris,
83  size_t max_payload_size)
84 {
85  struct m0_fop *fop;
86  struct m0_rpc_machine *machine = ris->ris_conn->c_rpc_machine;
87 
88  M0_UT_ASSERT(m0_rpc_machine_is_locked(machine));
89  get_item_calls++;
90  fop = m0_fop_alloc(&m0_rpc_arrow_fopt, NULL, machine);
91  M0_UT_ASSERT(fop != NULL);
92  item = &fop->f_item;
93  /* without this "get", the item will be freed as soon as it is
94  sent/failed. The reference is required to protect item until
95  item_source_test() performs its checks on the item.
96  */
97  m0_rpc_item_get(item);
98 
99  if (M0_FI_ENABLED("max"))
100  item->ri_size = m0_rpc_session_get_max_item_payload_size(
101  &cctx.rcx_session);
102 
103  if (M0_FI_ENABLED("not_multiple_of_8bytes"))
104  item->ri_size = max_payload_size - 1;
105 
106  M0_UT_ASSERT(m0_rpc_item_is_oneway(item) &&
107  m0_rpc_item_size(item) <= max_payload_size);
108  return item;
109 }

137 static void item_source_limits_test(void)
138 {
139  struct m0_rpc_item_source ris;
140  struct m0_rpc_frm *frm;
141  struct m0_rpc_packet *p;
142  int cond;
143 
144  m0_rpc_item_source_init(&ris, "test-item-source", &ris_ops);
145  M0_UT_ASSERT(ris.ris_ops == &ris_ops);
146  m0_rpc_item_source_register(conn, &ris);
147  frm = &conn->c_rpcchan->rc_frm;
148 
149  for (cond = 0; cond < 3; cond++) {
150  M0_ALLOC_PTR(p);
151  M0_UT_ASSERT(p != NULL);
152  has_item_calls = get_item_calls = 0;
153  m0_fi_enable_once("has_item", "yes");
154  switch (cond) {
155  case 0:
156  /* For the minimum item size */
157  break;
158  case 1:
159  m0_fi_enable_once("get_item", "max");
160  break;
161  case 2:
162  m0_fi_enable_once("get_item", "not_multiple_of_8bytes");
163  break;
164  default:
165  M0_IMPOSSIBLE("not supported");
166  }
167 
168  m0_rpc_machine_lock(conn->c_rpc_machine);
169  m0_rpc_packet_init(p, frm_rmachine(frm));
170  frm_fill_packet_from_item_sources(frm, p);
171  m0_rpc_machine_unlock(conn->c_rpc_machine);
172  M0_UT_ASSERT(has_item_calls > get_item_calls &&
173  get_item_calls == 1);
174  M0_UT_ASSERT(item->ri_sm.sm_state == M0_RPC_ITEM_SENDING);
175  m0_rpc_machine_lock(conn->c_rpc_machine);
176  m0_rpc_item_put(item);
177  m0_rpc_item_change_state(item, M0_RPC_ITEM_SENT);
178  m0_rpc_packet_discard(p);
179  m0_rpc_item_put(item);
180  m0_rpc_machine_unlock(conn->c_rpc_machine);
181  }
182  m0_rpc_item_source_deregister(&ris);
183  m0_rpc_item_source_fini(&ris);
184 }

186 static void item_source_test(void)
187 {
188  struct m0_rpc_item_source *ris;
189  int trigger;
190  int rc;
191 
192  /*
193  Test:
194  - Confirm that formation correctly pulls items and sends them.
195  - Also verify that periodic item-source drain works.
196  */
197  M0_ALLOC_PTR(ris);
198  M0_UT_ASSERT(ris != NULL);
199  m0_rpc_item_source_init(ris, "test-item-source", &ris_ops);
200  m0_rpc_item_source_register(conn, ris);
201 
202  for (trigger = 0; trigger < 2; trigger++) {
203  m0_fi_enable_once("has_item", "yes");
204  m0_fi_enable("frm_is_ready", "ready");
205  has_item_calls = get_item_calls = 0;
206  switch (trigger) {
207  case 0:
208  m0_rpc_machine_lock(conn->c_rpc_machine);
209  m0_rpc_frm_run_formation(&conn->c_rpcchan->rc_frm);
210  m0_rpc_machine_unlock(conn->c_rpc_machine);
211  break;
212  case 1:
213  m0_rpc_machine_lock(conn->c_rpc_machine);
214  m0_rpc_machine_drain_item_sources(conn->c_rpc_machine,
215  128);
216  m0_rpc_machine_unlock(conn->c_rpc_machine);
217  break;
218  default:
219  M0_IMPOSSIBLE("only two triggers");
220  }
221  M0_UT_ASSERT(has_item_calls > get_item_calls &&
222  get_item_calls == 1);
223  rc = m0_rpc_item_timedwait(item, M0_BITS(M0_RPC_ITEM_SENT,
224  M0_RPC_ITEM_FAILED),
225  m0_time_from_now(2, 0));
226  M0_UT_ASSERT(rc == 0);
227  M0_UT_ASSERT(item->ri_sm.sm_state == M0_RPC_ITEM_SENT);
228 
229  /* riso_has_item() is set to return false.
230  Test that get_item does not get called when has_item
231  returns false
232  */
233  has_item_calls = get_item_calls = 0;
234  m0_rpc_machine_lock(conn->c_rpc_machine);
235  m0_rpc_frm_run_formation(&conn->c_rpcchan->rc_frm);
236  m0_rpc_machine_unlock(conn->c_rpc_machine);
237  M0_UT_ASSERT(has_item_calls > get_item_calls &&
238  get_item_calls == 0);
239 
240  m0_fi_disable("frm_is_ready", "ready");
241  m0_rpc_machine_lock(conn->c_rpc_machine);
242  m0_rpc_item_put(item);
243  m0_rpc_machine_unlock(conn->c_rpc_machine);
244  }
245  m0_rpc_item_source_deregister(ris);
246  m0_rpc_item_source_fini(ris);
247  m0_free(ris);
248 }
