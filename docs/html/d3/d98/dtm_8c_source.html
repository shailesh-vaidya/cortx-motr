101 M0_INTERNAL int m0_dtx_open_sync(struct m0_dtx *tx)
102 {
103  int rc;
104 
105  m0_dtx_open(tx);
106  rc = m0_be_tx_timedwait(&tx->tx_betx,
107  M0_BITS(M0_BTS_ACTIVE, M0_BTS_FAILED),
108  M0_TIME_NEVER);
109  if (m0_be_tx_state(&tx->tx_betx) == M0_BTS_ACTIVE)
110  m0_dtx_opened(tx);
111 
112  return M0_RC(rc);
113 }
