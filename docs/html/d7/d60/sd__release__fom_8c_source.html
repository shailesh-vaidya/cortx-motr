123 int imitate_release_fop_recv(struct test_rpc_env *env)
124 {
125  int rc;
126  struct m0_fop *fop;
127  struct m0_fop_fdmi_rec_release *fop_data;
128  struct m0_reqh *reqh;
129  struct m0_rpc_item *rpc_item;
130 
131  M0_ENTRY();
132 
133  reqh = &g_sd_ut.motr.cc_reqh_ctx.rc_reqh;
134 
135  fop = m0_fop_alloc(&m0_fop_fdmi_rec_release_fopt, NULL,
136  &env->tre_rpc_machine);
137  M0_UT_ASSERT(fop != NULL);
138 
139  /* Overwrite to use my own fop release */
140  m0_ref_init(&fop->f_ref, 1, my_fop_release);
141 
142  fop_data = m0_fop_data(fop);
143  fop_data->frr_frid = rec_id_to_release;
144  fop_data->frr_frt = M0_FDMI_REC_TYPE_TEST;
145  rpc_item = &fop->f_item;
146 
147  m0_fop_rpc_machine_set(fop, &env->tre_rpc_machine);
148  m0_rpc_item_sm_init(rpc_item, M0_RPC_ITEM_INCOMING);
149 
150  rpc_item->ri_session = env->tre_session;
151 
152  m0_rpc_machine_lock(&env->tre_rpc_machine);
153  m0_rpc_item_change_state(rpc_item, M0_RPC_ITEM_ACCEPTED);
154  m0_rpc_machine_unlock(&env->tre_rpc_machine);
155 
156  m0_rpc_machine_lock(&env->tre_rpc_machine);
157  m0_rpc_session_hold_busy(env->tre_session);
158  m0_rpc_machine_unlock(&env->tre_rpc_machine);
159 
160  rc = m0_reqh_fop_handle(reqh, fop);
161  m0_fop_put_lock(fop);
162 
163  M0_LEAVE();
164  return rc;
165 }

170 void fdmi_sd_release_fom(void)
171 {
172  struct m0_fdmi_src *src = src_alloc();
173  int rc;
174  static struct m0_rpc_conn rpc_conn;
175  static struct m0_rpc_session rpc_session;
176  bool ok;
177  struct m0_fdmi_src_dock *src_dock = m0_fdmi_src_dock_get();
178 
179  M0_ENTRY();
180 
181  fdmi_serv_start_ut(&filterc_stub_ops);
182  prepare_rpc_env(&g_rpc_env, &g_sd_ut.motr.cc_reqh_ctx.rc_reqh,
183  &test_frm_ops, false, &rpc_conn, &rpc_session);
184  m0_semaphore_init(&g_sem, 0);
185  m0_semaphore_init(&g_sem2, 0);
186  m0_semaphore_init(&g_sem3, 0);
187  rc = m0_fdmi_source_register(src);
188  M0_UT_ASSERT(rc == 0);
189  g_src_rec.fsr_src = src;
190  m0_fdmi__record_init(&g_src_rec);
191  m0_fdmi__rec_id_gen(&g_src_rec);
192  m0_ref_init(&g_src_rec.fsr_ref, 1, g_src_rec_free);
193  m0_fdmi__fs_get(&g_src_rec);
194  fdmi_record_inflight_tlist_add_tail(
195  &src_dock->fsdc_rec_inflight, &g_src_rec);
196  rec_id_to_release = g_src_rec.fsr_rec_id;
197  rc = imitate_release_fop_recv(&g_rpc_env);
198  M0_UT_ASSERT(rc == 0);
199 
204  ok = m0_semaphore_timeddown(&g_sem, m0_time_from_now(10, 0));
205  M0_UT_ASSERT(ok);
206 
211  ok = m0_semaphore_timeddown(&g_sem, m0_time_from_now(10, 0));
212  M0_UT_ASSERT(ok);
213  fdmi_ut_packet_send_failed(&g_rpc_env.tre_rpc_machine,
214  g_sent_rpc_packet);
215  ok = m0_semaphore_timeddown(&g_sem2, m0_time_from_now(10, 0));
216  M0_UT_ASSERT(ok);
217  m0_fdmi__record_deinit(&g_src_rec);
218  m0_fdmi_source_deregister(src);
219  m0_fdmi_source_free(src);
220  m0_semaphore_fini(&g_sem);
221  m0_semaphore_fini(&g_sem2);
222  ok = m0_semaphore_timeddown(&g_sem3, m0_time_from_now(10, 0));
223  M0_UT_ASSERT(ok);
224  m0_semaphore_fini(&g_sem3);
225  unprepare_rpc_env(&g_rpc_env);
226  fdmi_serv_stop_ut();
227  M0_LEAVE();
228 }
