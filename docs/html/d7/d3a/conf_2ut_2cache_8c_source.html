66 
67  m0_conf_cache_lock(&m0_conf_ut_cache);

69  m0_conf_cache_unlock(&m0_conf_ut_cache);
70  M0_UT_ASSERT(rc == 0);
71 
72  M0_UT_ASSERT(m0_conf_cache_lookup(&m0_conf_ut_cache, fid) == obj);
73  *result = obj;
74 }

104 static void test_cache(void)
105 {
106  int rc;
107  struct m0_conf_obj *obj;
108  struct m0_fid samples[] = {
109  M0_FID_TINIT('p', ~0, 0),
110  M0_FID_TINIT('D', 7, 3)
111  };
112 
113  m0_forall(i, ARRAY_SIZE(samples),
114  ut_conf_obj_create(&samples[i], &obj), true);
115  ut_conf_obj_delete(obj); /* delete the last object */
116 
117  /* Duplicated identity. */
118  obj = m0_conf_obj_create(&samples[0], &m0_conf_ut_cache);
119  M0_UT_ASSERT(obj != NULL);
120 
121  m0_conf_cache_lock(&m0_conf_ut_cache);
122  rc = m0_conf_cache_add(&m0_conf_ut_cache, obj);
123  m0_conf_cache_unlock(&m0_conf_ut_cache);
124  M0_UT_ASSERT(rc == -EEXIST);
125 
126  m0_conf_cache_lock(&m0_conf_ut_cache);
127  m0_conf_obj_delete(obj);
128  m0_conf_cache_unlock(&m0_conf_ut_cache);
129 }
