58 M0_INTERNAL int m0t1fs_init(void)
59 {
60  struct m0_addb2_sys *sys = m0_addb2_global_get();
61  int rc;
62  int cpus;
63 
64  M0_ENTRY();
65  m0_mutex_init(&m0t1fs_mutex);
66  /*
67  * [1 - M0_NET_LNET_TMID_MAX / 2] for clients.
68  * [M0_NET_LNET_TMID_MAX / 2 - M0_NET_LNET_TMID_MAX] for server ep.
69  */
70  rc = m0_bitmap_init(&m0t1fs_client_ep_tmid, M0_NET_LNET_TMID_MAX / 2);
71  if (rc != 0)
72  goto out;
73  m0_bitmap_set(&m0t1fs_client_ep_tmid, 0, true);
74 
75  rc = m0_mdservice_fsync_fop_init(NULL);
76  if (rc != 0)
77  goto bitmap_fini;
78 
79  rc = m0_ioservice_fop_init();
80  if (rc != 0)
81  goto fsync_fini;
82 
83  rc = m0_mdservice_fop_init();
84  if (rc != 0)
85  goto ioservice_fop_fini;
86 
87  rc = m0_ss_process_fops_init();
88  if (rc != 0)
89  goto mdservice_fop_fini;
90 
91  rc = m0t1fs_inode_cache_init();
92  if (rc != 0)
93  goto process_fops_fini;
94 
95  m0_addb2_sys_sm_start(sys);
96  rc = m0_addb2_sys_net_start(sys);
97  if (rc != 0)
98  goto icache_fini;
99 
100  rc = register_filesystem(&m0t1fs_fs_type);
101  if (rc != 0)
102  goto addb2_fini;
103  /*
104  * Limit the number of concurrent parity calculations
105  * to avoid starving other threads (especially LNet) out.
106  *
107  * Note: the exact threshold number may come from configuration
108  * database later where it can be specified per-node.
109  */
110  cpus = (num_online_cpus() / 2) ?: 1;
111  printk(KERN_INFO "motr: max CPUs for parity calcs: %d\n", cpus);
112  m0_semaphore_init(&m0t1fs_cpus_sem, cpus);
113  return M0_RC(0);
114 
115 addb2_fini:
116  m0_addb2_sys_net_stop(sys);
117  m0_addb2_sys_sm_stop(sys);
118 icache_fini:
119  m0t1fs_inode_cache_fini();
120 process_fops_fini:
121  m0_ss_process_fops_fini();
122 mdservice_fop_fini:
123  m0_mdservice_fop_fini();
124 ioservice_fop_fini:
125  m0_ioservice_fop_fini();
126 fsync_fini:
127  m0_mdservice_fsync_fop_fini();
128 bitmap_fini:
129  m0_bitmap_fini(&m0t1fs_client_ep_tmid);
130  m0_mutex_fini(&m0t1fs_mutex);
131 out:
132  return M0_ERR(rc);
133 }

139  M0_ENTRY();

141  sys = m0_addb2_global_get();

144 
145  m0_addb2_sys_net_stop(sys);
146  m0_addb2_sys_sm_stop(sys);
147  m0t1fs_inode_cache_fini();
148  m0_ha_state_fop_fini();
149  m0_mdservice_fop_fini();
150  m0_ioservice_fop_fini();
151  m0_mdservice_fsync_fop_fini();
152  m0_bitmap_fini(&m0t1fs_client_ep_tmid);
153  m0_mutex_fini(&m0t1fs_mutex);
154  M0_LEAVE();
155 }
