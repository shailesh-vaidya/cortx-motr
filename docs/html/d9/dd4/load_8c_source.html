149 static void conf_save_load(void)
150 {
151  struct m0_stob *stob;
152  char linux_location[] = "linuxstob:./__s-conf";
153  struct m0_fid confd_fid = M0_CONFD_FID(1, 2, 12345);
154  int rc;
155  char *str_write;
156  char *str_read;
157  int i;
158 
159  rc = m0_confd_stob_init(&stob, linux_location, &confd_fid);
160  M0_UT_ASSERT(rc == 0);
161 
162  str_write = m0_alloc_aligned(STR_LEN, m0_stob_block_shift(stob));
163  for (i = 0; i < STR_LEN; ++i)
164  str_write[i] = '0' + (i % 10);
165  str_write[STR_LEN - 1] = 0;
166 
167  rc = m0_confd_stob_write(stob, str_write);
168  M0_UT_ASSERT(rc == 0);
169 
170  rc = m0_confd_stob_read(stob, &str_read);
171  M0_UT_ASSERT(rc == 0);
172  M0_UT_ASSERT(strcmp(str_write, str_read) == 0);
173 
174  m0_free_aligned(str_read, STR_LEN, m0_stob_block_shift(stob));
175  m0_free_aligned(str_write, STR_LEN, m0_stob_block_shift(stob));
176 
177  rc = m0_stob_domain_destroy_location(linux_location);
178  M0_UT_ASSERT(rc == 0);
179 
180  m0_confd_stob_fini(stob);
181 }

188 static void conf_save_load_fail(void)
189 {
190  struct m0_stob *stob;
191  char linux_location[] = "linuxstob:./__s-conf";
192  char fail_location[] = "fail_location";
193  struct m0_fid confd_fid = M0_CONFD_FID(1, 2, 12345);
194  int rc;

196  char *str_read;
197  int i;
198 
199  /* Test create fail*/
200  rc = m0_confd_stob_init(&stob, fail_location, &confd_fid);
201  M0_UT_ASSERT(rc == -EINVAL);
202 
203  rc = m0_confd_stob_init(&stob, linux_location, &confd_fid);
204  M0_UT_ASSERT(rc == 0);
205 
206  /* test save fail */
207  str_write = m0_alloc_aligned(STR_LEN, m0_stob_block_shift(stob));
208  for (i = 0; i < STR_LEN; ++i)
209  str_write[i] = '0' + (i % 10);

211 
212  rc = m0_confd_stob_write(stob, str_write);
213  M0_UT_ASSERT(rc == 0);
214 
215  /* test load fail */
216  m0_fi_enable_once("m0_alloc_aligned", "fail_allocation");
217  rc = m0_confd_stob_read(stob, &str_read);

242 static int conf_ut_reqh_init(struct m0_conf_ut_reqh *conf_reqh,
243  const char *ep_addr)
244 {
245  struct m0_net_xprt *xprt = m0_net_xprt_default_get();
246  enum { NR_TMS = 1 };
247  int rc;
248 
249  M0_SET0(conf_reqh);
250  rc = m0_net_domain_init(&conf_reqh->cur_net_dom, xprt);
251  if (rc != 0)
252  return rc;
253 
254  rc = m0_rpc_net_buffer_pool_setup(&conf_reqh->cur_net_dom,
255  &conf_reqh->cur_buf_pool,
256  m0_rpc_bufs_nr(
257  M0_NET_TM_RECV_QUEUE_DEF_LEN,
258  NR_TMS),
259  NR_TMS);
260  if (rc != 0)
261  goto net;
262 
263  rc = M0_REQH_INIT(&conf_reqh->cur_reqh,
264  .rhia_dtm = (void *)1,
265  .rhia_mdstore = (void *)1,
266  .rhia_fid = &g_process_fid);
267  if (rc != 0)
268  goto buf_pool;
269  m0_reqh_start(&conf_reqh->cur_reqh);
270 
271  rc = m0_rpc_machine_init(&conf_reqh->cur_rmachine,
272  &conf_reqh->cur_net_dom, ep_addr,
273  &conf_reqh->cur_reqh, &conf_reqh->cur_buf_pool,
274  M0_BUFFER_ANY_COLOUR,
275  M0_RPC_DEF_MAX_RPC_MSG_SIZE,
276  M0_NET_TM_RECV_QUEUE_DEF_LEN);
277 
278  if (rc == 0) {
279  return 0;
280  }
281 buf_pool:
282  m0_rpc_net_buffer_pool_cleanup(&conf_reqh->cur_buf_pool);
283 net:
284  m0_net_domain_fini(&conf_reqh->cur_net_dom);
285  return rc;
286 }
