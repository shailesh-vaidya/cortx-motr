50 int m0_rpc_server_start(struct m0_rpc_server_ctx *sctx)
51 {
52  int rc;
53 
54  M0_ENTRY("server_ctx: %p", sctx);
55  M0_PRE(sctx->rsx_argv != NULL && sctx->rsx_argc > 0);
56 
57  /* Open error log file */
58  sctx->rsx_log_file = fopen(sctx->rsx_log_file_name, "w+");
59  if (sctx->rsx_log_file == NULL)
60  return M0_ERR_INFO(errno, "Open of error log file");
61 
62  rc = m0_cs_init(&sctx->rsx_motr_ctx, sctx->rsx_xprts,
63  sctx->rsx_xprts_nr, sctx->rsx_log_file, true);
64  M0_LOG(M0_DEBUG, "cs_init: rc=%d", rc);
65  if (rc != 0)
66  goto fclose;
67 
68  rc = m0_cs_setup_env(&sctx->rsx_motr_ctx, sctx->rsx_argc,
69  sctx->rsx_argv);
70  M0_LOG(M0_DEBUG, "cs_setup_env: rc=%d", rc);
71  if (rc != 0)
72  goto error;
73 
74  rc = m0_cs_start(&sctx->rsx_motr_ctx);
75  if (rc == 0)
76  return M0_RC(0);
77 
78 error:
79  m0_cs_fini(&sctx->rsx_motr_ctx);
80 fclose:
81  fclose(sctx->rsx_log_file);
82  return M0_RC(rc);
83 }

102 M0_INTERNAL int m0_rpc_client_connect(struct m0_rpc_conn *conn,
103  struct m0_rpc_session *session,
104  struct m0_rpc_machine *rpc_mach,
105  const char *remote_addr,
106  struct m0_fid *svc_fid,
107  uint64_t max_rpcs_in_flight,
108  m0_time_t abs_timeout)
109 {
110  struct m0_net_end_point *ep;
111  int rc;
112 
113  M0_ENTRY("conn=%p session=%p rpc_mach=%p remote_addr=%s",
114  conn, session, rpc_mach, remote_addr);
115 
116  rc = m0_net_end_point_create(&ep, &rpc_mach->rm_tm, remote_addr);
117  if (rc != 0)
118  return M0_RC(rc);
119 
120  rc = m0_rpc_conn_create(conn, svc_fid, ep, rpc_mach, max_rpcs_in_flight,
121  abs_timeout);
122  m0_net_end_point_put(ep);
123  if (rc != 0)
124  return M0_RC(rc);
125 
126  rc = m0_rpc_session_create(session, conn, abs_timeout);
127  if (rc != 0)
128  (void)m0_rpc_conn_destroy(conn, abs_timeout);
129 
130  return M0_RC(rc);
131 }

160 int m0_rpc_client_start(struct m0_rpc_client_ctx *cctx)
161 {
162  enum { NR_TM = 1 }; /* number of TMs */
163  int rc;
164 
165  M0_ENTRY("client_ctx: %p", cctx);
166 
167  if (cctx->rcx_recv_queue_min_length == 0)
168  cctx->rcx_recv_queue_min_length = M0_NET_TM_RECV_QUEUE_DEF_LEN;
169 
170  rc = m0_rpc_net_buffer_pool_setup(
171  cctx->rcx_net_dom, &cctx->rcx_buffer_pool,
172  m0_rpc_bufs_nr(cctx->rcx_recv_queue_min_length, NR_TM),
173  NR_TM);
174  if (rc != 0)
175  return M0_RC(rc);
176 
177  M0_SET0(&cctx->rcx_reqh);
178  rc = M0_REQH_INIT(&cctx->rcx_reqh,
179  .rhia_dtm = (void*)1,
180  .rhia_mdstore = (void*)1,
181  .rhia_fid = cctx->rcx_fid,
182  );
183  if (rc != 0)
184  goto err;
185  m0_reqh_start(&cctx->rcx_reqh);
186 
187  rc = m0_rpc_machine_init(&cctx->rcx_rpc_machine,
188  cctx->rcx_net_dom, cctx->rcx_local_addr,
189  &cctx->rcx_reqh,
190  &cctx->rcx_buffer_pool, M0_BUFFER_ANY_COLOUR,
191  cctx->rcx_max_rpc_msg_size,
192  cctx->rcx_recv_queue_min_length);
193  if (rc != 0) {
194  m0_reqh_services_terminate(&cctx->rcx_reqh);
195  m0_reqh_fini(&cctx->rcx_reqh);
196  goto err;
197  }
198 
199  rc = m0_rpc_client_connect(&cctx->rcx_connection, &cctx->rcx_session,
200  &cctx->rcx_rpc_machine,
201  cctx->rcx_remote_addr, NULL,
202  cctx->rcx_max_rpcs_in_flight,
203  cctx->rcx_abs_timeout?:M0_TIME_NEVER);
204  if (rc == 0)
205  return M0_RC(0);
206 
207  m0_reqh_services_terminate(&cctx->rcx_reqh);
208  m0_rpc_machine_fini(&cctx->rcx_rpc_machine);
209  m0_reqh_fini(&cctx->rcx_reqh);
210 err:
211  m0_rpc_net_buffer_pool_cleanup(&cctx->rcx_buffer_pool);
212  return M0_RC(rc);
213 }

223 int m0_rpc_client_stop_stats(struct m0_rpc_client_ctx *cctx,
224  void (*printout)(struct m0_rpc_machine *))
225 {
226  int rc0;
227  int rc1;
228 
229  M0_ENTRY("client_ctx: %p", cctx);
230 
231  rc0 = m0_rpc_session_destroy(&cctx->rcx_session, M0_TIME_NEVER);
232  if (rc0 != 0)
233  M0_LOG(M0_ERROR, "Failed to terminate session %d", rc0);
234 
235  rc1 = m0_rpc_conn_destroy(&cctx->rcx_connection, M0_TIME_NEVER);
236  if (rc1 != 0)
237  M0_LOG(M0_ERROR, "Failed to terminate connection %d", rc1);
238  if (printout != NULL)
239  printout(&cctx->rcx_rpc_machine);
240  m0_reqh_services_terminate(&cctx->rcx_reqh);
241  m0_rpc_machine_fini(&cctx->rcx_rpc_machine);
242  m0_reqh_fini(&cctx->rcx_reqh);
243  m0_rpc_net_buffer_pool_cleanup(&cctx->rcx_buffer_pool);
244 
245  return M0_RC(rc0 ?: rc1);
246 }

249 int m0_rpc_post_with_timeout_sync(struct m0_fop *fop,
250  struct m0_rpc_session *session,
251  const struct m0_rpc_item_ops *ri_ops,
252  m0_time_t deadline,
253  m0_time_t timeout)
254 {
255  struct m0_rpc_item *item;
256  int rc;
257 
258  M0_ENTRY("fop: %p, session: %p", fop, session);
259  M0_PRE(session != NULL);
260 
261  item = &fop->f_item;
262  item->ri_ops = ri_ops;
263  item->ri_session = session;
264  item->ri_prio = M0_RPC_ITEM_PRIO_MID;
265  item->ri_deadline = deadline;
266 
267  /* Set default value if user hasn't set anything else */
268  if (item->ri_nr_sent_max == ~(uint64_t)0)
269  item->ri_nr_sent_max = M0_RPCLIB_MAX_RETRIES;
270 
271  /*
272  * Add a ref so that the item does not get vanished, say due to
273  * session cancellation, while it is being waited for.
274  */
275  m0_fop_get(fop);
276  rc = m0_rpc_post(item) ?:
277  m0_rpc_item_wait_for_reply(item, timeout) ?:
278  m0_rpc_item_generic_reply_rc(item->ri_reply);
279  m0_fop_put_lock(fop);
280  return M0_RC(rc);
281 }
