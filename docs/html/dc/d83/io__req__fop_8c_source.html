134 static void application_attribute_copy(struct m0_indexvec *rep_ivec,
135  struct target_ioreq *ti,
136  struct m0_op_io *ioo,
137  struct m0_buf *buf)
138 {
139  uint32_t unit_size;
140  uint32_t off;
141  uint32_t cs_sz;
142  m0_bindex_t rep_index;
143  m0_bindex_t ti_cob_index;
144  m0_bindex_t ti_goff_index;
145  struct m0_ivec_cursor rep_cursor;
146  struct m0_ivec_cursor ti_cob_cursor;
147  struct m0_ivec_cursor ti_goff_cursor;
148  struct m0_indexvec *ti_ivec = &ti->ti_ivec;
149  struct m0_indexvec *ti_goff_ivec = &ti->ti_goff_ivec;
150  void *dst;
151  void *src;
152 
153  if (!m0__obj_is_di_enabled(ioo)) {
154  return;
155  }
156  src = buf->b_addr;
157 
158  if (!buf->b_nob) {
159  /* Return as no checksum is present */
160  return;
161  }
162 
163  unit_size = m0_obj_layout_id_to_unit_size(m0__obj_lid(ioo->ioo_obj));
164  cs_sz = ioo->ioo_attr.ov_vec.v_count[0];
165 
166  m0_ivec_cursor_init(&rep_cursor, rep_ivec);
167  m0_ivec_cursor_init(&ti_cob_cursor, ti_ivec);
168  m0_ivec_cursor_init(&ti_goff_cursor, ti_goff_ivec);
169 
170  rep_index = m0_ivec_cursor_index(&rep_cursor);
171  ti_cob_index = m0_ivec_cursor_index(&ti_cob_cursor);
172  ti_goff_index = m0_ivec_cursor_index(&ti_goff_cursor);
173 
174  /* Move rep_cursor on unit boundary */
175  off = rep_index % unit_size;
176  if (off) {
177  if (!m0_ivec_cursor_move(&rep_cursor, unit_size - off))
178  rep_index = m0_ivec_cursor_index(&rep_cursor);
179  else
180  return;
181  }
182  off = ti_cob_index % unit_size;
183  if (off != 0) {
184  if (!m0_ivec_cursor_move(&ti_cob_cursor, unit_size - off)) {
185  ti_cob_index = m0_ivec_cursor_index(&ti_cob_cursor);
186  }
187  }
188  off = ti_goff_index % unit_size;
189  if (off != 0) {
190  if (!m0_ivec_cursor_move(&ti_goff_cursor, unit_size - off)) {
191  ti_goff_index = m0_ivec_cursor_index(&ti_goff_cursor);
192  }
193  }
194  M0_ASSERT(ti_cob_index <= rep_index);
195 
208  do {
209  rep_index = m0_ivec_cursor_index(&rep_cursor);
210  while (ti_cob_index != rep_index) {
211  if (m0_ivec_cursor_move(&ti_cob_cursor, unit_size) ||
212  m0_ivec_cursor_move(&ti_goff_cursor, unit_size)) {
213  M0_ASSERT(0);
214  }
215  ti_cob_index = m0_ivec_cursor_index(&ti_cob_cursor);
216  ti_goff_index = m0_ivec_cursor_index(&ti_goff_cursor);
217  }
218 
219  /* GOB offset should be in span of application provided GOB extent */
220  M0_ASSERT(ti_goff_index <=
221  (ioo->ioo_ext.iv_index[ioo->ioo_ext.iv_vec.v_nr-1] +
222  ioo->ioo_ext.iv_vec.v_count[ioo->ioo_ext.iv_vec.v_nr-1]));
223 
224  dst = m0_extent_vec_get_checksum_addr(&ioo->ioo_attr,
225  ti_goff_index,
226  &ioo->ioo_ext,
227  unit_size, cs_sz);
228  M0_ASSERT(dst != NULL);
229  memcpy(dst, src, cs_sz);
230  src = (char *)src + cs_sz;
231 
232  /* Source is m0_buf and we have to copy all the checksum one at a time */
233  M0_ASSERT(src <= (buf->b_addr + buf->b_nob));
234 
235  } while (!m0_ivec_cursor_move(&rep_cursor, unit_size));
236 }

246 static void io_bottom_half(struct m0_sm_group *grp, struct m0_sm_ast *ast)
247 {
248  int rc;
249  uint64_t actual_bytes = 0;
250  struct m0_client *instance;
251  struct m0_op *op;
252  struct m0_op_io *ioo;
253  struct nw_xfer_request *xfer;
254  struct m0_io_fop *iofop;
255  struct ioreq_fop *irfop;
256  struct target_ioreq *tioreq;
257  struct m0_fop *reply_fop = NULL;
258  struct m0_rpc_item *req_item;
259  struct m0_rpc_item *reply_item;
260  struct m0_rpc_bulk *rbulk;
261  struct m0_fop_cob_rw_reply *rw_reply;
262  struct m0_indexvec rep_attr_ivec;
263  struct m0_fop_generic_reply *gen_rep;
264  struct m0_fop_cob_rw *rwfop;
265 
266  M0_ENTRY("sm_group %p sm_ast %p", grp, ast);
267 
268  M0_PRE(grp != NULL);
269  M0_PRE(ast != NULL);
270 
271  irfop = bob_of(ast, struct ioreq_fop, irf_ast, &iofop_bobtype);
272  iofop = &irfop->irf_iofop;
273  tioreq = irfop->irf_tioreq;
274  xfer = tioreq->ti_nwxfer;
275 
276  ioo = bob_of(xfer, struct m0_op_io, ioo_nwxfer, &ioo_bobtype);
277  op = &ioo->ioo_oo.oo_oc.oc_op;
278  instance = m0__op_instance(op);
279  M0_PRE(instance != NULL);
280  M0_PRE(M0_IN(irfop->irf_pattr, (PA_DATA, PA_PARITY)));
281  M0_PRE(M0_IN(ioreq_sm_state(ioo),
282  (IRS_READING, IRS_WRITING,
283  IRS_DEGRADED_READING, IRS_DEGRADED_WRITING,
284  IRS_FAILED)));
285 
286  /* Check errors in rpc items of an IO reqest and its reply. */
287  rbulk = &iofop->if_rbulk;
288  req_item = &iofop->if_fop.f_item;
289  rwfop = io_rw_get(&iofop->if_fop);
290  reply_item = req_item->ri_reply;
291  rc = req_item->ri_error;
292  if (reply_item != NULL) {
293  reply_fop = m0_rpc_item_to_fop(reply_item);
294  rc = rc?: m0_rpc_item_generic_reply_rc(reply_item);
295  }
296  if (rc < 0 || reply_item == NULL) {
297  M0_ASSERT(ergo(reply_item == NULL, rc != 0));
298  M0_LOG(M0_ERROR, "[%p] rpc item %p rc=%d", ioo, req_item, rc);
299  goto ref_dec;
300  }
301  M0_ASSERT(!m0_rpc_item_is_generic_reply_fop(reply_item));
302  M0_ASSERT(m0_is_io_fop_rep(reply_fop));
303 
304  /* Check errors in an IO request's reply. */
305  gen_rep = m0_fop_data(m0_rpc_item_to_fop(reply_item));
306  rw_reply = io_rw_rep_get(reply_fop);
307 
308  /*
309  * Copy attributes to client if reply received from read operation
310  * Skipping attribute_copy() if cksum validation is not allowed.
311  */
312  if (m0_is_read_rep(reply_fop) && op->op_code == M0_OC_READ &&
313  m0__obj_is_cksum_validation_allowed(ioo)) {
314  m0_indexvec_wire2mem(&rwfop->crw_ivec,
315  rwfop->crw_ivec.ci_nr, 0,
316  &rep_attr_ivec);
317 
318  application_attribute_copy(&rep_attr_ivec, tioreq, ioo,
319  &rw_reply->rwr_di_data_cksum);
320 
321  m0_indexvec_free(&rep_attr_ivec);
322  }
323  ioo->ioo_sns_state = rw_reply->rwr_repair_done;
324  M0_LOG(M0_DEBUG, "[%p] item %p[%u], reply received = %d, "
325  "sns state = %d", ioo, req_item,
326  req_item->ri_type->rit_opcode, rc, ioo->ioo_sns_state);
327  actual_bytes = rw_reply->rwr_count;
328  rc = gen_rep->gr_rc;
329  rc = rc ?: rw_reply->rwr_rc;
330  irfop->irf_reply_rc = rc;
331 
332  /* Update pending transaction number */
333  sync_record_update(
334  m0_reqh_service_ctx_from_session(reply_item->ri_session),
335  &ioo->ioo_obj->ob_entity, op, &rw_reply->rwr_mod_rep.fmr_remid);
336 
337 ref_dec:
338  /* For whatever reason, io didn't complete successfully.
339  * Reduce expected read bulk count */
340  if (rc < 0 && m0_is_read_fop(&iofop->if_fop))
341  m0_atomic64_sub(&xfer->nxr_rdbulk_nr,
342  m0_rpc_bulk_buf_length(rbulk));
343 
344  /* Propogate the error up as many stashed-rc layers as we can */
345  if (tioreq->ti_rc == 0)
346  tioreq->ti_rc = rc;
347 
348  /*
349  * Note: this is not necessary mean that this is 'real' error in the
350  * case of CROW is used (object is only created when it is first
351  * write)
352  */
353  if (xfer->nxr_rc == 0 && rc != 0) {
354  xfer->nxr_rc = rc;
355 
356 #define LOGMSG(ioo, rc, tireq) "ioo=%p from=%s rc=%d ti_rc=%d @"FID_F,\
357  (ioo), m0_rpc_conn_addr((tioreq)->ti_session->s_conn),\
358  (rc), (tioreq)->ti_rc, FID_P(&(tioreq)->ti_fid)
359 
360  if (rc == -ENOENT) /* normal for CROW */
361  M0_LOG(M0_DEBUG, LOGMSG(ioo, rc, tireq));
362  else
363  M0_LOG(M0_ERROR, LOGMSG(ioo, rc, tireq));
364 #undef LOGMSG
365  }
366 
367  if (irfop->irf_pattr == PA_DATA)
368  tioreq->ti_databytes += rbulk->rb_bytes;
369  else
370  tioreq->ti_parbytes += rbulk->rb_bytes;
371 
372  M0_LOG(M0_INFO, "[%p] fop %p, Returned no of bytes = %llu, "
373  "expected = %llu",
374  ioo, &iofop->if_fop, (unsigned long long)actual_bytes,
375  (unsigned long long)rbulk->rb_bytes);
376 
377  /* Drops reference on reply fop. */
378  m0_fop_put0_lock(&iofop->if_fop);
379  m0_fop_put0_lock(reply_fop);
380  m0_atomic64_dec(&instance->m0c_pending_io_nr);
381 
382  m0_mutex_lock(&xfer->nxr_lock);
383  m0_atomic64_dec(&xfer->nxr_iofop_nr);
384  if (should_ioreq_sm_complete(ioo)) {
385  m0_sm_state_set(&ioo->ioo_sm,
386  (M0_IN(ioreq_sm_state(ioo),
387  (IRS_READING, IRS_DEGRADED_READING)) ?
388  IRS_READ_COMPLETE:IRS_WRITE_COMPLETE));
389 
390  /* post an ast to run iosm_handle_executed */
391  ioo->ioo_ast.sa_cb = ioo->ioo_ops->iro_iosm_handle_executed;
392  m0_sm_ast_post(ioo->ioo_oo.oo_sm_grp, &ioo->ioo_ast);
393  }
394  m0_mutex_unlock(&xfer->nxr_lock);
395 
396  M0_LOG(M0_DEBUG, "[%p] irfop=%p bulk=%p "FID_F
397  " Pending fops = %"PRIu64" bulk=%"PRIu64,
398  ioo, irfop, rbulk, FID_P(&tioreq->ti_fid),
399  m0_atomic64_get(&xfer->nxr_iofop_nr),
400  m0_atomic64_get(&xfer->nxr_rdbulk_nr));
401 
402  M0_LEAVE();
403 }

413 {
414  struct m0_fop *fop;
415  struct m0_fop *rep_fop;
416  struct m0_io_fop *iofop;
417  struct ioreq_fop *reqfop;
418  struct m0_op_io *ioo;

421  M0_ENTRY("rpc_item %p", item);
422 
423  fop = m0_rpc_item_to_fop(item);
424  iofop = M0_AMB(iofop, fop, if_fop);
425  reqfop = bob_of(iofop, struct ioreq_fop, irf_iofop, &iofop_bobtype);
426  ioo = bob_of(reqfop->irf_tioreq->ti_nwxfer, struct m0_op_io,
427  ioo_nwxfer, &ioo_bobtype);
428  /*
429  * NOTE: RPC errors are handled in io_bottom_half(), which is called
430  * by reqfop->irf_ast.
431  */
432 
433  /*
434  * Acquires a reference on IO reply fop since its contents
435  * are needed for policy decisions in io_bottom_half().
436  * io_bottom_half() takes care of releasing the reference.
437  */

441  }
442  M0_LOG(M0_INFO, "ioreq_fop %p, target_ioreq %p io_request %p",
443  reqfop, reqfop->irf_tioreq, ioo);
444 
445  m0_fop_get(&reqfop->irf_iofop.if_fop);
446  m0_sm_ast_post(ioo->ioo_sm.sm_grp, &reqfop->irf_ast);
447 
448  M0_LEAVE();
449 }

451 static void ioreq_cc_bottom_half(struct m0_sm_group *grp,
452  struct m0_sm_ast *ast)
453 {
454  struct nw_xfer_request *xfer;
455  struct target_ioreq *ti;
456  struct cc_req_fop *cc_fop;
457  struct m0_op *op;
458  struct m0_op_io *ioo;
459  struct m0_fop_cob_op_reply *reply;
460  struct m0_fop *reply_fop = NULL;
461  struct m0_rpc_item *req_item;
462  struct m0_rpc_item *reply_item;
463  struct m0_be_tx_remid *remid = NULL;
464  int rc;
465 
466  ti = (struct target_ioreq *)ast->sa_datum;
467  ioo = bob_of(ti->ti_nwxfer, struct m0_op_io, ioo_nwxfer,
468  &ioo_bobtype);
469  op = &ioo->ioo_oo.oo_oc.oc_op;
470  xfer = ti->ti_nwxfer;
471  cc_fop = &ti->ti_cc_fop;
472  req_item = &cc_fop->crf_fop.f_item;
473  reply_item = req_item->ri_reply;
474  rc = req_item->ri_error;
475  if (reply_item != NULL) {
476  reply_fop = m0_rpc_item_to_fop(reply_item);
477  rc = rc ?: m0_rpc_item_generic_reply_rc(reply_item);
478  }
479  if (rc < 0 || reply_item == NULL) {
480  M0_ASSERT(ergo(reply_item == NULL, rc != 0));
481  goto ref_dec;
482  }
483  reply = m0_fop_data(m0_rpc_item_to_fop(reply_item));
484  /*
485  * Ignoring the case when an attempt is made to create a cob on target
486  * where previous IO had created it.
487  */
488  rc = rc ? M0_IN(reply->cor_rc, (0, -EEXIST)) ? 0 : reply->cor_rc : 0;
489 
490  remid = &reply->cor_common.cor_mod_rep.fmr_remid;
491 
492  /* Update pending transaction number */
493  sync_record_update(
494  m0_reqh_service_ctx_from_session(reply_item->ri_session),
495  &ioo->ioo_obj->ob_entity, op, remid);
496  /*
497  * @todo: in case confd is updated, a check is necessary similar to
498  * that present in m0t1fs. See
499  * m0t1fs/linux_kernel/file.c::io_bottom_half().
500  */
501 
502 ref_dec:
503  if (ti->ti_rc == 0 && rc != 0)
504  ti->ti_rc = rc;
505  if (xfer->nxr_rc == 0 && rc != 0)
506  xfer->nxr_rc = rc;
507  if (ioo->ioo_rc == 0 && rc != 0)
508  ioo->ioo_rc = rc;
509  m0_fop_put0_lock(&cc_fop->crf_fop);
510  if (reply_fop != NULL)
511  m0_fop_put0_lock(reply_fop);
512  m0_mutex_lock(&xfer->nxr_lock);
513  m0_atomic64_dec(&xfer->nxr_ccfop_nr);
514  if (should_ioreq_sm_complete(ioo)) {
515  if (ioreq_sm_state(ioo) == IRS_TRUNCATE)
516  m0_sm_state_set(&ioo->ioo_sm, IRS_TRUNCATE_COMPLETE);
517  else
518  m0_sm_state_set(&ioo->ioo_sm, IRS_WRITE_COMPLETE);
519  ioo->ioo_ast.sa_cb = ioo->ioo_ops->iro_iosm_handle_executed;
520  m0_sm_ast_post(ioo->ioo_oo.oo_sm_grp, &ioo->ioo_ast);
521  }
522  m0_mutex_unlock(&xfer->nxr_lock);
523 }

879 static void ioreq_fop_release(struct m0_ref *ref)
880 {
881  struct m0_fop *fop;
882  struct m0_io_fop *iofop;
883  struct ioreq_fop *reqfop;
884  struct m0_fop_cob_rw *rwfop;
885  struct m0_rpc_bulk *rbulk;
886  struct nw_xfer_request *xfer;
887  struct m0_rpc_machine *rmach;
888  struct m0_rpc_item *item;
889

891  M0_PRE(ref != NULL);
892 
893  fop = M0_AMB(fop, ref, f_ref);
894  rmach = m0_fop_rpc_machine(fop);

896  reqfop = bob_of(iofop, struct ioreq_fop, irf_iofop, &iofop_bobtype);
897  rbulk = &iofop->if_rbulk;
898  xfer = reqfop->irf_tioreq->ti_nwxfer;
899  item = &fop->f_item;
900 
901  /*
902  * Release the net buffers if rpc bulk object is still dirty.
903  * And wait on channel till all net buffers are deleted from
904  * transfer machine.
905  */
906  m0_mutex_lock(&xfer->nxr_lock);
907  m0_mutex_lock(&rbulk->rb_mutex);
908  if (!m0_tlist_is_empty(&rpcbulk_tl, &rbulk->rb_buflist)) {
909  struct m0_clink clink;
910  size_t buf_nr;
911  size_t non_queued_buf_nr;

913  m0_clink_init(&clink, NULL);
914  m0_clink_add(&rbulk->rb_chan, &clink);
915  buf_nr = rpcbulk_tlist_length(&rbulk->rb_buflist);
916  non_queued_buf_nr = m0_rpc_bulk_store_del_unqueued(rbulk);
917  m0_mutex_unlock(&rbulk->rb_mutex);
918 
919  m0_rpc_bulk_store_del(rbulk);
920  M0_LOG(M0_DEBUG, "fop %p, %p[%u], bulk %p, buf_nr %llu, "
921  "non_queued_buf_nr %llu", &iofop->if_fop, item,
922  item->ri_type->rit_opcode, rbulk,
923  (unsigned long long)buf_nr,
924  (unsigned long long)non_queued_buf_nr);
925 
926  if (m0_is_read_fop(&iofop->if_fop))
927  m0_atomic64_sub(&xfer->nxr_rdbulk_nr,
928  non_queued_buf_nr);
929  if (item->ri_sm.sm_state == M0_RPC_ITEM_UNINITIALISED)
930  /* rio_replied() is not invoked for this item. */
931  m0_atomic64_dec(&xfer->nxr_iofop_nr);
932  m0_mutex_unlock(&xfer->nxr_lock);
933 
934  /*
935  * If there were some queued net bufs which had to be deleted,
936  * then it is required to wait for their callbacks.
937  */
938  if (buf_nr > non_queued_buf_nr) {
939  /*
940  * rpc_machine_lock may be needed from nlx_tm_ev_worker
941  * thread, which is going to wake us up. So we should
942  * release it to avoid deadlock.
943  */
944  m0_rpc_machine_unlock(rmach);
945  m0_chan_wait(&clink);
946  m0_rpc_machine_lock(rmach);
947  }
948  m0_clink_del_lock(&clink);
949  m0_clink_fini(&clink);
950  } else {
951  m0_mutex_unlock(&rbulk->rb_mutex);
952  m0_mutex_unlock(&xfer->nxr_lock);
953  }
954  M0_ASSERT(m0_rpc_bulk_is_empty(rbulk));
955 
956  rwfop = io_rw_get(&iofop->if_fop);
957  M0_ASSERT(rwfop != NULL);
958  ioreq_fop_fini(reqfop);
959  /* see ioreq_fop_fini(). */
960  ioreq_fop_bob_fini(reqfop);
961  m0_io_fop_fini(iofop);
962  m0_free(reqfop);
963 
964  M0_LEAVE();
965 }
