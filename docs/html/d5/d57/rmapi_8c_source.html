46 static void credits_api_test (void)
47 {
48  int rc;
49 
50  rings_utdata_ops_set(&rm_test_data);
51  rm_utdata_init(&rm_test_data, OBJ_OWNER);
52 
53  /* 1. Test m0_rm_incoming_init() */
54  m0_rm_incoming_init(&rm_test_data.rd_in, rm_test_data.rd_owner,
55  M0_RIT_LOCAL, RIP_NONE, RIF_LOCAL_WAIT);
56  M0_UT_ASSERT(rm_test_data.rd_in.rin_sm.sm_state == RI_INITIALISED);
57  M0_UT_ASSERT(rm_test_data.rd_in.rin_type == M0_RIT_LOCAL);
58  M0_UT_ASSERT(rm_test_data.rd_in.rin_policy == RIP_NONE);
59  M0_UT_ASSERT(rm_test_data.rd_in.rin_flags == RIF_LOCAL_WAIT);
60  M0_UT_ASSERT(rm_test_data.rd_in.rin_want.cr_datum == 0);
61  M0_UT_ASSERT(rm_test_data.rd_in.rin_rc == 0);
62 
63  /* 2. Test m0_rm_credit_init */
64  m0_rm_credit_init(&rm_test_data.rd_credit, rm_test_data.rd_owner);
65  M0_UT_ASSERT(rm_test_data.rd_credit.cr_datum == 0);
66  M0_UT_ASSERT(rm_test_data.rd_credit.cr_owner == rm_test_data.rd_owner);
67 
68  /* 3. Test m0_rm_owner_selfadd. Test memory failure */
69  rm_test_data.rd_credit.cr_datum = ALLRINGS;
70  m0_fi_enable_once("rings_credit_copy", "fail_copy");
71  rc = m0_rm_owner_selfadd(rm_test_data.rd_owner,
72  &rm_test_data.rd_credit);
73  M0_UT_ASSERT(rc == -ENOMEM);
74 
75  /* 4. Test m0_rm_owner_selfadd. Indirectly tests m0_rm_loan_init */
76  rc = m0_rm_owner_selfadd(rm_test_data.rd_owner,
77  &rm_test_data.rd_credit);
78  M0_UT_ASSERT(rc == 0);
79  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
80  &rm_test_data.rd_owner->ro_borrowed));
81  M0_UT_ASSERT(!m0_rm_ur_tlist_is_empty(
82  &rm_test_data.rd_owner->ro_owned[OWOS_CACHED]));
83 
84  /*
85  * 5. Test m0_rm_credit_get for memory failure.
86  */
87  m0_rm_credit_init(&rm_test_data.rd_in.rin_want, rm_test_data.rd_owner);
88  rm_test_data.rd_in.rin_want.cr_datum = rm_test_data.rd_credit.cr_datum;
89  rm_test_data.rd_in.rin_ops = &rings_incoming_ops;
90  m0_fi_enable_once("rings_credit_copy", "fail_copy");
91  m0_rm_credit_get(&rm_test_data.rd_in);
92  M0_UT_ASSERT(rm_test_data.rd_in.rin_rc == -ENOMEM);
93  M0_UT_ASSERT(rm_test_data.rd_in.rin_sm.sm_state == RI_FAILURE);
94  /* Test m0_rm_incoming_fini */
95  m0_rm_incoming_fini(&rm_test_data.rd_in);
96 
97  /*
98  * 6. Test m0_rm_credit_get - Success case.
99  * Indirectly tests owner_balance, incoming_check, incoming_check_with,
100  * incoming_complete, m0_pin_pin_add.
101  */
102  m0_rm_incoming_init(&rm_test_data.rd_in, rm_test_data.rd_owner,
103  M0_RIT_LOCAL, RIP_NONE, RIF_LOCAL_WAIT);
104  rm_test_data.rd_in.rin_want.cr_datum = rm_test_data.rd_credit.cr_datum;
105  rm_test_data.rd_in.rin_ops = &rings_incoming_ops;
106  m0_rm_credit_get(&rm_test_data.rd_in);
107  M0_UT_ASSERT(rm_test_data.rd_in.rin_rc == 0);
108  M0_UT_ASSERT(rm_test_data.rd_in.rin_sm.sm_state == RI_SUCCESS);
109 
110  /* Test m0_rm_credit_put. Indirectly tests incoming_release, pin_del */
111  m0_rm_credit_put(&rm_test_data.rd_in);
112 
113  /* Test m0_rm_incoming_fini */
114  m0_rm_incoming_fini(&rm_test_data.rd_in);
115 
116  /*
117  * 7. Test m0_rm_credit_get(), incorrect owner state.
118  */
119  m0_rm_owner_windup(rm_test_data.rd_owner);
120  rc = m0_rm_owner_timedwait(rm_test_data.rd_owner, M0_BITS(ROS_FINAL),
121  M0_TIME_NEVER);
122  M0_ASSERT(rc == 0);
123  m0_rm_incoming_init(&rm_test_data.rd_in, rm_test_data.rd_owner,
124  M0_RIT_LOCAL, RIP_NONE, RIF_LOCAL_WAIT);
125  rm_test_data.rd_in.rin_want.cr_datum = rm_test_data.rd_credit.cr_datum;
126  rm_test_data.rd_in.rin_ops = &rings_incoming_ops;
127  m0_rm_credit_get(&rm_test_data.rd_in);
128  M0_UT_ASSERT(rm_test_data.rd_in.rin_rc == -ENODEV);
129  M0_UT_ASSERT(rm_test_data.rd_in.rin_sm.sm_state == RI_FAILURE);
130  m0_rm_incoming_fini(&rm_test_data.rd_in);
131 
132  rm_utdata_fini(&rm_test_data, OBJ_OWNER);
133 }

137  rings_utdata_ops_set(&rm_test_data);
138 
139  /*
140  * 1. Test m0_rm_owner_init
141  * Indirectly tests resource_get(), owner_internal_init(),

151  M0_UT_ASSERT(rm_test_data.rd_owner->ro_sm.sm_state == ROS_FINAL);
152  M0_UT_ASSERT(rm_test_data.rd_owner->ro_resource == rm_test_data.rd_res);
153  M0_UT_ASSERT(rm_test_data.rd_res->r_ref == 1);
154 
155  /* 3. Test m0_rm_owner_fini. Indirectly tests resource_put(). */
156  m0_rm_owner_fini(rm_test_data.rd_owner);
157  M0_UT_ASSERT(rm_test_data.rd_owner->ro_sm.sm_state == ROS_FINAL);
158  M0_UT_ASSERT(rm_test_data.rd_owner->ro_creditor == NULL);
159  M0_UT_ASSERT(rm_test_data.rd_owner->ro_resource == NULL);
160  M0_UT_ASSERT(rm_test_data.rd_res->r_ref == 0);
161 
162  m0_free0(&rm_test_data.rd_owner);
163  rm_utdata_fini(&rm_test_data, OBJ_RES);
164 }

166 static void res_api_test(void)
167 {

172  m0_mutex_lock(&rm_test_data.rd_rt->rt_lock);

176  m0_mutex_unlock(&rm_test_data.rd_rt->rt_lock);
177 
178  M0_UT_ASSERT(rm_test_data.rd_res->r_type == rm_test_data.rd_rt);
179 
180  /* 2. Test m0_rm_resource_del */
181  rm_test_data.rd_ops->resource_unset(&rm_test_data);
182 
183  m0_mutex_lock(&rm_test_data.rd_rt->rt_lock);
184  M0_UT_ASSERT(rm_test_data.rd_rt->rt_nr_resources == 0);
185  M0_UT_ASSERT(res_tlist_is_empty(&rm_test_data.rd_rt->rt_resources));
186  m0_mutex_unlock(&rm_test_data.rd_rt->rt_lock);
187 
188  rm_utdata_fini(&rm_test_data, OBJ_RES_TYPE);
