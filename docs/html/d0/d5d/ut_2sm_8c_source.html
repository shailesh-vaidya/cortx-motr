286 static void timeout(void)
287 {
288  struct m0_sm_timeout t0;
289  struct m0_sm_timeout t1;
290  const long delta = M0_TIME_ONE_SECOND/100;
291  int result;
292 
293  result = M0_THREAD_INIT(&ath, int, NULL, &ast_thread, 0, "ast_thread");
294  M0_UT_ASSERT(result == 0);
295 
296  m0_sm_group_lock(&G);
297  M0_SET0(&m);
298  m0_sm_init(&m, &tmo_sm_conf, S_TMO_INITIAL, &G);
299 
300  /* check that timeout initialisation and finalisation work. */
301  m0_sm_timeout_init(&t1);
302  m0_sm_timeout_fini(&t1);
303 
304  m0_sm_timeout_init(&t0);
305 
306  /* check that timeout works */
307  result = m0_sm_timeout_arm(&m, &t0, m0_time_from_now(0, delta),
308  S_TMO_0, 0);
309  M0_UT_ASSERT(result == 0);
310 
311  result = m0_sm_timedwait(&m, ~M0_BITS(S_TMO_INITIAL), M0_TIME_NEVER);
312  M0_UT_ASSERT(result == 0);
313  M0_UT_ASSERT(m.sm_state == S_TMO_0);
314 
315  m0_sm_timeout_fini(&t0);
316  m0_sm_timeout_init(&t1);
317 
318  /* check that state transition cancels the timeout */
319  result = m0_sm_timeout_arm(&m, &t1, m0_time_from_now(0, delta),
320  S_TMO_1, 0);
321  M0_UT_ASSERT(result == 0);
322 
323  m0_sm_state_set(&m, S_TMO_2);
324  M0_UT_ASSERT(m.sm_state == S_TMO_2);
325 
326  result = m0_sm_timedwait(&m, ~M0_BITS(S_TMO_2), m0_time_from_now(0,
327  2 * delta));
328  M0_UT_ASSERT(result == -ETIMEDOUT);
329  M0_UT_ASSERT(m.sm_state == S_TMO_2);
330 
331  m0_sm_timeout_fini(&t1);
332  m0_sm_timeout_init(&t1);
333 
334  /* check that timeout with a bitmask is not cancelled by a state
335  transition */
336  m0_sm_state_set(&m, S_TMO_0);
337  result = m0_sm_timeout_arm(&m, &t1, m0_time_from_now(0, delta), S_TMO_2,
338  M0_BITS(S_TMO_1));
339  M0_UT_ASSERT(result == 0);
340 
341  m0_sm_state_set(&m, S_TMO_1);
342  M0_UT_ASSERT(m.sm_state == S_TMO_1);
343 
344  result = m0_sm_timedwait(&m, M0_BITS(S_TMO_2), M0_TIME_NEVER);
345  M0_UT_ASSERT(result == 0);
346  M0_UT_ASSERT(m.sm_state == S_TMO_2);
347 
348  m0_sm_timeout_fini(&t1);
349 
350  m0_sm_state_set(&m, S_TMO_TERMINAL);
351  M0_UT_ASSERT(m.sm_state == S_TMO_TERMINAL);
352 
353  m0_sm_fini(&m);
354  m0_sm_group_unlock(&G);
355 }

480 static int over(struct m0_sm *mach)
481 {
