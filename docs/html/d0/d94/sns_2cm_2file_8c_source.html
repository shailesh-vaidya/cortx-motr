246 M0_INTERNAL int m0_sns_cm_fctx_init(struct m0_sns_cm *scm,
247  const struct m0_fid *fid,
248  struct m0_sns_cm_file_ctx **sc_fctx)
249 {
250  struct m0_sns_cm_file_ctx *fctx;
251 
252  M0_ENTRY();
253  M0_PRE(sc_fctx != NULL || scm != NULL || fid!= NULL);
254  M0_PRE(!m0_fid_eq(fid, &M0_COB_ROOT_FID) ||
255  !m0_fid_eq(fid, &M0_COB_ROOT_FID) ||
256  !m0_fid_eq(fid, &M0_MDSERVICE_SLASH_FID));
257  M0_PRE(m0_mutex_is_locked(&scm->sc_file_ctx_mutex));
258  M0_PRE(m0_sns_cm_fid_is_valid(scm, fid));
259 
260  M0_ALLOC_PTR(fctx);
261  if (fctx == NULL)
262  return M0_ERR(-ENOMEM);
263  m0_fid_set(&fctx->sf_fid, fid->f_container, fid->f_key);
264 
265  fctx->sf_pd = NULL;
266  fctx->sf_layout = NULL;
267  fctx->sf_pi = NULL;
268  fctx->sf_nr_ios_visited = 0;
269  fctx->sf_max_group = 0;
270  m0_scmfctx_tlink_init(fctx);
271  m0_ref_init(&fctx->sf_ref, 1, sns_cm_fctx_release);
272  m0_sm_init(&fctx->sf_sm, &fctx_sm_conf, M0_SCFS_INIT,
273  &scm->sc_base.cm_sm_group);
274  m0_clink_init(&fctx->sf_fini_clink, fctx_fini_clink_cb);
275  m0_mutex_init(&fctx->sf_lock);
276  fctx->sf_group = &scm->sc_base.cm_sm_group;
277  fctx->sf_scm = scm;
278  if (!m0_sns_cm2reqh(scm)->rh_oostore)
279  sns_cm_fctx_rm_init(fctx);
280  *sc_fctx = fctx;
281 
282  return M0_RC(0);
283 }

321  /* XXX: Ideally m0_file_lock should be called, but
322  * it internally calls m0_rm_incoming_init(). This has already been
323  * called here in m0_sns_cm_file_lock_init().
324  * m0_file_lock(&fctx->sf_owner, &fctx->sf_rin);
325  */
326  fctx->sf_rin.rin_want.cr_datum = RM_FILE_LOCK;
327  fctx->sf_rin.rin_want.cr_group_id = m0_rm_sns_cm_group;
328  fctx->sf_rin.rin_ops = &file_lock_incoming_ops;
329 
330  m0_rm_credit_get(&fctx->sf_rin);
331 
332  M0_POST(m0_sns_cm_fctx_locate(scm, &fctx->sf_fid) == fctx);

369 M0_INTERNAL int
370 m0_sns_cm_file_lock_wait(struct m0_sns_cm_file_ctx *fctx, struct m0_fom *fom)
371 {
372  struct m0_chan *rm_chan;
373  uint32_t state;
374 
375  M0_ENTRY();
376 
377  M0_PRE(fctx != NULL && fctx->sf_scm != NULL && fom != NULL);
378  M0_PRE(m0_cm_is_locked(&fctx->sf_scm->sc_base));
379  M0_PRE(m0_mutex_is_locked(&fctx->sf_scm->sc_file_ctx_mutex));
380  M0_PRE(m0_sns_cm_fid_is_valid(fctx->sf_scm, &fctx->sf_fid));
381  M0_PRE(m0_sns_cm_fctx_locate(fctx->sf_scm, &fctx->sf_fid) != NULL);
382 
383  m0_rm_owner_lock(&fctx->sf_owner);
384  state = fctx->sf_rin.rin_sm.sm_state;
385  if (state == RI_SUCCESS || state == RI_FAILURE) {
386  m0_rm_owner_unlock(&fctx->sf_owner);
387  if (state == RI_FAILURE) {
388  m0_ref_put(&fctx->sf_ref);
389  return M0_ERR_INFO(-EFAULT,
390  "Failed to lock file "FID_F,
391  FID_P(&fctx->sf_fid));
392  } else {
393  if (m0_sns_cm_fctx_state_get(fctx) == M0_SCFS_LOCK_WAIT)
394  _fctx_status_set(fctx, M0_SCFS_LOCKED);
395  return M0_RC(0);
396  }
397  }
398  rm_chan = &fctx->sf_rin.rin_sm.sm_chan;
399  m0_fom_wait_on(fom, rm_chan, &fom->fo_cb);
400  m0_rm_owner_unlock(&fctx->sf_owner);
401  return M0_RC(-EAGAIN);
402 }

404 M0_INTERNAL int m0_sns_cm_file_lock(struct m0_sns_cm *scm,
405  const struct m0_fid *fid,
406  struct m0_sns_cm_file_ctx **out)
407 {
408  struct m0_sns_cm_file_ctx *fctx;
409  struct m0_fid f = *fid;
410  int rc;
411 
412  M0_ENTRY();
413 
414  M0_PRE(scm != NULL || fid != NULL);
415  M0_PRE(m0_cm_is_locked(&scm->sc_base));
416  M0_PRE(m0_sns_cm_fid_is_valid(scm, fid));
417  M0_PRE(m0_mutex_is_locked(&scm->sc_file_ctx_mutex));
418 
419  fctx = m0_scmfctx_htable_lookup(&scm->sc_file_ctx, &f);
420  if ( fctx != NULL) {
421  *out = fctx;
422  m0_ref_get(&fctx->sf_ref);
423  M0_LOG(M0_DEBUG, "fid: "FID_F, FID_P(fid));
424  if (m0_sns_cm_fctx_state_get(fctx) >= M0_SCFS_LOCKED) {
425  return M0_RC(0);
426  }
427  if (m0_sns_cm_fctx_state_get(fctx) == M0_SCFS_LOCK_WAIT)
428  return M0_RC(-EAGAIN);
429  }
430  rc = m0_sns_cm_fctx_init(scm, fid, &fctx);
431  if (rc == 0) {
432  rc = !m0_sns_cm2reqh(scm)->rh_oostore ?
433  __sns_cm_file_lock(fctx) :
434  __sns_cm_file_oo_lock(fctx);
435  *out = fctx;
436  }
437  return M0_RC(rc);
438 }
