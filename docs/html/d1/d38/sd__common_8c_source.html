66 void fdmi_serv_start_ut(const struct m0_filterc_ops *filterc_ops)
67 {
68  int rc;
69  struct m0_reqh_fdmi_svc_params *fdms_start_params;
70  struct m0_reqh_context *reqh_ctx = &g_sd_ut.motr.cc_reqh_ctx;
71  struct m0_reqh *reqh = &reqh_ctx->rc_reqh;
72 
73  M0_ENTRY();
74  M0_LOG(M0_DEBUG, ">> fdmi_serv_start_ut ");
75 
76  memset(reqh, 0, sizeof(struct m0_reqh));
77  M0_SET0(&g_sd_ut);
78  M0_SET0(&md);
79 
80  sd_ut_lfile = fopen(LOG_FILE_NAME, "w+");
81  M0_UT_ASSERT(sd_ut_lfile != NULL);
82 
83  rc = m0_cs_init(&g_sd_ut.motr, m0_net_all_xprt_get(),
84  m0_net_xprt_nr(), sd_ut_lfile, false);
85  M0_UT_ASSERT(rc == 0);
86 
87  rc = M0_REQH_INIT(reqh,
88  .rhia_fid = &g_process_fid,
89  .rhia_dtm = (void *)1,
90  .rhia_mdstore = &md);
91  M0_UT_ASSERT(rc == 0);
92 
93  rc = m0_reqh_service_allocate(&g_sd_ut.fdmi_service, &m0_fdmi_service_type, NULL);
94  M0_UT_ASSERT(rc == 0);
95 
96  m0_reqh_service_init(g_sd_ut.fdmi_service, reqh, NULL);
97 
98  /* Patch filterc instance used by source dock FOM */
99  M0_ALLOC_PTR(fdms_start_params);
100  M0_ASSERT(fdms_start_params != NULL);
101  if (filterc_ops != NULL) {
102  fdms_start_params->filterc_ops = filterc_ops;
103  }
104 
105  m0_buf_init(&g_sd_ut.fdmi_service->rs_ss_param,
106  fdms_start_params,
107  sizeof(struct m0_reqh_fdmi_svc_params));
108 
109  m0_reqh_rpc_mach_tlink_init_at_tail(
110  &g_sd_ut.rpc_machine,
111  &reqh->rh_rpc_machines);
112 
113  m0_reqh_service_start(g_sd_ut.fdmi_service);
114 
115  m0_reqh_start(reqh);
116 
117  M0_LOG(M0_DEBUG, "<< fdmi_serv_start_ut ");
118  M0_LEAVE();
119 }

171 void prepare_rpc_env(struct test_rpc_env *env,
172  struct m0_reqh *reqh,
173  const struct m0_rpc_frm_ops *frm_ops,
174  bool sender,
175  struct m0_rpc_conn *rpc_conn,
176  struct m0_rpc_session *rpc_session)
177 {
178  enum { TEST_TM_NR = 1 }; /* Number of TMs. */
179  int rc;
180 
181  M0_ENTRY();
182 
183  M0_ASSERT(rpc_conn != NULL && rpc_session != NULL);
184 
185  m0_net_domain_init(&env->tre_net_dom, m0_net_xprt_default_get());
186 
187  rc = m0_rpc_net_buffer_pool_setup(
188  &env->tre_net_dom, &env->tre_buffer_pool,
189  m0_rpc_bufs_nr(M0_NET_TM_RECV_QUEUE_DEF_LEN, TEST_TM_NR),
190  TEST_TM_NR);
191  M0_ASSERT(rc == 0);
192 
193  env->ep_addr_local = "0@lo:12345:32:123";
194  env->ep_addr_remote = "0@lo:12345:32:123";
195  rc = m0_rpc_machine_init(&env->tre_rpc_machine,
196  &env->tre_net_dom,
197  env->ep_addr_local,
198  reqh, &env->tre_buffer_pool, M0_BUFFER_ANY_COLOUR,
199  2048, /* max RPC message size */
200  M0_NET_TM_RECV_QUEUE_DEF_LEN);
201  M0_ASSERT(rc == 0);
202 
203  rc = m0_net_end_point_create(&env->tre_ep,
204  &env->tre_rpc_machine.rm_tm,
205  env->ep_addr_remote);
206  M0_ASSERT(rc == 0);
207 
208  m0_rpc_machine_lock(&env->tre_rpc_machine);
209  env->tre_rpc_chan = rpc_chan_get(&env->tre_rpc_machine, env->tre_ep,
210  10 /* max packets in flight */);
211  env->tre_rpc_chan->rc_frm.f_ops = frm_ops;
212  M0_ASSERT(env->tre_rpc_chan != NULL);
213 
214  env->tre_conn = rpc_conn;
215  if (!sender) {
216  rc = m0_rpc_rcv_conn_init(env->tre_conn, env->tre_ep,
217  &env->tre_rpc_machine, &M0_UINT128(10, 10));
218  } else {
219  m0_rpc_machine_unlock(&env->tre_rpc_machine);
220  rc = m0_rpc_conn_init(env->tre_conn, NULL, env->tre_ep,
221  &env->tre_rpc_machine, 10);
222  m0_rpc_machine_lock(&env->tre_rpc_machine);
223  }
224 
225  conn_state_set(env->tre_conn, M0_RPC_CONN_ACTIVE);
226  env->tre_conn->c_sender_id = 0xFEFE;
227  m0_rpc_machine_unlock(&env->tre_rpc_machine);
228  M0_ASSERT(rc == 0);
229 
230  env->tre_session = rpc_session;
231  rc = m0_rpc_session_init(env->tre_session, env->tre_conn);
232  env->tre_session->s_session_id = 10;
233  m0_rpc_machine_lock(&env->tre_rpc_machine);
234  session_state_set(env->tre_session, M0_RPC_SESSION_IDLE);
235  m0_rpc_machine_unlock(&env->tre_rpc_machine);
236  M0_ASSERT(rc == 0);
237 
238  M0_LEAVE();
239 }

241 void unprepare_rpc_env(struct test_rpc_env *env)
242 {
243  M0_ENTRY("env %p", env);
244 
245  if (session_state(env->tre_session) == M0_RPC_SESSION_BUSY) {
246  m0_rpc_machine_lock(&env->tre_rpc_machine);
247  session_state_set(env->tre_session, M0_RPC_SESSION_IDLE);
248  env->tre_session->s_hold_cnt = 0;
249  env->tre_rpc_chan->rc_frm.f_state = FRM_IDLE;
250  env->tre_rpc_chan->rc_frm.f_nr_items = 0;
251  env->tre_rpc_chan->rc_frm.f_nr_packets_enqed = 0;
252  m0_sm_fail(&env->tre_session->s_sm,
253  M0_RPC_SESSION_FAILED, -EBUSY);
254  m0_rpc_machine_unlock(&env->tre_rpc_machine);
255  } else {
256  m0_rpc_machine_lock(&env->tre_rpc_machine);
257  session_state_set(env->tre_session, M0_RPC_SESSION_TERMINATED);
258  m0_rpc_machine_unlock(&env->tre_rpc_machine);
259  }
260  m0_rpc_session_fini(env->tre_session);
261 
262  m0_rpc_machine_lock(&env->tre_rpc_machine);
263  conn_state_set(env->tre_conn, M0_RPC_CONN_TERMINATED);
264  m0_rpc_machine_unlock(&env->tre_rpc_machine);
265  m0_rpc_conn_fini(env->tre_conn);
266 
267  m0_rpc_machine_lock(&env->tre_rpc_machine);
268  rpc_chan_put(env->tre_rpc_chan);
269  m0_rpc_machine_unlock(&env->tre_rpc_machine);
270 
271  m0_net_end_point_put(env->tre_ep);
272 
273  m0_rpc_machine_fini(&env->tre_rpc_machine);
274  m0_rpc_net_buffer_pool_cleanup(&env->tre_buffer_pool);
275 
276  m0_net_domain_fini(&env->tre_net_dom);
277 
278  M0_LEAVE();
279 }
