60 static int ha_dispatcher_level_enter(struct m0_module *module)
61 {
62  enum ha_dispatcher_level level = module->m_cur + 1;
63  struct m0_ha_dispatcher *hd;
64 
65  hd = container_of(module, struct m0_ha_dispatcher, hds_module);
66  M0_ENTRY("hd=%p level=%d", hd, level);
67  switch (level) {
68  case HA_DISPATCHER_LEVEL_TLIST:
69  ha_dispatcher_handlers_tlist_init(&hd->hds_handlers);
70  return M0_RC(0);
71  case HA_DISPATCHER_LEVEL_NOTE_HANDLER_ALLOC:
72  if (!hd->hds_cfg.hdc_enable_note)
73  return M0_RC(0);
74  M0_ALLOC_PTR(hd->hds_note_handler);
75  return hd->hds_note_handler == NULL ? M0_ERR(-ENOMEM) :
76  M0_RC(0);
77  case HA_DISPATCHER_LEVEL_NOTE_HANDLER_INIT:
78  if (!hd->hds_cfg.hdc_enable_note)
79  return M0_RC(0);
80  return M0_RC(m0_ha_note_handler_init(hd->hds_note_handler,
81  hd));
82  case HA_DISPATCHER_LEVEL_FVEC_HANDLER_ALLOC:
83  if (!hd->hds_cfg.hdc_enable_fvec)
84  return M0_RC(0);
85  M0_ALLOC_PTR(hd->hds_fvec_handler);
86  return hd->hds_fvec_handler == NULL ? M0_ERR(-ENOMEM) :
87  M0_RC(0);
88  case HA_DISPATCHER_LEVEL_FVEC_HANDLER_INIT:
89  if (!hd->hds_cfg.hdc_enable_fvec)
90  return M0_RC(0);
91  return M0_RC(m0_ha_fvec_handler_init(hd->hds_fvec_handler,
92  hd));
93  case HA_DISPATCHER_LEVEL_KEEPALIVE_HANDLER_ALLOC:
94  if (!hd->hds_cfg.hdc_enable_keepalive)
95  return M0_RC(0);
96  M0_ALLOC_PTR(hd->hds_keepalive_handler);
97  return hd->hds_keepalive_handler == NULL ? M0_ERR(-ENOMEM) :
98  M0_RC(0);
99  case HA_DISPATCHER_LEVEL_KEEPALIVE_HANDLER_INIT:
100  if (!hd->hds_cfg.hdc_enable_keepalive)
101  return M0_RC(0);
102  return M0_RC(m0_ha_keepalive_handler_init(
103  hd->hds_keepalive_handler, hd));
104  case HA_DISPATCHER_LEVEL_READY:
105  M0_IMPOSSIBLE("can't be here");
106  }
107  return M0_ERR(-ENOSYS);
108 }

110 static void ha_dispatcher_level_leave(struct m0_module *module)
111 {
112  enum ha_dispatcher_level level = module->m_cur;
113  struct m0_ha_dispatcher *hd;
114 
115  hd = container_of(module, struct m0_ha_dispatcher, hds_module);
116  M0_ENTRY("hd=%p level=%d", hd, level);
117  switch (level) {
118  case HA_DISPATCHER_LEVEL_TLIST:
119  ha_dispatcher_handlers_tlist_fini(&hd->hds_handlers);
120  break;
121  case HA_DISPATCHER_LEVEL_NOTE_HANDLER_ALLOC:
122  if (hd->hds_cfg.hdc_enable_note)
123  m0_free(hd->hds_note_handler);
124  break;
125  case HA_DISPATCHER_LEVEL_NOTE_HANDLER_INIT:
126  if (hd->hds_cfg.hdc_enable_note)
127  m0_ha_note_handler_fini(hd->hds_note_handler);
128  break;
129  case HA_DISPATCHER_LEVEL_FVEC_HANDLER_ALLOC:
130  if (hd->hds_cfg.hdc_enable_fvec)
131  m0_free(hd->hds_fvec_handler);
132  break;
133  case HA_DISPATCHER_LEVEL_FVEC_HANDLER_INIT:
134  if (hd->hds_cfg.hdc_enable_fvec)
135  m0_ha_fvec_handler_fini(hd->hds_fvec_handler);
136  break;
137  case HA_DISPATCHER_LEVEL_KEEPALIVE_HANDLER_ALLOC:
138  if (hd->hds_cfg.hdc_enable_keepalive)
139  m0_free(hd->hds_keepalive_handler);
140  break;
141  case HA_DISPATCHER_LEVEL_KEEPALIVE_HANDLER_INIT:
142  if (hd->hds_cfg.hdc_enable_keepalive)
143  m0_ha_keepalive_handler_fini(hd->hds_keepalive_handler);
144  break;
145  case HA_DISPATCHER_LEVEL_READY:
146  M0_IMPOSSIBLE("can't be here");
147  }
148  M0_LEAVE();
149 }

192 M0_INTERNAL int m0_ha_dispatcher_init(struct m0_ha_dispatcher *hd,
193  struct m0_ha_dispatcher_cfg *hd_cfg)
194 {
195  int rc;
196 
197  M0_ENTRY("hd=%p", hd);
198  M0_PRE(M0_IS0(hd));
199 
200  hd->hds_cfg = *hd_cfg;
201  m0_module_setup(&hd->hds_module, "m0_ha_dispatcher",
202  ha_dispatcher_levels,
203  ARRAY_SIZE(ha_dispatcher_levels), m0_get());
204  rc = m0_module_init(&hd->hds_module, HA_DISPATCHER_LEVEL_READY);
205  if (rc != 0) {
206  m0_module_fini(&hd->hds_module, M0_MODLEV_NONE);
207  return M0_ERR(rc);
208  }
209  return M0_RC(0);
210 }
