48 void file_di_init(void)
49 {
50  int rc;
51  int i;
52 
53  rc = m0_bufvec_alloc(&data, SEGS_NR, BUFFER_SIZE);
54  M0_ASSERT(rc == 0);
55  for (i = 0; i < data.ov_vec.v_nr; ++i)
56  memset(data.ov_buf[i], 'a' + i, BUFFER_SIZE);
57 
58  for (i = 0; i < data.ov_vec.v_nr; ++i) {
59  offsets[i] = i * size;
60  counts[i] = size;
61  }
62 
63  io_vec.iv_vec.v_nr = data.ov_vec.v_nr;
64  io_vec.iv_index = offsets;
65  io_vec.iv_vec.v_count = counts;
66 
67  res_type.rt_id = M0_RM_FLOCK_RT;
68  res_type.rt_ops = &file_lock_type_ops;
69  m0_rm_domain_init(&res_dom);
70  rc = m0_rm_type_register(&res_dom, &res_type);
71  M0_UT_ASSERT(rc == 0);
72  m0_fid_set(&fid, 1, 0);
73  m0_file_init(&file, &fid, &res_dom, M0_DI_CRC32_4K);
74 
75  file_di_info_setup(&file, &io_vec, &di_param);
76 
77  size = file.fi_di_ops->do_out_shift(&file) *
78  io_vec.iv_vec.v_nr * M0_DI_ELEMENT_SIZE;
79  di_data = m0_alloc(size);
80  M0_UT_ASSERT(di_data != NULL);
81 }
