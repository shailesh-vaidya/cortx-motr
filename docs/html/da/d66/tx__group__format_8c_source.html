62 static int be_group_format_level_enter(struct m0_module *module)
63 {
64  struct m0_be_group_format *gft = be_group_format_module2gft(module);
65  m0_bcount_t size_group;
66  m0_bcount_t size_cblock;
67  int level = module->m_cur + 1;
68 
69  M0_LOG(M0_DEBUG, "gft=%p level=%d", gft, level);
70 
71  switch (level) {
72  case M0_BE_GROUP_FORMAT_LEVEL_ASSIGNS:
73  gft->gft_log = gft->gft_cfg.gfc_log;
74  gft->gft_fmt_group_decoded = NULL;
75  gft->gft_fmt_cblock_decoded = NULL;
76  return 0;
77  case M0_BE_GROUP_FORMAT_LEVEL_OP_INIT:
78  m0_be_op_init(&gft->gft_pd_io_get);
79  m0_be_op_init(&gft->gft_log_discard_get);
80  m0_be_op_init(&gft->gft_all_get);
81  return 0;
82  case M0_BE_GROUP_FORMAT_LEVEL_FMT_GROUP_INIT:
83  return m0_be_fmt_group_init(&gft->gft_fmt_group,
84  &gft->gft_cfg.gfc_fmt_cfg);
85  case M0_BE_GROUP_FORMAT_LEVEL_FMT_CBLOCK_INIT:
86  return m0_be_fmt_cblock_init(&gft->gft_fmt_cblock, NULL);
87  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_INIT:
88  m0_be_log_record_init(&gft->gft_log_record, gft->gft_log);
89  return 0;
90  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_IO_CREATE_GROUP:
91  size_group = m0_be_fmt_group_size_max(&gft->gft_cfg.gfc_fmt_cfg);
92  return m0_be_log_record_io_create(&gft->gft_log_record,
93  size_group);
94  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_IO_CREATE_CBLOCK:
95  size_cblock = m0_be_fmt_cblock_size_max(NULL);
96  return m0_be_log_record_io_create(&gft->gft_log_record,
97  size_cblock);
98  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_ITER_INIT:
99  return m0_be_log_record_iter_init(&gft->gft_log_record_iter);
100  case M0_BE_GROUP_FORMAT_LEVEL_INITED:
101  return 0;
102  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_ALLOCATE:
103  return m0_be_log_record_allocate(&gft->gft_log_record);
104  case M0_BE_GROUP_FORMAT_LEVEL_ALLOCATED:
105  return 0;
106  default:
107  return M0_ERR(-ENOSYS);
108  }
109 }

111 static void be_group_format_level_leave(struct m0_module *module)
112 {
113  struct m0_be_group_format *gft = be_group_format_module2gft(module);
114  int level = module->m_cur;
115 
116  M0_LOG(M0_DEBUG, "gft=%p level=%d", gft, level);
117 
118  switch (level) {
119  case M0_BE_GROUP_FORMAT_LEVEL_ASSIGNS:
120  break;
121  case M0_BE_GROUP_FORMAT_LEVEL_OP_INIT:
122  m0_be_op_fini(&gft->gft_all_get);
123  m0_be_op_fini(&gft->gft_log_discard_get);
124  m0_be_op_fini(&gft->gft_pd_io_get);
125  break;
126  case M0_BE_GROUP_FORMAT_LEVEL_FMT_GROUP_INIT:
127  m0_be_fmt_group_fini(&gft->gft_fmt_group);
128  break;
129  case M0_BE_GROUP_FORMAT_LEVEL_FMT_CBLOCK_INIT:
130  m0_be_fmt_cblock_fini(&gft->gft_fmt_cblock);
131  break;
132  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_INIT:
133  m0_be_log_record_fini(&gft->gft_log_record);
134  break;
135  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_IO_CREATE_GROUP:
136  break;
137  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_IO_CREATE_CBLOCK:
138  break;
139  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_ITER_INIT:
140  m0_be_log_record_iter_fini(&gft->gft_log_record_iter);
141  break;
142  case M0_BE_GROUP_FORMAT_LEVEL_INITED:
143  break;
144  case M0_BE_GROUP_FORMAT_LEVEL_LOG_RECORD_ALLOCATE:
145  m0_be_log_record_deallocate(&gft->gft_log_record);
146  break;
147  case M0_BE_GROUP_FORMAT_LEVEL_ALLOCATED:
148  if (gft->gft_fmt_group_decoded != NULL) {
149  m0_be_fmt_group_decoded_free(gft->gft_fmt_group_decoded);
150  }
151  if (gft->gft_fmt_cblock_decoded != NULL) {
152  m0_be_fmt_cblock_decoded_free(
153  gft->gft_fmt_cblock_decoded);
154  }
155  if (gft->gft_pd_io != NULL)
156  m0_be_pd_io_put(gft->gft_cfg.gfc_pd, gft->gft_pd_io);
157  break;
158  default:
159  M0_IMPOSSIBLE("Unexpected m0_module level");
160  }
161 }

163 M0_INTERNAL void m0_be_group_format_reset(struct m0_be_group_format *gft)
164 {
165  m0_be_fmt_group_reset(&gft->gft_fmt_group);
166  m0_be_fmt_cblock_reset(&gft->gft_fmt_cblock);
167  m0_be_log_record_reset(&gft->gft_log_record);
168  if (gft->gft_pd_io != NULL) {
169  m0_be_pd_io_put(gft->gft_cfg.gfc_pd, gft->gft_pd_io);
170  gft->gft_pd_io = NULL;
171  }
172  m0_be_op_reset(&gft->gft_pd_io_get);
173  m0_be_op_reset(&gft->gft_log_discard_get);
174  m0_be_op_reset(&gft->gft_all_get);
175 
176  if (gft->gft_fmt_group_decoded != NULL) {
177  m0_time_t time = m0_time_now();
178  m0_be_fmt_group_decoded_free(gft->gft_fmt_group_decoded);
179  M0_LOG(M0_DEBUG, "time=%"PRId64,
180  (unsigned long)m0_time_now() - time);
181  gft->gft_fmt_group_decoded = NULL;
182  }
183  if (gft->gft_fmt_cblock_decoded != NULL) {
184  m0_be_fmt_cblock_decoded_free(
185  gft->gft_fmt_cblock_decoded);
186  gft->gft_fmt_cblock_decoded = NULL;
187  }
188 }

327 M0_INTERNAL void m0_be_group_format_encode(struct m0_be_group_format *gft)
328 {

335  rc = m0_be_fmt_group_encode(&gft->gft_fmt_group, &cur);

339  m0_bufvec_cursor_init(&cur, bvec);
340  rc = m0_be_fmt_cblock_encode(&gft->gft_fmt_cblock, &cur);
341  M0_ASSERT_INFO(rc == 0, "due to preallocated buffers "
342  "encode can't fail here: rc = %d", rc);
343 }

345 M0_INTERNAL int m0_be_group_format_decode(struct m0_be_group_format *gft)
346 {
347  struct m0_bufvec_cursor cur;
348  struct m0_bufvec *bvec;
349  int rc;
350 
351  M0_PRE(gft->gft_fmt_group_decoded == NULL);
352  M0_PRE(gft->gft_fmt_cblock_decoded == NULL);
353 
354  bvec = m0_be_log_record_io_bufvec(&gft->gft_log_record, GFT_GROUP_IO);
355  m0_bufvec_cursor_init(&cur, bvec);
356  rc = m0_be_fmt_group_decode(&gft->gft_fmt_group_decoded, &cur,
357  M0_BE_FMT_DECODE_CFG_DEFAULT);
358 
359  bvec = m0_be_log_record_io_bufvec(&gft->gft_log_record, GFT_GROUP_CB_IO);
360  m0_bufvec_cursor_init(&cur, bvec);
361  rc = rc ?: m0_be_fmt_cblock_decode(&gft->gft_fmt_cblock_decoded,
362  &cur, M0_BE_FMT_DECODE_CFG_DEFAULT);
363  return rc;
364 }
