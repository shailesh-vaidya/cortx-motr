98 static int send_fop(struct m0_rpc_session *session)
99 {
100  int rc;
101  struct m0_fop *fop;
102  struct cs_ds2_req_fop *cs_ds2_fop;
103 
104  fop = m0_fop_alloc_at(session, &cs_ds2_req_fop_fopt);
105  M0_UT_ASSERT(fop != NULL);
106  if (fop == NULL) {
107  rc = -ENOMEM;
108  goto out;
109  }
110 
111  cs_ds2_fop = m0_fop_data(fop);
112  cs_ds2_fop->csr_value = 0xaaf5;
113 
114  rc = m0_rpc_post_sync(fop, session, &cs_ds_req_fop_rpc_item_ops,
115  0 /* deadline */);
116  M0_UT_ASSERT(rc == 0);
117  M0_UT_ASSERT(fop->f_item.ri_error == 0);
118  M0_UT_ASSERT(fop->f_item.ri_reply != 0);
119 
120  m0_fop_put_lock(fop);
121 out:
122  return rc;
123 }

125 static void test_rpclib(void)
126 {
127  int rc;
128 
129  /*
130  * There is no need to initialize xprt explicitly if client and server
131  * run within a single process, because in this case transport is
132  * initialized by m0_rpc_server_start().
133  */
134 
135  sctx_reset();
136  rc = m0_rpc_server_start(&sctx);
137  M0_UT_ASSERT(rc == 0);
138  if (rc != 0)
139  return;
140  rc = m0_rpc_client_start(&cctx);
141  M0_UT_ASSERT(rc == 0);
142  if (rc != 0)
143  goto server_fini;
144 
145  rc = send_fop(&cctx.rcx_session);
146  M0_UT_ASSERT(rc == 0);
147 
148  rc = m0_rpc_client_stop(&cctx);
149  M0_UT_ASSERT(rc == 0);
150 
151 server_fini:
152  m0_rpc_server_stop(&sctx);
153  return;
154 }
