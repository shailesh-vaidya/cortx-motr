183 static void session_terminate_reply_and_fini(int err)
184 {
185  /* Session transition from TERMINATING => TERMINATED | FAILED */
186  term_fop.f_item.ri_error = err;
187  m0_rpc_machine_lock(&machine);
188  m0_rpc_session_terminate_reply_received(&term_fop.f_item);
189  m0_rpc_machine_unlock(&machine);
190  M0_UT_ASSERT(err == 0 ?
191  session_state(&session) == M0_RPC_SESSION_TERMINATED :
192  session_state(&session) == M0_RPC_SESSION_FAILED);
193 
194  m0_rpc_session_fini(&session);
195 }

197 static void session_hold_release(void)
198 {
199  /* Session transition from IDLE => BUSY => IDLE */

203  m0_rpc_session_release(&session);
204  M0_UT_ASSERT(session_state(&session) == M0_RPC_SESSION_IDLE);
205  m0_rpc_machine_unlock(&machine);

208 static void session_check(void)
209 {
210  /* Checks for session states transitions,
211  INITIALISED => ESTABLISHING => IDLE => BUSY => IDLE =>
212  TERMINATING => FINALISED.
213  */
214  session_init_and_establish();
215  session_establish_reply(0);
216  M0_UT_ASSERT(session_state(&session) == M0_RPC_SESSION_IDLE);
217  session_hold_release();
218  session_terminate();
219  session_terminate_reply_and_fini(0);
220 }
