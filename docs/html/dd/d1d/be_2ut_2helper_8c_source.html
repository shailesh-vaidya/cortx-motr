185 static void be_ut_sm_group_thread_func(struct m0_be_ut_sm_group_thread *sgt)
186 {
187  struct m0_sm_group *grp = &sgt->sgt_grp;
188  bool need_to_up_asts_run_sem;
189 
190  while (!m0_semaphore_trydown(&sgt->sgt_stop_sem)) {
191  m0_chan_wait(&grp->s_clink);
192  m0_sm_group_lock(grp);
193  need_to_up_asts_run_sem =
194  m0_semaphore_trydown(&sgt->sgt_asts_run_sem);
195  m0_sm_asts_run(grp);
196  if (need_to_up_asts_run_sem)
197  m0_semaphore_up(&sgt->sgt_asts_run_sem);
198  m0_sm_group_unlock(grp);
199  }
200 }

202 static int m0_be_ut_sm_group_thread_init(struct m0_be_ut_sm_group_thread **sgtp,
203  bool lock_new)
204 {
205  struct m0_be_ut_sm_group_thread *sgt;
206  int rc;
207 
208  M0_ALLOC_PTR(*sgtp);
209  sgt = *sgtp;
210  if (sgt != NULL) {
211  sgt->sgt_tid = gettid_impl();
212  sgt->sgt_lock_new = lock_new;
213 
214  m0_sm_group_init(&sgt->sgt_grp);
215  m0_semaphore_init(&sgt->sgt_stop_sem, 0);
216  m0_semaphore_init(&sgt->sgt_asts_run_sem, 0);
217  rc = M0_THREAD_INIT(&sgt->sgt_thread,
218  struct m0_be_ut_sm_group_thread *, NULL,
219  &be_ut_sm_group_thread_func, sgt,
220  "be_ut sgt");
221  if (rc == 0) {
222  if (sgt->sgt_lock_new)
223  m0_sm_group_lock(&sgt->sgt_grp);
224  } else {
225  m0_semaphore_fini(&sgt->sgt_stop_sem);
226  m0_sm_group_fini(&sgt->sgt_grp);
227  m0_free(sgt);
228  }
229  } else {
230  rc = -ENOMEM;
231  }
232  return rc;
233 }

235 static void m0_be_ut_sm_group_thread_fini(struct m0_be_ut_sm_group_thread *sgt)
236 {
237  int rc;
238 
239  m0_semaphore_up(&sgt->sgt_stop_sem);
240 
241  m0_clink_signal(&sgt->sgt_grp.s_clink);
242  if (sgt->sgt_lock_new)
243  m0_sm_group_unlock(&sgt->sgt_grp);
244 
245  rc = m0_thread_join(&sgt->sgt_thread);
246  M0_ASSERT(rc == 0);
247  m0_thread_fini(&sgt->sgt_thread);
248 
249  m0_semaphore_fini(&sgt->sgt_asts_run_sem);
250  m0_semaphore_fini(&sgt->sgt_stop_sem);
251  m0_sm_group_fini(&sgt->sgt_grp);
252  m0_free(sgt);
253 }

351 M0_INTERNAL int m0_be_ut_backend_init_cfg(struct m0_be_ut_backend *ut_be,
352  const struct m0_be_domain_cfg *cfg,
353  bool mkfs)
354 {
355  static bool mkfs_executed = false;
356  struct m0_be_domain_cfg *c;
357  int rc;
358 
359  m0_bob_init(&m0_ut_be_backend_bobtype, ut_be);
360  if (!mkfs_executed && cfg == NULL &&
361  ut_be->but_stob_domain_location == NULL)
362  mkfs = mkfs_executed = true;
363 
364  ut_be->but_sm_groups_unlocked = false;
365  m0_mutex_init(&ut_be->but_sgt_lock);
366 
367  if (cfg == NULL)
368  m0_be_ut_backend_cfg_default(&ut_be->but_dom_cfg);
369  else
370  /*
371  * Make a copy of `cfg': it can be an automatic variable,
372  * which scope we should not rely on.
373  */
374  ut_be->but_dom_cfg = *cfg;
375  c = &ut_be->but_dom_cfg;
376 
377  /* Create reqh, if necessary. */
378  if (c->bc_engine.bec_reqh == NULL)
379  m0_be_ut_reqh_create(&c->bc_engine.bec_reqh);
380 
381  /* Use m0_be_ut_backend's stob domain location, if possible. */
382  if (ut_be->but_stob_domain_location != NULL)
383  c->bc_stob_domain_location = ut_be->but_stob_domain_location;
384 
385  c->bc_mkfs_mode = mkfs;
386 
387 check_mkfs:
388  m0_be_domain_module_setup(&ut_be->but_dom, c);
389  rc = m0_module_init(&ut_be->but_dom.bd_module,
390  M0_BE_DOMAIN_LEVEL_READY);
391  if (!c->bc_mkfs_mode && rc == -ENOENT) {
392  m0_module_fini(&ut_be->but_dom.bd_module, M0_MODLEV_NONE);
393  M0_SET0(&ut_be->but_dom);
394  c->bc_mkfs_mode = true;
395  goto check_mkfs;
396  }
397  if (rc != 0)
398  m0_mutex_fini(&ut_be->but_sgt_lock);
399 
400  return rc;
401 }

536 static struct m0_sm_group *
537 be_ut_backend_sm_group_lookup(struct m0_be_ut_backend *ut_be, bool lock_new)
538 {
539  struct m0_be_ut_sm_group_thread *sgt;
540  struct m0_sm_group *grp;
541  pid_t tid = gettid_impl();
542  unsigned i;
543  int rc;
544 
545  m0_mutex_lock(&ut_be->but_sgt_lock);
546  grp = NULL;
547  for (i = 0; i < ut_be->but_sgt_size; ++i) {
548  sgt = ut_be->but_sgt[i];
549  if (sgt->sgt_tid == tid) {
550  grp = &sgt->sgt_grp;
551  break;
552  }
553  }
554  if (grp == NULL) {
555  rc = m0_be_ut_sm_group_thread_init(&sgt, lock_new);
556  M0_ASSERT(rc == 0);
557  be_ut_sm_group_thread_add(ut_be, sgt);
558  grp = &sgt->sgt_grp;
559  }
560  m0_mutex_unlock(&ut_be->but_sgt_lock);
561  M0_POST(grp != NULL);
562  return grp;
563 }
