149 static int tx_group_fom_tick(struct m0_fom *fom)
150 {
151  enum tx_group_fom_state phase = m0_fom_phase(fom);
152  struct m0_be_tx_group_fom *m = fom2tx_group_fom(fom);
153  struct m0_be_tx_group *gr = m->tgf_group;
154  struct m0_be_op *op = &m->tgf_op;
155  int rc;
156 
157  M0_ENTRY("tx_group_fom=%p group=%p phase=%s", m, gr,
158  m0_fom_phase_name(fom, phase));
159 
160  switch (phase) {
161  case TGS_INIT:
162  rc = m0_be_tx_group__allocate(gr);
163  m0_fom_phase_move(fom, rc, rc == 0 ? TGS_OPEN : TGS_FAILED);
164  m0_semaphore_up(&m->tgf_start_sem);
165  return M0_FSO_WAIT;
166  case TGS_OPEN:
167  if (m->tgf_stopping) {
168  m0_fom_phase_set(fom, TGS_STOPPING);
169  return M0_FSO_AGAIN;
170  }
171  return M0_FSO_WAIT;
172  case TGS_PREPARE:
173  m0_be_op_reset(op);
174  m0_be_tx_group_prepare(gr, op);
175  return m0_be_op_tick_ret(op, fom, TGS_LOGGING);
176  case TGS_LOGGING:
177  m0_be_op_reset(op);
178  if (m->tgf_recovery_mode) {
179  m0_be_tx_group_log_read(gr, op);
180  return m0_be_op_tick_ret(op, fom, TGS_RECONSTRUCT);
181  } else {
182  m0_be_tx_group_encode(gr);
183  m0_be_tx_group_log_write(gr, op);
184  return m0_be_op_tick_ret(op, fom, TGS_PLACING);
185  }
186  case TGS_RECONSTRUCT:
187  rc = m0_be_tx_group_decode(gr);
188  M0_ASSERT_INFO(rc == 0, "rc = %d", rc); /* XXX notify engine */
189  rc = m0_be_tx_group_reconstruct(gr,
190  &m->tgf_gen.fo_loc->fl_group);
191  M0_ASSERT_INFO(rc == 0, "rc = %d", rc); /* XXX notify engine */
192  m0_fom_phase_set(fom, TGS_TX_OPEN);
193  return M0_FSO_AGAIN;
194  case TGS_TX_OPEN:
195  m0_be_op_reset(op);
196  m0_be_tx_group_reconstruct_tx_open(gr, op);
197  return m0_be_op_tick_ret(op, fom, TGS_TX_CLOSE);
198  case TGS_TX_CLOSE:
199  m0_be_op_reset(&m->tgf_op_gc);
200  /* m0_be_op_tick_ret() for the op is in TGS_TX_GC_WAIT phase */
201  m0_be_tx_group_reconstruct_tx_close(gr, &m->tgf_op_gc);
202  m0_fom_phase_set(fom, TGS_REAPPLY);
203  return M0_FSO_AGAIN;
204  case TGS_REAPPLY:
205  m0_be_op_reset(op);
206  rc = m0_be_tx_group_reapply(gr, op);
207  M0_ASSERT_INFO(rc == 0, "rc = %d", rc); /* XXX notify engine */
208  return m0_be_op_tick_ret(op, fom, TGS_PLACING);
209  case TGS_PLACING:
210  m0_be_tx_group__tx_state_post(gr, M0_BTS_LOGGED, false);
211  m0_be_op_reset(op);
212  m0_be_tx_group_seg_place_prepare(gr);
213  m0_be_tx_group_seg_place(gr, op);
214  return m0_be_op_tick_ret(op, fom, TGS_PLACED);
215  case TGS_PLACED:
216  m0_be_tx_group__tx_state_post(gr, M0_BTS_PLACED, true);
217  m0_fom_phase_set(fom, TGS_STABILIZING);
218  return M0_FSO_AGAIN;
219  case TGS_STABILIZING:
220  if (m->tgf_stable) {
221  m0_fom_phase_set(fom, TGS_STABLE);
222  return M0_FSO_AGAIN;
223  }
224  return M0_FSO_WAIT;
225  case TGS_STABLE:
226  m0_fom_phase_set(fom, m->tgf_recovery_mode ?
227  TGS_TX_GC_WAIT : TGS_RESET);
228  return M0_FSO_AGAIN;
229  case TGS_TX_GC_WAIT:
230  return m0_be_op_tick_ret(&m->tgf_op_gc, fom, TGS_RESET);
231  case TGS_RESET:
232  m0_be_tx_group_reset(gr);
233  m0_be_tx_group_open(gr);
234  m0_fom_phase_set(fom, TGS_OPEN);
235  return M0_FSO_AGAIN;
236  case TGS_STOPPING:
237  m0_be_tx_group__deallocate(gr);
238  m0_fom_phase_set(fom, TGS_FINISH);
239  return M0_FSO_WAIT;
240  case TGS_FAILED:
241  m0_fom_phase_set(fom, TGS_FINISH);
242  return M0_FSO_WAIT;
243  default:
244  M0_IMPOSSIBLE("Invalid phase: %d", phase);
245  }
246 
247  M0_LEAVE();
248  return M0_FSO_WAIT;
249 }

294 /*
295  * Wakes up tx_group_fom iff it is waiting.
296  * It is possible that multiple fom wakeup asts are posted through different
297  * code paths. Thus we avoid waking up of already running FOM.
298  */
299 static void be_tx_group_fom_iff_waiting_wakeup(struct m0_fom *fom)
300 {
301  M0_ENTRY();
302  if (m0_fom_is_waiting(fom)) {
303  M0_LOG(M0_DEBUG, "waking up");
304  m0_fom_ready(fom);
305  }
306  M0_LEAVE();
307 }

381 M0_INTERNAL int m0_be_tx_group_fom_start(struct m0_be_tx_group_fom *gf)
382 {
383  int rc;
384  struct m0_fom *fom = &gf->tgf_gen;
385 
386  m0_fom_queue(fom);
387  m0_semaphore_down(&gf->tgf_start_sem);
388  M0_ASSERT(M0_IN(m0_fom_phase(fom), (TGS_OPEN, TGS_FAILED)));
389  rc = m0_fom_rc(fom);
390  if (m0_fom_phase(fom) == TGS_FAILED) {
391  M0_ASSERT(rc != 0);
392  m0_fom_wakeup(fom);
393  m0_semaphore_down(&gf->tgf_finish_sem);
394  }
395 
396  return M0_RC(rc);
397 }
