217  &m0_cas_index_fid_type,

253  const struct m0_dtm0_ts *ts,
254  bool tbs)
255 {
256  uint64_t version = ts->dts_phys;
257 
258  M0_PRE(version <= M0_CRV_VER_MAX);
259  M0_PRE(version >= M0_CRV_VER_MIN);
260 
261  m0_crv_ts_set(crv, ts);
262  m0_crv_tbs_set(crv, tbs);
263 
264  M0_POST(equi(m0_crv_tbs(crv), tbs));
265  M0_POST(m0_crv_ts(crv).dts_phys == version);
266 }
