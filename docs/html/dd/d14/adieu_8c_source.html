84 static int test_adieu_init(const char *location,
85  const char *dom_cfg,
86  const char *stob_cfg)
87 {
88  int i;
89  int rc;
90  struct m0_stob_id stob_id;
91  char cs_char = 'a';
92 
93  rc = m0_stob_domain_create(location,
94  NULL, M0_STOB_UT_DOMAIN_KEY, dom_cfg, &dom);
95  M0_ASSERT(rc == 0);
96  M0_ASSERT(dom != NULL);
97 
98  m0_stob_id_make(0, M0_STOB_UT_STOB_KEY, &dom->sd_id, &stob_id);
99  rc = m0_stob_find(&stob_id, &obj);
100  M0_ASSERT(rc == 0);
101  rc = m0_stob_locate(obj);
102  M0_ASSERT(rc == 0);
103  rc = m0_ut_stob_create(obj, stob_cfg, NULL);
104  M0_ASSERT(rc == 0);
105 
106  block_shift = m0_stob_block_shift(obj);
107  /* buf_size is chosen so it would be at least MIN_BUF_SIZE in bytes
108  * or it would consist of at least MIN_BUF_SIZE_IN_BLOCKS blocks */
109  buf_size = max_check(MIN_BUF_SIZE,
110  (1 << block_shift) * MIN_BUF_SIZE_IN_BLOCKS);
111 
112  for (i = 0; i < ARRAY_SIZE(user_buf); ++i) {
113  user_buf[i] = m0_alloc_aligned(buf_size, block_shift);
114  M0_ASSERT(user_buf[i] != NULL);
115  }
116 
117  // Allocate contigious buffer for i/p checksums
118  user_cksm_buf[0] = m0_alloc(AD_ADIEU_CS_SZ * ARRAY_SIZE(user_cksm_buf));
119  M0_ASSERT(user_cksm_buf[0] != NULL);
120  memset( user_cksm_buf[0], cs_char++, AD_ADIEU_CS_SZ);
121  for (i = 1; i < ARRAY_SIZE(user_cksm_buf); ++i) {
122  user_cksm_buf[i] = user_cksm_buf[i-1] + AD_ADIEU_CS_SZ;
123  memset( user_cksm_buf[i], cs_char++, AD_ADIEU_CS_SZ);
124  }
125 
126  for (i = 0; i < ARRAY_SIZE(read_buf); ++i) {
127  read_buf[i] = m0_alloc_aligned(buf_size, block_shift);
128  M0_ASSERT(read_buf[i] != NULL);
129  }
130 
131  // Allocate contigious buffer for o/p checksums
132  read_cksm_buf[0] = m0_alloc(AD_ADIEU_CS_SZ * ARRAY_SIZE(read_cksm_buf));
133  M0_ASSERT(read_cksm_buf[0] != NULL);
134  memset( read_cksm_buf[0], 0, AD_ADIEU_CS_SZ);
135  for (i = 1; i < ARRAY_SIZE(read_cksm_buf); ++i) {
136  read_cksm_buf[i] = read_cksm_buf[i-1] + AD_ADIEU_CS_SZ;
137  memset( read_cksm_buf[i], 0, AD_ADIEU_CS_SZ);
138  }
139 
140  for (i = 0; i < NR; ++i) {
141  user_bufs[i] = m0_stob_addr_pack(user_buf[i], block_shift);
142  read_bufs[i] = m0_stob_addr_pack(read_buf[i], block_shift);
143  user_vec[i] = buf_size >> block_shift;
144  stob_vec[i] = (buf_size * (2 * i + 1)) >> block_shift;
145  memset(user_buf[i], ('a' + i)|1, buf_size);
146  }
147  return rc;
148 }

171 static void test_write(int i)
172 {
173  int rc;
174  struct m0_fol_frag *fol_frag;
175 
176  M0_ALLOC_PTR(fol_frag);
177  M0_UB_ASSERT(fol_frag != NULL);
178 
179  m0_stob_io_init(&io);
180 
181  io.si_opcode = SIO_WRITE;
182  io.si_flags = 0;
183  io.si_fol_frag = fol_frag;
184  io.si_user.ov_vec.v_nr = i;
185  io.si_user.ov_vec.v_count = user_vec;
186  io.si_user.ov_buf = (void **)user_bufs;
187 
188  io.si_stob.iv_vec.v_nr = i;
189  io.si_stob.iv_vec.v_count = user_vec;
190  io.si_stob.iv_index = stob_vec;
191 
192  io.si_unit_sz = (buf_size >> block_shift);
193  io.si_cksum_sz = AD_ADIEU_CS_SZ;
194  // Checksum for i buf_size blocks
195  io.si_cksum.b_addr = user_cksm_buf[0];
196  io.si_cksum.b_nob = ( i * AD_ADIEU_CS_SZ );
197 
198  m0_clink_init(&clink, NULL);
199  m0_clink_add_lock(&io.si_wait, &clink);
200 
201  rc = m0_stob_io_prepare_and_launch(&io, obj, NULL, NULL);
202  M0_ASSERT(rc == 0);
203 
204  m0_chan_wait(&clink);
205 
206  M0_ASSERT(io.si_rc == 0);
207  M0_ASSERT(io.si_count == (buf_size * i) >> block_shift);
208 
209  m0_clink_del_lock(&clink);
210  m0_clink_fini(&clink);
211 
212  m0_stob_io_fini(&io);
213 }

257 static void test_adieu(const char *path)
258 {
259  int ch;
260  int i;
261  int j;
262 
263  for (i = 1; i < NR; ++i) {
264  test_write(i);
265 
266  /* this works only for linuxstob */
267  f = fopen(path, "r");
268  for (j = 0; j < i; ++j) {
269  int k;
270 
271  for (k = 0; k < buf_size; ++k) {
272  ch = fgetc(f);
273  M0_ASSERT(ch == '\0');
274  M0_ASSERT(!feof(f));
275  }
276  for (k = 0; k < buf_size; ++k) {
277  ch = fgetc(f);
278  M0_ASSERT(ch != '\0');
279  M0_ASSERT(!feof(f));
280  }
281  }
282  ch = fgetc(f);
283  M0_ASSERT(ch == EOF);
284  fclose(f);
285  }
286 
287  for (i = 1; i < NR; ++i) {
288  test_read(i);
289  M0_ASSERT(memcmp(user_buf[i - 1], read_buf[i - 1], buf_size) == 0);

354 static void ub_iovec_invert()
355 {
356  int i;
357  bool swapped;
358 
359  /* Reverse sort index vecs. */
360  do {
361  swapped = false;
362  for (i = 0; i < NR_SORT - 1; i++) {
363  if (stob_vec1[i] < stob_vec1[i + 1]) {
364  m0_bindex_t tmp = stob_vec1[i];
365  stob_vec1[i] = stob_vec1[i + 1];
366  stob_vec1[i + 1] = tmp;
367  swapped = true;
368  }
369  }
370  } while(swapped);
371 }
