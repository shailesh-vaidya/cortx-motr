134 m0_time_t m0_time_now(void)
135 {
136  m0_time_t result;
137 
138  switch (M0_CLOCK_SOURCE) {
139  case M0_CLOCK_SOURCE_REALTIME_MONOTONIC:
140  M0_PRE(m0_time_monotonic_offset != 0);
141  result = m0_clock_gettime_wrapper(M0_CLOCK_SOURCE_MONOTONIC) +
142  m0_time_monotonic_offset;
143  break;
144  case M0_CLOCK_SOURCE_GTOD:
145  result = m0_clock_gettimeofday_wrapper();
146  break;
147  case M0_CLOCK_SOURCE_REALTIME:
148  case M0_CLOCK_SOURCE_MONOTONIC:
149  case M0_CLOCK_SOURCE_MONOTONIC_RAW:
150  result = m0_clock_gettime_wrapper(M0_CLOCK_SOURCE);
151  break;
152  default:
153  M0_IMPOSSIBLE("Unknown clock source");
154  result = M0_TIME_NEVER;
155  };
156  return result;
157 }

160 M0_INTERNAL m0_time_t m0_time_to_realtime(m0_time_t abs_time)
161 {
162  m0_time_t source_time;
163  m0_time_t realtime;
164  m0_time_t monotonic;
165 
166  if (abs_time != M0_TIME_NEVER && abs_time != 0) {
167  switch (M0_CLOCK_SOURCE) {
168  case M0_CLOCK_SOURCE_MONOTONIC:
169  case M0_CLOCK_SOURCE_MONOTONIC_RAW:
170  source_time = m0_clock_gettime_wrapper(M0_CLOCK_SOURCE);
171  realtime =
172  m0_clock_gettime_wrapper(M0_CLOCK_SOURCE_REALTIME);
173  abs_time += realtime - source_time;
174  break;
175  case M0_CLOCK_SOURCE_REALTIME_MONOTONIC:
176  monotonic =
177  m0_clock_gettime_wrapper(M0_CLOCK_SOURCE_MONOTONIC);
178  realtime =
179  m0_clock_gettime_wrapper(M0_CLOCK_SOURCE_REALTIME);
180  /* get monotonic time */
181  abs_time -= m0_time_monotonic_offset;
182  /* add offset for realtime */
183  abs_time += realtime - monotonic;
184  /* It will mitigate time jumps between call
185  * to m0_time_now() and call to this function. */
186  break;
187  case M0_CLOCK_SOURCE_GTOD:
188  case M0_CLOCK_SOURCE_REALTIME:
189  break;
190  default:
191  M0_IMPOSSIBLE("Unknown clock source");
192  abs_time = 0;
193  }
194  }
195  return abs_time;
196 } M0_EXPORTED(m0_time_to_realtime);
