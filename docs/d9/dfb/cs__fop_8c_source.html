107 static void cs_ut_rpc_item_reply_cb(struct m0_rpc_item *item)
108 {
109  struct m0_fop *reply;
110 
111  M0_PRE(item != NULL);
112  M0_PRE(M0_IN(m0_fop_opcode(m0_rpc_item_to_fop(item)),
113  (M0_CS_DS1_REQ_OPCODE, M0_CS_DS2_REQ_OPCODE)));
114 
115  if (m0_rpc_item_error(item) == 0) {
116  reply = m0_rpc_item_to_fop(item->ri_reply);
117  M0_ASSERT(M0_IN(m0_fop_opcode(reply), (M0_CS_DS1_REP_OPCODE,
118  M0_CS_DS2_REP_OPCODE)));
119  }

186 void m0_ut_fom_phase_set(struct m0_fom *fom, int phase)
187 {
188  switch (m0_fom_phase(fom)) {
189  case M0_FOPH_SUCCESS:
190  m0_fom_phase_set(fom, M0_FOPH_FOL_REC_ADD);
191  /* fall through */
192  case M0_FOPH_FAILURE:
193  m0_fom_phase_set(fom, M0_FOPH_TXN_COMMIT);
194  m0_fom_phase_set(fom, M0_FOPH_QUEUE_REPLY);
195  /* fall through */
196  default:
197  if (m0_fom_phase(fom) != phase)
198  m0_fom_phase_set(fom, phase);
199  }
200 }

203  Allocates and initialises a fom.
204  */
205 static int cs_ds_req_fop_fom_create(struct m0_fop *fop,
206  const struct m0_fom_ops *ops,
207  struct m0_fom **out, struct m0_reqh *reqh)
208 {
209  struct m0_fom *fom;
210  struct m0_fop *rfop;
211 
212  M0_PRE(fop != NULL);
213  M0_PRE(ops != NULL);
214  M0_PRE(out != NULL);
215  M0_PRE(M0_IN(m0_fop_opcode(fop), (M0_CS_DS1_REQ_OPCODE,
216  M0_CS_DS2_REQ_OPCODE)));
217 
218  M0_ALLOC_PTR(fom);
219  if (fom == NULL)
220  return -ENOMEM;
221 
222  rfop = m0_fop_opcode(fop) == M0_CS_DS1_REQ_OPCODE ?
223  m0_fop_reply_alloc(fop, &cs_ds1_rep_fop_fopt) :
224  m0_fop_reply_alloc(fop, &cs_ds2_rep_fop_fopt);
225  if (rfop == NULL) {
226  m0_free(fom);
227  return -ENOMEM;
228  }
229  m0_fom_init(fom, &fop->f_type->ft_fom_type, ops, fop, rfop, reqh);
230 
231  *out = fom;
232  return 0;
233 }

271 /*
272  Transitions fom through its generic phases and also
273  performs corresponding fop specific execution.
274  */
275 static int cs_req_fop_fom_tick(struct m0_fom *fom)
276 {
277  int rc;
278  struct cs_ds1_req_fop *ds1_reqfop;
279  struct cs_ds1_rep_fop *ds1_repfop;
280  struct cs_ds2_req_fop *ds2_reqfop;
281  struct cs_ds2_rep_fop *ds2_repfop;
282  uint64_t opcode;
283 
284  M0_PRE(M0_IN(m0_fop_opcode(fom->fo_fop), (M0_CS_DS1_REQ_OPCODE,
285  M0_CS_DS2_REQ_OPCODE)));
286  if (m0_fom_phase(fom) < M0_FOPH_NR) {
287  rc = m0_fom_tick_generic(fom);
288  } else {
289  opcode = m0_fop_opcode(fom->fo_fop);
290  switch (opcode) {
291  case M0_CS_DS1_REQ_OPCODE:
292  ds1_reqfop = m0_fop_data(fom->fo_fop);
293  ds1_repfop = m0_fop_data(fom->fo_rep_fop);
294  ds1_repfop->csr_rc = ds1_reqfop->csr_value;
295  m0_fom_phase_set(fom, M0_FOPH_SUCCESS);
296  rc = M0_FSO_AGAIN;
297  break;
298  case M0_CS_DS2_REQ_OPCODE:
299  ds2_reqfop = m0_fop_data(fom->fo_fop);
300  ds2_repfop = m0_fop_data(fom->fo_rep_fop);
301  ds2_repfop->csr_rc = ds2_reqfop->csr_value;
302  m0_fom_phase_set(fom, M0_FOPH_SUCCESS);
303  rc = M0_FSO_AGAIN;
304  break;
305  default:
306  M0_ASSERT("Invalid fop" == 0);
307  rc = 0;
308  break;
309  }
310  }
311  if (M0_FI_ENABLED("inject_delay"))
312  m0_nanosleep(m0_time(M0_RPC_ITEM_RESEND_INTERVAL +1, 0), NULL);
313 
314  return rc;
315 }
