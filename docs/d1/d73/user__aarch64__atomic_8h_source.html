94  " prfm pstl1strm, %2\n" \
95  "1:ldxr %0, %2\n" \

97  " stxr %w1, %0, %2\n" \
98  " cbnz %w1, 1b" \
99  : "=&r" (result), "=&r" (tmp), "+Q" (a->a_value) \
