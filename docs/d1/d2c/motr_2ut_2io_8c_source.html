623 static bool ut_helper_sort_is_sorted(struct m0_indexvec *ivec)
624 {
625  uint32_t i;
626  m0_bindex_t prev_ind = 0;
627 
628  for (i = 0; i < SEG_NR(ivec); ++i) {
629  if (prev_ind > INDEX(ivec, i))

631  prev_ind = INDEX(ivec, i);
