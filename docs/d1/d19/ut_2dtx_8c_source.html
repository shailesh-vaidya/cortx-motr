155 static void init0(void)
156 {
157  int i;
158  int j;
159  struct m0_tl uu;

162  M0_SET0(&update_remote);
163  M0_SET0(&control_local);

166  M0_SET0(&oper_remote);
167  M0_SET0(&history0);
168  history_local = history0;
169  for (i = 0; i < REM_NR; ++i)

171  m0_dtm_init(&dtm_local, &M0_UINT128(0, 0));
172  m0_dtm_history_type_register(&dtm_local, &htype);
173  m0_dtm_history_type_register(&dtm_local, &m0_dtm_dtx_htype);
174  m0_dtm_history_type_register(&dtm_local, &m0_dtm_fol_remote_htype);
175  for (i = 0; i < ARRAY_SIZE(dtm_remote); ++i) {
176  m0_dtm_init(&dtm_remote[i], &M0_UINT128(1, i + 1));
177  m0_dtm_history_type_register(&dtm_remote[i], &htype);
178  m0_dtm_history_type_register(&dtm_remote[i],
179  &m0_dtm_dtx_srv_htype);
180  m0_dtm_history_type_register(&dtm_remote[i],
181  &m0_dtm_fol_remote_htype);
182  m0_dtm_local_remote_init(&remote_local[i],
183  &dtm_remote[i].d_id, &dtm_local, NULL);
184  m0_dtm_local_remote_init(&remote_remote[i],
185  &dtm_local.d_id, &dtm_remote[i], NULL);
186  }
187  for (i = 0; i < UPDATE_NR; ++i) {
188  struct m0_dtm_history *history = &history_local[i];
189 
190  m0_dtm_history_init(history, &dtm_local);
191  history->h_hi.hi_ver = 1;
192  history->h_hi.hi_flags |= M0_DHF_OWNED;
193  history->h_ops = &hops;
194  history->h_rem =
195  &remote_local[i % ARRAY_SIZE(remote_local)].lre_rem;
196  }
197  for (i = 0; i < OPER_NR; ++i) {
198  m0_dtm_update_list_init(&uu);
199  m0_dtm_update_link(&uu, control_local[i], 2*DTM_NR);
200  m0_dtm_oper_init(&oper_local[i], &dtm_local, &uu);
201  oper_local[i].oprt_op.op_ops = &op_ops;
202  }
203  for (i = 0; i < REM_NR; ++i) {
204  m0_dtm_fol_init(&fol_remote[i], &dtm_remote[i]);
205  for (j = 0; j < OPER_NR; ++j) {
206  ode[i][j].od_updates.ou_update = udescr[i][j];
207  ode_reply[i][j].od_updates.ou_update = ureply[i][j];
208  m0_dtm_update_list_init(&uu);
209  m0_dtm_update_link(&uu, control_remote[i][j], 1);
210  m0_dtm_oper_init(&oper_remote[i][j],
211  &dtm_remote[i], &uu);
212  oper_remote[i][j].oprt_op.op_ops = &op_ops;
213  }
214  }
215 }
