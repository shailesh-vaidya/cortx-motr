123 static int flock_ut_fom_tick(struct m0_fom *fom, void *data, int *phase)
124 {
125  int rc;
126  struct m0_sns_cm_file_ctx *fctx;
127  struct m0_fid *fid;
128 
129  fid = &gfid;
130 
131  m0_cm_lock(cm);
132  m0_mutex_lock(&scm->sc_file_ctx_mutex);
133  switch (*phase) {
134  case M0_FOM_PHASE_INIT:
135  *phase = FILE_LOCK;
136  rc = M0_FSO_AGAIN;
137  break;
138  case FILE_LOCK:
139  rc = m0_sns_cm_file_lock(scm, fid, &fctx);
140  M0_UT_ASSERT(rc == 0 || rc == -EAGAIN);
141  if (rc == -EAGAIN)
142  rc = m0_sns_cm_file_lock_wait(fctx, fom);
143  if (rc == -EAGAIN) {
144  *phase = FILE_LOCK_WAIT;
145  rc = M0_FSO_WAIT;
146  }
147  if (rc == 0) {
148  *phase = FILE_LOCKED;
149  rc = M0_FSO_AGAIN;
150  }
151  break;
152  case FILE_LOCK_WAIT:
153  fctx = m0_scmfctx_htable_lookup(&scm->sc_file_ctx, fid);
154  M0_UT_ASSERT(fctx != NULL);
155  rc = m0_sns_cm_file_lock_wait(fctx, fom);
156  M0_UT_ASSERT(rc == 0 || rc == -EAGAIN);
157  if (rc == -EAGAIN)
158  rc = M0_FSO_WAIT;
159  if (rc == 0) {
160  *phase = FILE_LOCKED;
161  rc = M0_FSO_AGAIN;
162  }
163  break;
164  case FILE_LOCKED:
165  fctx = m0_scmfctx_htable_lookup(&scm->sc_file_ctx, fid);
166  file_lock_verify(scm, &gfid, NR);
167  M0_UT_ASSERT(m0_sns_cm_fctx_state_get(fctx) == M0_SCFS_LOCKED);
168  m0_fom_wait_on(fom, &fctx->sf_sm.sm_chan, &fom->fo_cb);
169  m0_sns_cm_file_unlock(scm, &fctx->sf_fid);
170  *phase = FILE_UNLOCK_WAIT;
171  rc = M0_FSO_WAIT;
172  break;
173  case FILE_UNLOCK_WAIT:
174  M0_UT_ASSERT(m0_scmfctx_htable_lookup(&scm->sc_file_ctx, fid) == NULL);
175  *phase = M0_FOM_PHASE_FINISH;
176  rc = M0_FSO_WAIT;
177  break;
178  default:
179  rc = -1;
180  break;
181  }
182 
183  m0_cm_unlock(cm);
184  m0_mutex_unlock(&scm->sc_file_ctx_mutex);
185 
186  return rc;
187 }

274 static void file_lock_wait(struct m0_sns_cm_file_ctx *fctx,
275  struct m0_clink *clink)
276 {
277  struct m0_chan *chan;
278  enum m0_rm_incoming_state state;
279 
280  chan = &fctx->sf_rin.rin_sm.sm_chan;
281  m0_rm_owner_lock(&fctx->sf_owner);
282  state = fctx->sf_rin.rin_sm.sm_state;
283  m0_clink_add(chan, clink);
284  m0_rm_owner_unlock(&fctx->sf_owner);
285  /* Check if the lock is already acquired before waiting. */
286  while (!M0_IN(state, (RI_SUCCESS, RI_FAILURE))) {
287  m0_chan_timedwait(clink, m0_time_from_now(1, 0));
288  state = fctx->sf_rin.rin_sm.sm_state;
289  }
290 
291  M0_UT_ASSERT(state == RI_SUCCESS);
292  m0_sm_group_lock(fctx->sf_sm.sm_grp);
293  m0_sm_state_set(&fctx->sf_sm, M0_SCFS_LOCKED);
294  m0_sm_group_unlock(fctx->sf_sm.sm_grp);
295  m0_rm_owner_lock(&fctx->sf_owner);
296  m0_clink_del(clink);
297  m0_rm_owner_unlock(&fctx->sf_owner);
298 }

310 static void sns_file_lock_unlock(void)
311 {
312  struct m0_sns_cm_file_ctx *fctx[NR_FIDS];
313  struct m0_clink tc_clink[NR_FIDS];
314  struct m0_fid fid;
315  uint64_t cont = 0;
316  uint64_t key = KEY_START;
317  int i;
318  int rc;
319 
320  m0_mutex_lock(&scm->sc_file_ctx_mutex);
321  for (i = 0; i < NR_FIDS; i++) {
322  fid_index = i;
323  m0_clink_init(&tc_clink[i], NULL);
324  m0_cm_lock(&scm->sc_base);
325  M0_SET0(&fid);
326  m0_fid_set(&fid, cont, key);
327  rc = m0_sns_cm_file_lock(scm, &fid, &fctx[i]);
328  M0_UT_ASSERT(rc == -EAGAIN);
329  m0_cm_unlock(&scm->sc_base);
330  file_lock_wait(fctx[i], &tc_clink[i]);
331  file_lock_verify(scm, &fid, 1);
332  ++key;
333  if (key == KEY_MAX) {
334  cont++;
335  key = KEY_START;
336  }
337  file_unlock_and_wait(fctx[i], &tc_clink[i]);
338  m0_clink_fini(&tc_clink[i]);
339  }
340  m0_mutex_unlock(&scm->sc_file_ctx_mutex);
341 }
