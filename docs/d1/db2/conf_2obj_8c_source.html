313 void m0_conf_obj_type_register(const struct m0_conf_obj_type *otype)
314 {
315  uint8_t id = otype->cot_ftype.ft_id;
316 
317  M0_PRE(IS_IN_ARRAY(id, obj_types));
318  M0_PRE(obj_types[id] == NULL);
319 
320  m0_fid_type_register(&otype->cot_ftype);
321  obj_types[id] = otype;
322 
323  if (otype->cot_xt != NULL) {
324  otype->cot_xc_init();
325  M0_PRE_EX(m0_xcode_type_flags(*otype->cot_xt,
326  M0_XCODE_TYPE_FLAG_DOM_CONF, 0,
327  M0_BITS(M0_XA_ATOM)));
328  /* Onwire representation must start with the header. */
329  M0_PRE((*otype->cot_xt)->xct_child[0].xf_type ==
330  m0_confx_header_xc);
331 
332  m0_xcode_union_add(m0_confx_obj_xc, otype->cot_branch,
333  *otype->cot_xt, id);
334  }
335  M0_POST(m0_forall(i, ARRAY_SIZE(obj_types),
336  ergo(obj_types[i] != NULL,
337  (obj_types[i]->cot_magic == otype->cot_magic) == (i == id))));
338 }

340 void m0_conf_obj_type_unregister(const struct m0_conf_obj_type *otype)
341 {
342  uint8_t id = otype->cot_ftype.ft_id;
343 
344  M0_PRE(IS_IN_ARRAY(id, obj_types));
345  M0_PRE(obj_types[id] == otype);
346  m0_fid_type_unregister(&otype->cot_ftype);
347  obj_types[id] = NULL;
348 }

426 {
427  /* Root cannot be a child, because it is the topmost object. */
428  M0_PRE(m0_conf_obj_type(child) != &M0_CONF_ROOT_TYPE);
